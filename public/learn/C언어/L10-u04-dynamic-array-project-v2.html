malloc, calloc, realloc을 종합 활용하여 자동으로 크기가 늘어나는 동적 배열(ArrayList)을 단계적으로 구현합니다.

  • 동적 배열의 구조(data, size, capacity)를 설계할 수 있습니다.

  • 용량 초과 시 자동 확장하는 로직을 작성할 수 있습니다.

프로젝트 개요

동적 배열 (ArrayList)
정적 배열은 크기가 고정입니다. 동적 배열은 데이터가 추가될 때
자동으로 용량을 늘립니다. Java의 ArrayList, Python의 list와 같은 원리입니다.

버전 1: 기본 — 추가, 출력, 해제
버전 2: 확장 — 삽입, 삭제, 검색, 정렬
버전 3: 완성 — 구조체 배열, 통계, 메뉴 시스템

버전 1: 기본 동적 배열

C
1#include <stdio.h>
2#include <stdlib.h>
3
4typedef struct {
5 int *data; /* 힙에 할당된 배열 */
6 int size; /* 현재 원소 수 */
7 int capacity; /* 현재 용량 */
8} IntArray;
9
10IntArray *createArray(int initCap) {
11 IntArray *arr = (IntArray*)malloc(sizeof(IntArray));
12 if (arr == NULL) return NULL;
13 arr->data = (int*)malloc(initCap * sizeof(int));
14 if (arr->data == NULL) { free(arr); return NULL; }
15 arr->size = 0;
16 arr->capacity = initCap;
17 return arr;
18}
19
20void append(IntArray *arr, int value) {
21 if (arr->size >= arr->capacity) {
22 arr->capacity *= 2;
23 int *temp = (int*)realloc(arr->data, arr->capacity * sizeof(int));
24 if (temp == NULL) { printf("확장 실패\n"); return; }
25 arr->data = temp;
26 printf(" [확장: 용량 %d]\n", arr->capacity);
27 }
28 arr->data[arr->size++] = value;
29}
30
31void printArray(const IntArray *arr) {
32 printf("[%d/%d] ", arr->size, arr->capacity);
33 for (int i = 0; i < arr->size; i++)
34 printf("%d ", arr->data[i]);
35 printf("\n");
36}
37
38void destroyArray(IntArray *arr) {
39 free(arr->data);
40 free(arr);
41}
42
43int main(void) {
44 IntArray *arr = createArray(2);
45 if (arr == NULL) return 1;
46
47 for (int i = 1; i <= 10; i++) {
48 printf("추가: %d ", i * 10);
49 append(arr, i * 10);
50 printArray(arr);
51 }
52
53 destroyArray(arr);
54 return 0;
55}

버전 2: 삽입 + 삭제 + 검색 + 정렬

undefined

C
1#include <stdio.h>
2#include <stdlib.h>
3
4typedef struct {
5 int *data;
6 int size;
7 int capacity;
8} IntArray;
9
10IntArray *create(int cap) {
11 IntArray *a = (IntArray*)malloc(sizeof(IntArray));
12 a->data = (int*)malloc(cap * sizeof(int));
13 a->size = 0; a->capacity = cap;
14 return a;
15}
16
17void ensureCapacity(IntArray *a) {
18 if (a->size >= a->capacity) {
19 a->capacity *= 2;
20 a->data = (int*)realloc(a->data, a->capacity * sizeof(int));
21 }
22}
23
24void append(IntArray *a, int val) {
25 ensureCapacity(a);
26 a->data[a->size++] = val;
27}
28
29void insertAt(IntArray *a, int idx, int val) {
30 if (idx < 0 || idx > a->size) return;
31 ensureCapacity(a);
32 for (int i = a->size; i > idx; i--)
33 a->data[i] = a->data[i-1];
34 a->data[idx] = val;
35 a->size++;
36}
37
38void removeAt(IntArray *a, int idx) {
39 if (idx < 0 || idx >= a->size) return;
40 for (int i = idx; i < a->size - 1; i++)
41 a->data[i] = a->data[i+1];
42 a->size--;
43}
44
45int indexOf(const IntArray *a, int val) {
46 for (int i = 0; i < a->size; i++)
47 if (a->data[i] == val) return i;
48 return -1;
49}
50
51void sortArray(IntArray *a) {
52 for (int i = 0; i < a->size-1; i++)
53 for (int j = i+1; j < a->size; j++)
54 if (a->data[i] > a->data[j]) {
55 int t = a->data[i]; a->data[i] = a->data[j]; a->data[j] = t;
56 }
57}
58
59void print(const IntArray *a) {
60 printf("[%d/%d] ", a->size, a->capacity);
61 for (int i = 0; i < a->size; i++) printf("%d ", a->data[i]);
62 printf("\n");
63}
64
65void destroy(IntArray *a) { free(a->data); free(a); }
66
67int main(void) {
68 IntArray *a = create(4);
69 int vals[] = {50, 30, 80, 10, 60};
70 for (int i = 0; i < 5; i++) append(a, vals[i]);
71 printf("원본: "); print(a);
72
73 insertAt(a, 2, 99);
74 printf("삽입: "); print(a);
75
76 removeAt(a, 0);
77 printf("삭제: "); print(a);
78
79 int idx = indexOf(a, 80);
80 printf("검색: 80은 인덱스 %d\n", idx);
81
82 sortArray(a);
83 printf("정렬: "); print(a);
84
85 destroy(a);
86 return 0;
87}

동적 배열 설계 패턴
구조체 3요소: data(포인터), size(현재 크기), capacity(용량)

생성: malloc(구조체) + malloc/calloc(data 배열)
확장: size >= capacity → capacity *= 2 → realloc
삭제: 뒤 원소 앞으로 이동 → size--
해제: free(data) → free(구조체) (순서 중요!)

이 패턴은 Java ArrayList, C++ vector의 기초입니다.

요약

핵심 정리
1. 동적 배열 구조: data(포인터) + size(현재) + capacity(용량)
2. 생성: malloc(구조체) + calloc(배열)
3. 확장: size >= capacity → capacity *= 2 → realloc
4. 삽입/삭제: 원소 이동 후 size 증감
5. 해제: free(data) → free(구조체) 순서
6. 패턴: Java ArrayList, C++ vector의 C 구현
