모듈형 계산기

malloc/free, #define 매크로, 조건부 컴파일, 헤더 가드, 멀티파일 프로그래밍, static/extern을 종합 활용하여 모듈형 계산기를 단계적으로 완성합니다.

이 유닛에서 배우는 것

  • Layer 10의 모든 개념을 하나의 프로젝트에 종합합니다.
  • 단일 파일 계산기를 멀티파일 구조로 리팩터링할 수 있습니다.
  • 헤더 가드와 static/extern으로 모듈 간 인터페이스를 설계합니다.
  • 조건부 컴파일로 디버그 모드와 기능 확장을 제어합니다.

프로젝트 개요

모듈형 계산기 (Modular Calculator)
동적 메모리, 전처리기, 멀티파일 프로그래밍을 종합 활용합니다.

버전 1: 기본 — 단일 파일, 사칙연산(+, -, *, /)
버전 2: 확장 — 멀티파일(calc.h/calc.c/main.c), 계산 이력, 메모리 기능
버전 3: 완성 — config.h 설정, 공학 함수 모듈, 메뉴 시스템, #ifdef 디버그

19History *createHistory(int cap) {
20 History *h = (History*)malloc(sizeof(History));
21 if (h == NULL) return NULL;
22 h->data = (Record*)malloc(cap * sizeof(Record));
23 if (h->data == NULL) { free(h); return NULL; }
24 h->size = 0;
25 h->capacity = cap;
26 return h;
27}
28
29void addRecord(History *h, double l, char op, double r, double res) {
30 if (h->size >= h->capacity) {
31 h->capacity *= 2;
32 Record *temp = (Record*)realloc(h->data,
33 h->capacity * sizeof(Record));
34 if (temp == NULL) { printf("확장 실패\n"); return; }
35 h->data = temp;
36 }
37 h->data[h->size++] = (Record){l, op, r, res};
38}
39
40void printHistory(const History *h) {
41 if (h->size == 0) { printf("이력이 없습니다.\n"); return; }
42 printf("\n%-4s %10s %2s %10s = %12s\n",
43 "#", "왼쪽", "연산", "오른쪽", "결과");
44 for (int i = 0; i < h->size; i++)
45 printf("%-4d %10.2f %2c %10.2f = %12.2f\n",
46 i + 1, h->data[i].left, h->data[i].op,
47 h->data[i].right, h->data[i].result);
48}
49
50void destroyHistory(History *h) {
51 free(h->data);
52 free(h);
53}
54
55double calculate(double a, char op, double b, int *ok) {
56 *ok = 1;
57 switch (op) {
58 case '+': return a + b;
59 case '-': return a - b;
60 case '*': return a * b;
61 case '/':
62 if (b == 0) { printf("0으로 나눌 수 없습니다.\n"); *ok = 0; return 0; }
63 return a / b;
64 default:
65 printf("알 수 없는 연산자: %c\n", op);
66 *ok = 0; return 0;
67 }
68}

72 if (hist == NULL) { printf("메모리 할당 실패\n"); return 1; }

74 printf("=== 사칙연산 계산기 ===\n");
75 printf("입력 형식: 숫자 연산자 숫자 (예: 10 + 3)\n");

78 char line[MAX_EXPR];
79 while (1) {
80 printf("> ");
81 if (fgets(line, MAX_EXPR, stdin) == NULL) break;
82 if (line[0] == 'q' || line[0] == 'Q') break;
83 if (line[0] == 'h' || line[0] == 'H') {
84 printHistory(hist); continue;
85 }
86
87 double a, b;
88 char op;
89 if (sscanf(line, "%lf %c %lf", &a, &op, &b) != 3) {
90 printf("입력 형식 오류\n"); continue;
91 }
92
93 int ok;
94 double result = calculate(a, op, b, &ok);
95 if (ok) {
96 printf("= %.2f\n", result);
97 addRecord(hist, a, op, b, result);
98 }
99 }
100
101 printf("\n--- 최종 이력 ---\n");
102 printHistory(hist);
103 destroyHistory(hist);
104 printf("메모리 해제 완료\n");
105 return 0;
106}

89 } while (choice != 0);

95#endif
