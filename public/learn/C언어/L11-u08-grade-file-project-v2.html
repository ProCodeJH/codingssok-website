Chapter 08

성적 파일 저장/불러오기

구조체 배열로 관리하는 성적 데이터를 파일에 저장하고 불러오는 프로그램을 구현합니다.

이 유닛에서 배우는 것

  • 구조체 배열을 텍스트 파일로 저장/불러오기할 수 있습니다.
  • 구조체 배열을 바이너리 파일로 저장/불러오기할 수 있습니다.
  • 파일 기반 데이터 영속성을 구현할 수 있습니다.
  • 텍스트/바이너리 저장 방식을 비교하고 선택할 수 있습니다.

프로젝트 개요

성적 파일 저장/불러오기 프로그램
구조체 배열의 성적 데이터를 파일로 저장하고 불러옵니다.

버전 1: 기본 — 텍스트 파일로 성적 저장/불러오기 (fprintf, fscanf)
버전 2: 확장 — 바이너리 파일 + 메뉴 시스템 (fwrite, fread)
버전 3: 완성 — 자동 저장/불러오기 + 검색 + 정렬 + 에러 처리

94void compareFileSize(const char *textFile, const char *binFile) {
95 FILE *ft = fopen(textFile, "r");
96 FILE *fb = fopen(binFile, "rb");
97 if (ft == NULL || fb == NULL) {
98 printf("파일 크기 비교를 위해 먼저 저장하십시오.\n");
99 if (ft) fclose(ft);
100 if (fb) fclose(fb);
101 return;
102 }
103 fseek(ft, 0, SEEK_END);
104 long textSize = ftell(ft);
105 fseek(fb, 0, SEEK_END);
106 long binSize = ftell(fb);
107 fclose(ft);
108 fclose(fb);
109
110 printf("\n=== 파일 크기 비교 ===\n");
111 printf("텍스트 파일(%s): %ld바이트\n", textFile, textSize);
112 printf("바이너리 파일(%s): %ld바이트\n", binFile, binSize);
113 if (textSize > binSize)
114 printf("바이너리가 %ld바이트 더 작습니다.\n", textSize - binSize);
115 else
116 printf("텍스트가 %ld바이트 더 작습니다.\n", binSize - textSize);
117}

28int autoLoad(Student arr[], const char *filename) {
29 FILE *fp = fopen(filename, "rb");
30 if (fp == NULL) {
31 printf("저장 파일이 없습니다. 새로 시작합니다.\n");
32 return 0;
33 }
34
35 FileHeader header;
36 if (fread(&header, sizeof(FileHeader), 1, fp) != 1) {

38 fclose(fp);

40 }
41

44 fclose(fp);

46 }

52 return 0;
53 }
54
55 if (header.count < 0 || header.count > MAX_STUDENTS) {

57 fclose(fp);
58 return 0;
59 }
60
61 int read = fread(arr, sizeof(Student), header.count, fp);
62 fclose(fp);
63
64 if (read != header.count) {
65 printf("데이터 손상: %d명 중 %d명만 읽혔습니다.\n",
66 header.count, read);
67 return read;
68 }

매직 넘버와 파일 헤더
바이너리 파일의 앞부분에 매직 넘버와 버전 정보를 저장하면
파일 형식을 검증할 수 있습니다.

FileHeader 구조:
magic — 파일 형식 식별 (0x47524445 = "GRDE")
version — 파일 버전 (호환성 확인)
count — 저장된 데이터 수

불러올 때 이 값들을 확인하여
잘못된 파일이나 손상된 데이터를 걸러냅니다.
