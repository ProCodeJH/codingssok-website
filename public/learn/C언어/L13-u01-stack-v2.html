배열 기반 스택 구조체와 함수

C
1#include <stdio.h>
2#include <stdlib.h>
3
4#define MAX_SIZE 100
5
6typedef struct {
7 int data[MAX_SIZE];
8 int top;
9} Stack;
10
11// 스택 초기화
12void init(Stack *s) {
13 s->top = -1;
14}
15
16// 스택이 비었는지 확인
17int isEmpty(Stack *s) {
18 return s->top == -1;
19}
20
21// 스택이 가득 찼는지 확인
22int isFull(Stack *s) {
23 return s->top == MAX_SIZE - 1;
24}
25
26// push: 스택에 값 삽입
27void push(Stack *s, int value) {
28 if (isFull(s)) {
29 printf("스택 오버플로!\n");
30 return;
31 }
32 s->data[++(s->top)] = value;
33}
34
35// pop: 스택에서 값 제거 후 반환
36int pop(Stack *s) {
37 if (isEmpty(s)) {
38 printf("스택 언더플로!\n");
39 return -1;
40 }
41 return s->data[(s->top)--];
42}
43
44// peek: 맨 위 값 확인 (제거하지 않음)
45int peek(Stack *s) {
46 if (isEmpty(s)) {
47 printf("스택이 비어 있습니다!\n");
48 return -1;
49 }
50 return s->data[s->top];
51}
52
53int main(void) {
54 Stack s;
55 init(&s);
56
57 push(&s, 10);
58 push(&s, 20);
59 push(&s, 30);
60
61 printf("peek: %d\n", peek(&s)); // 30
62 printf("pop: %d\n", pop(&s)); // 30
63 printf("pop: %d\n", pop(&s)); // 20
64 printf("peek: %d\n", peek(&s)); // 10
65 printf("isEmpty: %d\n", isEmpty(&s)); // 0 (false)
66 printf("pop: %d\n", pop(&s)); // 10
67 printf("isEmpty: %d\n", isEmpty(&s)); // 1 (true)
68
69 return 0;
70}

스택 연산 추적표

오버플로와 언더플로

스택 오버플로(Stack Overflow)
스택이 가득 찬 상태(top == MAX_SIZE - 1)에서 push를 시도하면 발생합니다.
배열 범위를 초과하여 메모리를 덮어쓰므로 반드시 isFull() 검사가 필요합니다.
스택 언더플로(Stack Underflow)
스택이 빈 상태(top == -1)에서 pop을 시도하면 발생합니다.
존재하지 않는 데이터를 참조하므로 반드시 isEmpty() 검사가 필요합니다.

스택 활용 사례

스택이 사용되는 곳
1. 함수 호출 스택 — main() → funcA() → funcB() 호출 시 복귀 주소를 스택에 저장합니다.
2. 실행 취소(Ctrl+Z) — 작업 내역을 스택에 push하고, 취소 시 pop합니다.
3. 괄호 검사 — 여는 괄호를 push, 닫는 괄호를 만나면 pop하여 짝을 확인합니다.
4. 후위 표기식 계산 — 피연산자를 push, 연산자를 만나면 pop하여 계산합니다.

배열 스택 vs 연결리스트 스택 비교

에러 체험

에러 1: top 초기값을 0으로 설정
void init(Stack *s) {
s->top = 0; // 잘못! data[0]에 값이 있다고 오인
}

→ top은 "마지막 데이터의 인덱스"입니다. 빈 스택은 데이터가 없으므로 -1이 올바릅니다.
→ s->top = -1;
에러 2: pop 전에 isEmpty 미확인
int val = pop(&s); // 빈 스택이면 쓰레기값 또는 비정상 종료!

→ pop 전에 반드시 isEmpty()로 확인하거나, pop 내부에서 검사해야 합니다.
→ if (!isEmpty(&s)) { int val = pop(&s); }
에러 3: push 시 배열 크기 초과
// MAX_SIZE가 3인 스택에 4번 push
push(&s, 10); push(&s, 20); push(&s, 30);
push(&s, 40); // data[3] 접근 → 배열 범위 초과!

→ push 전에 isFull()로 검사해야 합니다.
→ if (!isFull(&s)) push(&s, 40);
