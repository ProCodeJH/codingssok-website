스택 계산기 -- 후위 표기

스택을 활용하여 중위 표기식을 후위로 변환하고 계산하는 계산기를 구현합니다.

이 유닛에서 배우는 것

  • 스택을 활용하여 후위 표기식을 계산할 수 있습니다.
  • 중위 표기식을 후위 표기식으로 변환하는 알고리즘을 구현할 수 있습니다.
  • 연산자 우선순위와 괄호 처리를 정확하게 구현할 수 있습니다.
  • 다자릿수, 음수, 실수를 포함한 완전한 수식 계산기를 구현할 수 있습니다.

프로젝트 개요

스택 계산기 (Stack Calculator)
스택 자료구조를 활용하여 수학 수식을 계산하는 프로그램을 구현합니다. 후위 표기법(Reverse Polish Notation)은 괄호 없이 연산 순서를 명확하게 표현하며, 스택 하나로 효율적으로 계산할 수 있습니다.

버전 1: 기본 -- 후위 표기식 계산기 (사용자가 후위식 입력, 스택으로 계산)
버전 2: 확장 -- 중위->후위 변환 + 계산 (괄호 지원, 사칙연산)
버전 3: 완성 -- 완전한 수식 계산기 (다자릿수, 음수, 실수, 에러 처리, 히스토리)

50/* 후위 표기식 계산 */
51int evaluatePostfix(const char *expr) {
52 IntStack stack;
53 initStack(&stack);
54
55 int i = 0;
56 while (expr[i] != '\0') {
57 /* 공백 건너뛰기 */
58 if (expr[i] == ' ') {
59 i++;
60 continue;
61 }
62
63 /* 숫자인 경우: 스택에 push */
64 if (isdigit(expr[i])) {
65 push(&stack, expr[i] - '0');
66 i++;
67 }
68 /* 연산자인 경우: 두 값을 pop하여 계산 */
69 else {
70 int b = pop(&stack); /* 나중에 push된 값이 오른쪽 피연산자 */
71 int a = pop(&stack); /* 먼저 push된 값이 왼쪽 피연산자 */
72 int result = 0;
73
74 switch (expr[i]) {
75 case '+': result = a + b; break;
76 case '-': result = a - b; break;
77 case '*': result = a * b; break;
78 case '/':
79 if (b == 0) {
80 printf("오류: 0으로 나눌 수 없습니다.\n");
81 return 0;
82 }
83 result = a / b;
84 break;
85 default:
86 printf("오류: 알 수 없는 연산자 '%c'\n", expr[i]);
87 return 0;
88 }
89 push(&stack, result);
90 i++;
91 }
92 }
93
94 return pop(&stack);
95}

48/* 중위 -> 후위 변환 (Shunting Yard 알고리즘) */
49int infixToPostfix(const char *infix, char postfix[][16], int maxTokens) {
50 CharStack opStack;
51 charInit(&opStack);
52 int count = 0; /* 후위 토큰 개수 */
53 int i = 0;
54
55 while (infix[i] != '\0') {
56 /* 공백 건너뛰기 */
57 if (infix[i] == ' ') {
58 i++;
59 continue;
60 }
61
62 /* 숫자: 후위 출력에 추가 */
63 if (isdigit(infix[i])) {
64 char num[16];
65 int j = 0;
66 while (isdigit(infix[i])) {
67 num[j++] = infix[i++];
68 }
69 num[j] = '\0';
70 strcpy(postfix[count++], num);
71 }
72 /* 여는 괄호: 연산자 스택에 push */
73 else if (infix[i] == '(') {
74 charPush(&opStack, '(');
75 i++;
76 }
77 /* 닫는 괄호: '('를 만날 때까지 pop */
78 else if (infix[i] == ')') {
79 while (!charIsEmpty(&opStack) &&
80 charPeek(&opStack) != '(') {
81 char op[2] = { charPop(&opStack), '\0' };
82 strcpy(postfix[count++], op);
83 }
84 if (!charIsEmpty(&opStack)) {
85 charPop(&opStack); /* '(' 제거 */
86 }
87 i++;
88 }
89 /* 연산자: 우선순위 비교 후 처리 */
90 else if (isOperator(infix[i])) {
91 while (!charIsEmpty(&opStack) &&
92 charPeek(&opStack) != '(' &&
93 precedence(charPeek(&opStack)) >= precedence(infix[i])) {
94 char op[2] = { charPop(&opStack), '\0' };
95 strcpy(postfix[count++], op);
96 }
97 charPush(&opStack, infix[i]);
98 i++;
99 }
100 else {
101 i++; /* 알 수 없는 문자 건너뛰기 */
102 }
103 }
104
105 /* 스택에 남은 연산자 모두 출력 */
106 while (!charIsEmpty(&opStack)) {
107 char op[2] = { charPop(&opStack), '\0' };
108 strcpy(postfix[count++], op);
109 }
110
111 return count;
112}

114/* 후위 표기식 계산 (토큰 배열 버전) */
115int evaluatePostfix(char tokens[][16], int tokenCount) {
116 IntStack stack;
117 intInit(&stack);
118
119 for (int i = 0; i < tokenCount; i++) {
120 /* 숫자 토큰 */
121 if (isdigit(tokens[i][0])) {
122 intPush(&stack, atoi(tokens[i]));
123 }
124 /* 연산자 토큰 */
125 else if (isOperator(tokens[i][0]) && tokens[i][1] == '\0') {
126 int b = intPop(&stack);
127 int a = intPop(&stack);
128 int result = 0;
129
130 switch (tokens[i][0]) {
131 case '+': result = a + b; break;
132 case '-': result = a - b; break;
133 case '*': result = a * b; break;
134 case '/':
135 if (b == 0) {
136 printf("오류: 0으로 나눌 수 없습니다.\n");
137 return 0;
138 }
139 result = a / b;
140 break;
141 }
142 intPush(&stack, result);
143 }
144 }
145
146 return intPop(&stack);
147}

94/* 단항 마이너스인지 판별 */
95int isUnaryMinus(const char *expr, int pos) {
96 if (expr[pos] != '-') return 0;
97 if (pos == 0) return 1;
98
99 /* 바로 앞 유효 문자 찾기 */
100 int k = pos - 1;
101 while (k >= 0 && expr[k] == ' ') k--;
102 if (k < 0) return 1;
103 if (isOperator(expr[k]) || expr[k] == '(') return 1;
104 return 0;
105}

121 i++;

128 int j = 0;

135 list->tokens[list->count][j] = '\0';
136 list->count++;
137 }

139 else if (isOperator(expr[i]) || expr[i] == '(' || expr[i] == ')') {
140 list->tokens[list->count][0] = expr[i];
141 list->tokens[list->count][1] = '\0';
142 list->count++;
143 i++;
144 }
145 else {
146 printf("오류: 잘못된 문자 '%c' (위치 %d)\n", expr[i], i);

155int infixToPostfix(TokenList *infix, TokenList *postfix) {
156 CharStack opStack;

159
160 for (int i = 0; i < infix->count; i++) {

162
163 /* 숫자 토큰 (음수 포함: 첫 문자가 숫자이거나 '-' 뒤 숫자) */
164 if (isdigit(token[0]) || token[0] == '.' ||
165 (token[0] == '-' && token[1] != '\0')) {
166 strcpy(postfix->tokens[postfix->count++], token);
167 }
168 /* 여는 괄호 */
169 else if (token[0] == '(') {
170 charPush(&opStack, '(');
171 }
172 /* 닫는 괄호 */
173 else if (token[0] == ')') {
174 while (!charIsEmpty(&opStack) &&
175 charPeek(&opStack) != '(') {
176 char op[2] = { charPop(&opStack), '\0' };
177 strcpy(postfix->tokens[postfix->count++], op);
178 }
179 if (charIsEmpty(&opStack)) {
180 printf("오류: 괄호가 일치하지 않습니다.\n");
181 return 0;
182 }
183 charPop(&opStack); /* '(' 제거 */
184 }
185 /* 연산자 */
186 else if (isOperator(token[0]) && token[1] == '\0') {
187 while (!charIsEmpty(&opStack) &&
188 charPeek(&opStack) != '(' &&
189 precedence(charPeek(&opStack)) >= precedence(token[0])) {
190 char op[2] = { charPop(&opStack), '\0' };
191 strcpy(postfix->tokens[postfix->count++], op);
192 }
193 charPush(&opStack, token[0]);
194 }
195 else {
196 printf("오류: 잘못된 토큰 '%s'\n", token);
197 return 0;
198 }
199 }
200
201 while (!charIsEmpty(&opStack)) {
202 if (charPeek(&opStack) == '(') {
203 printf("오류: 괄호가 일치하지 않습니다.\n");
204 return 0;
205 }
206 char op[2] = { charPop(&opStack), '\0' };
207 strcpy(postfix->tokens[postfix->count++], op);

213/* ===== 후위 표기식 계산 (실수) ===== */
214
215int evaluatePostfix(TokenList *postfix, double *result) {
216 DoubleStack stack;
217 dblInit(&stack);
218
219 for (int i = 0; i < postfix->count; i++) {
220 char *token = postfix->tokens[i];
221
222 /* 숫자 토큰 */
223 if (isdigit(token[0]) || token[0] == '.' ||
224 (token[0] == '-' && token[1] != '\0')) {
225 dblPush(&stack, atof(token));
226 }
227 /* 연산자 토큰 */
228 else if (isOperator(token[0]) && token[1] == '\0') {
229 if (dblSize(&stack) < 2) {
230 printf("오류: 피연산자가 부족합니다.\n");
231 return 0;
232 }
233 double b = dblPop(&stack);
234 double a = dblPop(&stack);
235 double r = 0.0;
236
237 switch (token[0]) {
238 case '+': r = a + b; break;
239 case '-': r = a - b; break;
240 case '*': r = a * b; break;
241 case '/':
242 if (fabs(b) < 1e-12) {
243 printf("오류: 0으로 나눌 수 없습니다.\n");
244 return 0;
245 }
246 r = a / b;
247 break;
248 }
249 dblPush(&stack, r);
250 }
251 }
252
253 if (dblSize(&stack) != 1) {
254 printf("오류: 수식이 올바르지 않습니다.\n");
255 return 0;
256 }
257
258 *result = dblPop(&stack);
259 return 1;
260}

293

295

297 if (!tokenize(expression, &infix)) break;
298

303 printPostfix(&postfix);
304

306 double result;
307 if (evaluatePostfix(&postfix, &result)) {
308 printf("결과: %g\n", result);
309 addHistory(expression, result);
310 }
311 break;
312
313 case 2:
314 showHistory();
315 break;
316
317 case 0:
318 printf("종료합니다.\n");
319 break;
320
321 default:
322 printf("잘못된 선택입니다.\n");
323 break;
324 }
325 } while (choice != 0);

요약

핵심 정리
1. 후위 표기법: 연산자가 피연산자 뒤에 위치, 괄호와 우선순위 불필요
2. 후위식 계산: 숫자는 push, 연산자는 pop 2회 -> 계산 -> push
3. pop 순서 주의: 나중에 push된 값이 오른쪽 피연산자 (b), 먼저 push된 값이 왼쪽 (a)
4. Shunting Yard: 연산자 스택으로 우선순위 비교하며 중위 -> 후위 변환
5. 괄호 처리: '(' push, ')' 만나면 '('까지 모두 pop하여 출력
6. 우선순위: *, / (2) > +, - (1), 같은 우선순위는 왼쪽 결합 (>=)
7. 토크나이저: 다자릿수, 실수, 단항 마이너스를 분리하여 토큰 배열 생성
8. 에러 처리: 괄호 불일치, 0 나누기, 피연산자 부족, 잘못된 문자 감지
9. 타입별 스택: IntStack (정수), DoubleStack (실수), CharStack (연산자)
10. 히스토리: 원형 버퍼로 최근 계산 기록을 관리하여 조회 가능
