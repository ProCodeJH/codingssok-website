알고리즘 선택 흐름도
문제 분석 순서:

1. 입력 크기(n)를 확인합니다
- n <= 10: 완전탐색(순열) 가능
- n <= 20: 완전탐색(부분집합) 가능
- n <= 10,000: O(n^2) 가능
- n <= 1,000,000: O(n log n) 필요

2. 문제 유형을 판별합니다
- '최솟값/최댓값을 구하라' -> DP 또는 그리디
- '모든 경우를 나열하라' -> 브루트포스 또는 백트래킹
- '연결/경로를 찾아라' -> 그래프 (DFS/BFS)
- '패턴을 찾아라' -> 문자열 알고리즘

3. 세부 조건을 확인합니다
- 그리디 적용 가능? (탐욕적 선택이 전체 최적인가?)
- DP 적용 가능? (점화식을 세울 수 있는가?)
- 가지치기 가능? (백트래킹으로 탐색 공간 축소)

13 printf("=== 문제 1: 두 수의 합 (브루트포스 O(n^2)) ===\n");

23 printf(" arr[%d] + arr[%d] = %d + %d = %d\n",

41 int totalCount = 0;

62/* ===== 문제 3: 구간 합 최대 (브루트포스 vs 카데인) ===== */
63/* 연속 부분 배열의 최대 합을 구합니다.
64 * 브루트포스 O(n^2)와 DP(카데인) O(n)을 비교합니다. */
65
66void maxSubarrayBrute(int arr[], int n) {
67 printf("=== 문제 3a: 최대 구간 합 (브루트포스 O(n^2)) ===\n");
68
69 int maxSum = arr[0];
70 int bestL = 0, bestR = 0;
71
72 for (int i = 0; i < n; i++) {
73 int sum = 0;
74 for (int j = i; j < n; j++) {
75 sum += arr[j];
76 if (sum > maxSum) {
77 maxSum = sum;
78 bestL = i;
79 bestR = j;
80 }
81 }
82 }
83
84 printf(" 최대 구간 합: %d\n", maxSum);
85 printf(" 구간: arr[%d..%d] = ", bestL, bestR);
86 for (int i = bestL; i <= bestR; i++)
87 printf("%d ", arr[i]);
88 printf("\n\n");
89}

91void maxSubarrayKadane(int arr[], int n) {
92 printf("=== 문제 3b: 최대 구간 합 (카데인 DP O(n)) ===\n");
93
94 int maxSum = arr[0];
95 int curSum = arr[0];
96 int bestL = 0, bestR = 0;
97 int tempL = 0;
98
99 for (int i = 1; i < n; i++) {
100 if (curSum + arr[i] > arr[i]) {
101 curSum = curSum + arr[i];
102 } else {
103 curSum = arr[i];
104 tempL = i;
105 }
106 if (curSum > maxSum) {
107 maxSum = curSum;
108 bestL = tempL;
109 bestR = i;
110 }
111 }
112
113 printf(" 최대 구간 합: %d\n", maxSum);
114 printf(" 구간: arr[%d..%d] = ", bestL, bestR);
115 for (int i = bestL; i <= bestR; i++)

1#include <stdio.h>
2#include <string.h>
3
4/* ============================================================
5 * KOI 기출 도전 v2: DP + 그래프 기출 유형
6 * ============================================================ */
7
8#define MAX_N 101
9#define MAX_MAZE 20
10#define QUEUE_SIZE 1000
11
12/* ===== 문제 1: 계단 오르기 (1차원 DP) ===== */
13/* n개의 계단이 있을 때, 1칸 또는 2칸씩 올라갈 수 있습니다.
14 * 꼭대기까지 올라가는 경우의 수를 구합니다.
15 * 점화식: dp[i] = dp[i-1] + dp[i-2] */
16
17void stairClimb(int n) {
18 int dp[MAX_N];
19
20 printf("=== 문제 1: 계단 오르기 (1차원 DP) ===\n");
21 printf("계단 수: %d\n\n", n);
22
23 dp[0] = 1; /* 바닥에 서 있는 경우 */
24 dp[1] = 1; /* 1칸 올라가는 방법 1가지 */
25
26 for (int i = 2; i <= n; i++) {
27 dp[i] = dp[i - 1] + dp[i - 2];
28 }
29
30 printf(" DP 테이블:\n ");
31 for (int i = 0; i <= n; i++)
32 printf("dp[%d]=%d ", i, dp[i]);
33 printf("\n\n");
34
35 printf(" 점화식: dp[i] = dp[i-1] + dp[i-2]\n");
36 printf(" %d칸 계단을 오르는 경우의 수: %d\n\n", n, dp[n]);
37}
38
39/* ===== 문제 2: 격자 경로 (2차원 DP) ===== */
40/* (0,0)에서 (m-1,n-1)까지 오른쪽 또는 아래로만 이동할 때
41 * 경로의 수를 구합니다.
42 * 점화식: dp[i][j] = dp[i-1][j] + dp[i][j-1] */
43
44void gridPaths(int rows, int cols) {
45 int dp[MAX_N][MAX_N];
46
47 printf("=== 문제 2: 격자 경로 (2차원 DP) ===\n");
48 printf("격자 크기: %d x %d\n\n", rows, cols);
49
50 /* 첫 행과 첫 열은 모두 1 (한 방향으로만 이동) */
51 for (int i = 0; i < rows; i++) dp[i][0] = 1;
52 for (int j = 0; j < cols; j++) dp[0][j] = 1;
53
54 /* 나머지 칸 채우기 */
55 for (int i = 1; i < rows; i++) {
56 for (int j = 1; j < cols; j++) {
57 dp[i][j] = dp[i - 1][j] + dp[i][j - 1];
58 }
59 }
60
61 /* DP 테이블 출력 */
62 printf(" DP 테이블:\n");
63 for (int i = 0; i < rows; i++) {
64 printf(" ");
65 for (int j = 0; j < cols; j++) {
66 printf("%4d", dp[i][j]);
67 }
68 printf("\n");
69 }
70
71 printf("\n 점화식: dp[i][j] = dp[i-1][j] + dp[i][j-1]\n");
72 printf(" (0,0) -> (%d,%d) 경로 수: %d\n\n",
73 rows - 1, cols - 1, dp[rows - 1][cols - 1]);
74}
75
76/* ===== 문제 3: 미로 탐색 (BFS) ===== */
77/* 미로에서 (0,0)부터 (N-1,M-1)까지의 최단 거리를 구합니다.
78 * BFS는 가중치가 동일한 그래프에서 최단 경로를 보장합니다. */
79
80typedef struct {
81 int r, c;
82} Point;
83
84void mazeBFS(void) {
85 /* 1: 이동 가능, 0: 벽 */
86 int maze[MAX_MAZE][MAX_MAZE] = {
87 {1, 0, 1, 1, 1, 1},
88 {1, 0, 1, 0, 0, 1},
89 {1, 1, 1, 0, 1, 1},
90 {0, 0, 1, 1, 1, 0},
91 {1, 1, 1, 0, 1, 1}
92 };
93 int rows = 5, cols = 6;
94 int dist[MAX_MAZE][MAX_MAZE];
95
96 /* 방향 배열: 상, 하, 좌, 우 */
97 int dr[] = {-1, 1, 0, 0};
98 int dc[] = {0, 0, -1, 1};
99
100 /* 초기화 */
101 memset(dist, -1, sizeof(dist));
102
103 /* BFS 큐 */
104 Point queue[QUEUE_SIZE];
105 int front = 0, rear = 0;
106
107 /* 시작점 */
108 dist[0][0] = 1;
109 queue[rear].r = 0;
110 queue[rear].c = 0;
111 rear++;
112
113 printf("=== 문제 3: 미로 탐색 (BFS) ===\n\n");
114
115 /* 미로 출력 */
116 printf(" 미로:\n");
117 for (int i = 0; i < rows; i++) {
118 printf(" ");
119 for (int j = 0; j < cols; j++)
120 printf("%d ", maze[i][j]);
121 printf("\n");
122 }
123 printf("\n");
124
125 /* BFS 탐색 */
126 while (front < rear) {
127 Point cur = queue[front++];
128
129 for (int d = 0; d < 4; d++) {
130 int nr = cur.r + dr[d];
131 int nc = cur.c + dc[d];
132
133 /* 범위 검사 */
134 if (nr < 0 || nr >= rows || nc < 0 || nc >= cols)
135 continue;
136 /* 벽이거나 이미 방문 */
137 if (maze[nr][nc] == 0 || dist[nr][nc] != -1)
138 continue;
139
140 dist[nr][nc] = dist[cur.r][cur.c] + 1;
141 queue[rear].r = nr;
142 queue[rear].c = nc;
143 rear++;
144 }
145 }
146
147 /* 거리 테이블 출력 */
148 printf(" 거리 테이블:\n");
149 for (int i = 0; i < rows; i++) {
150 printf(" ");
151 for (int j = 0; j < cols; j++) {
152 if (dist[i][j] == -1)
153 printf(" .");
154 else
155 printf("%3d", dist[i][j]);
156 }
157 printf("\n");
158 }
159
160 int result = dist[rows - 1][cols - 1];
161 if (result == -1)
162 printf("\n 도달 불가능\n");
163 else
164 printf("\n 최단 거리: %d칸\n", result);
165 printf("\n");
166}
167
168/* ===== 문제 4: 연결 요소 개수 (DFS) ===== */
169/* 그래프에서 연결 요소(Connected Component)의 개수를 구합니다.
170 * DFS로 하나의 연결 요소를 탐색한 뒤, 미방문 노드에서 다시 DFS를 시작합니다. */
171
172#define MAX_V 10
173
174int adj[MAX_V][MAX_V]; /* 인접 행렬 */
175int visited[MAX_V];
176int numVertices;
177
178void dfs(int v) {
179 visited[v] = 1;
180 for (int u = 0; u < numVertices; u++) {
181 if (adj[v][u] && !visited[u]) {
182 dfs(u);
183 }
184 }
185}
186
187void connectedComponents(void) {
188 /* 그래프 설정: 7개 정점, 3개 연결 요소 */
189 numVertices = 7;
190 memset(adj, 0, sizeof(adj));
191 memset(visited, 0, sizeof(visited));
192
193 /* 연결 요소 1: {0, 1, 2} */
194 adj[0][1] = adj[1][0] = 1;
195 adj[1][2] = adj[2][1] = 1;
196
197 /* 연결 요소 2: {3, 4} */
198 adj[3][4] = adj[4][3] = 1;
199
200 /* 연결 요소 3: {5, 6} */
201 adj[5][6] = adj[6][5] = 1;
202
203 printf("=== 문제 4: 연결 요소 개수 (DFS) ===\n\n");
204 printf(" 정점 수: %d\n", numVertices);
205 printf(" 간선: (0-1), (1-2), (3-4), (5-6)\n\n");
206
207 int components = 0;
208 for (int v = 0; v < numVertices; v++) {
209 if (!visited[v]) {
210 components++;
211 printf(" 연결 요소 %d: { ", components);
212
213 /* DFS 전에 현재 방문 상태를 저장하여 어떤 노드가 추가되는지 확인 */
214 int before[MAX_V];
215 memcpy(before, visited, sizeof(visited));
216
217 dfs(v);
218
219 for (int u = 0; u < numVertices; u++) {
220 if (!before[u] && visited[u])
221 printf("%d ", u);
222 }
223 printf("}\n");
224 }
225 }
226
227 printf("\n 연결 요소 개수: %d\n\n", components);
228}
229

231 printf("========================================\n");
232 printf(" KOI 기출 도전 v2: DP + 그래프 풀이\n");
233 printf("========================================\n\n");
234
235 /* 1차원 DP: 계단 오르기 */

248}

12/* ===== 문제 1: LIS (최장 증가 부분 수열) -- 3가지 풀이 비교 ===== */
13
14/* 풀이 1: 브루트포스 O(2^n) -- n이 작을 때만 가능 */
15int lisBrute_helper(int arr[], int n, int idx, int prev) {
16 if (idx == n) return 0;
17
18 /* 현재 원소를 포함하지 않는 경우 */
19 int skip = lisBrute_helper(arr, n, idx + 1, prev);
20
21 /* 현재 원소를 포함하는 경우 (이전보다 클 때만) */
22 int take = 0;
23 if (prev == -1 || arr[idx] > arr[prev]) {
24 take = 1 + lisBrute_helper(arr, n, idx + 1, idx);
25 }
26
27 return (skip > take) ? skip : take;
28}
29
30int lisBrute(int arr[], int n) {
31 return lisBrute_helper(arr, n, 0, -1);
32}
33
34/* 풀이 2: DP O(n^2) -- 실용적 풀이 */
35int lisDP(int arr[], int n) {
36 int dp[MAX_N];
37 int maxLen = 1;
38
39 for (int i = 0; i < n; i++) {
40 dp[i] = 1; /* 자기 자신만 포함 */
41 for (int j = 0; j < i; j++) {
42 if (arr[j] < arr[i] && dp[j] + 1 > dp[i]) {
43 dp[i] = dp[j] + 1;
44 }
45 }
46 if (dp[i] > maxLen)
47 maxLen = dp[i];
48 }
49
50 return maxLen;
51}
52
53/* 풀이 3: 이분 탐색 O(n log n) -- 최적 풀이 */
54int tails[MAX_N]; /* tails[i]: 길이 i+1인 LIS의 마지막 원소 최솟값 */
55
56int lowerBound(int arr[], int len, int target) {
57 int lo = 0, hi = len;
58 while (lo < hi) {
59 int mid = (lo + hi) / 2;
60 if (arr[mid] < target)
61 lo = mid + 1;
62 else
63 hi = mid;
64 }
65 return lo;
66}
67
68int lisBinary(int arr[], int n) {
69 int length = 0;
70
71 for (int i = 0; i < n; i++) {
72 int pos = lowerBound(tails, length, arr[i]);
73 tails[pos] = arr[i];
74 if (pos == length)
75 length++;
76 }
77
78 return length;
79}

87 for (int i = 0; i < n; i++)
88 printf("%d ", arr[i]);

121typedef struct {

127void knapsack(void) {
128 Item items[] = {
129 {"보석", 2, 40},

131 {"은화", 4, 60},
132 {"동전", 5, 70},
133 {"왕관", 1, 10}
134 };
135 int n = sizeof(items) / sizeof(items[0]);
136 int capacity = 7;
137 int dp[6][MAX_N]; /* dp[i][w]: i번째 아이템까지 고려, 용량 w일 때 최대 가치 */
138
139 printf("=== 문제 2: 0/1 배낭 문제 (DP) ===\n");
140 printf("배낭 용량: %d\n\n", capacity);
141
142 printf(" %-8s %6s %6s\n", "아이템", "무게", "가치");
143 printf(" --------------------------\n");
144 for (int i = 0; i < n; i++) {
145 printf(" %-8s %6d %6d\n",
146 items[i].name, items[i].weight, items[i].value);
147 }
148 printf("\n");
149
150 /* DP 테이블 초기화 */
151 memset(dp, 0, sizeof(dp));
152
153 for (int i = 1; i <= n; i++) {
154 for (int w = 0; w <= capacity; w++) {
155 /* 현재 아이템을 넣지 않는 경우 */
156 dp[i][w] = dp[i - 1][w];
157
158 /* 현재 아이템을 넣는 경우 */
159 if (items[i - 1].weight <= w) {
160 int withItem = dp[i - 1][w - items[i - 1].weight]
161 + items[i - 1].value;
162 if (withItem > dp[i][w])
163 dp[i][w] = withItem;
164 }
165 }
166 }
167
168 /* DP 테이블 출력 */
169 printf(" DP 테이블 (행: 아이템, 열: 용량):\n");
170 printf(" %8s", "");
171 for (int w = 0; w <= capacity; w++)
172 printf("%4d", w);
173 printf("\n");
174
175 for (int i = 0; i <= n; i++) {
176 if (i == 0)
177 printf(" %-8s", "(없음)");
178 else
179 printf(" %-8s", items[i - 1].name);
180
181 for (int w = 0; w <= capacity; w++)
182 printf("%4d", dp[i][w]);
183 printf("\n");
184 }
185
186 printf("\n 최대 가치: %d\n", dp[n][capacity]);
187
188 /* 선택된 아이템 역추적 */
189 printf(" 선택된 아이템: ");
190 int w = capacity;
191 for (int i = n; i >= 1; i--) {
192 if (dp[i][w] != dp[i - 1][w]) {
193 printf("%s(%d/%d) ",
194 items[i - 1].name,
195 items[i - 1].weight,
196 items[i - 1].value);
197 w -= items[i - 1].weight;
198 }
199 }
200 printf("\n\n");
201}

핵심 정리
1. 알고리즘 분류: 입력 크기(n)와 문제 특성으로 적합한 알고리즘을 판별
2. 브루트포스: n이 작을 때(<=20) 모든 경우를 검사, 정답 보장
3. 그리디: 매 단계 최선 선택이 전체 최적일 때 사용 (동전, 회의실)
4. DP: 부분 문제 중복 + 최적 부분 구조 (점화식으로 테이블 채움)
5. 1차원 DP: dp[i] = f(dp[i-1], dp[i-2], ...) -- 계단, 피보나치
6. 2차원 DP: dp[i][j] = f(dp[i-1][j], dp[i][j-1]) -- 격자, 배낭
7. BFS: 가중치 동일 그래프의 최단 경로 보장 (큐 사용)
8. DFS: 연결 요소, 백트래킹에 활용 (스택/재귀 사용)
9. 시간 복잡도 역산: n=10^6이면 O(n log n), n=10^4이면 O(n^2) 가능
10. 최적화 전략: 브루트포스 -> 메모이제이션 -> DP -> 이분탐색 순서로 개선
