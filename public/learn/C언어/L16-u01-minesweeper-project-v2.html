150 qRear++;

153 int r = bfsQueue[qFront].r;
154 int c = bfsQueue[qFront].c;
155 qFront++;
156
157 if (count[r][c] == 0) {
158 for (int d = 0; d < 8; d++) {
159 int nr = r + dr[d];
160 int nc = c + dc[d];
161 if (nr >= 0 && nr < gRows &&
162 nc >= 0 && nc < gCols &&
163 !revealed[nr][nc] &&
164 !flagged[nr][nc] &&
165 mine[nr][nc] == 0) {
166 revealed[nr][nc] = 1;
167 bfsQueue[qRear].r = nr;
168 bfsQueue[qRear].c = nc;
169 qRear++;
170 }
