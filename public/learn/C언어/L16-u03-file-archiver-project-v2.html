Chapter 03

파일 아카이버 (tar 구현)

여러 파일을 하나로 묶는 아카이브 프로그램을 단계적으로 구현하며, RLE 압축과 CRC 무결성 검증까지 완성합니다.

이 유닛에서 배우는 것

  • 파일 이름과 크기 헤더를 설계하고 여러 파일을 하나의 아카이브로 묶고 풀 수 있습니다.
  • 아카이브 내 파일 목록 조회, 개별 추출, 삭제 기능을 구현할 수 있습니다.
  • RLE 압축/해제를 아카이브에 적용하여 저장 공간을 절약할 수 있습니다.
  • CRC 체크섬으로 아카이브 파일의 무결성을 검증할 수 있습니다.

fread와 fwrite 함수에서 바이너리 모드("rb", "wb")를 사용하는 이유는 무엇입니까?

61unsigned int rleEncode(const unsigned char *src,
62 unsigned int srcLen,
63 unsigned char *dst,
64 unsigned int dstMax) {
65 unsigned int di = 0;
66 unsigned int i = 0;
67
68 while (i < srcLen) {
69 unsigned char ch = src[i];
70 unsigned int count = 1;
71 while (i + count < srcLen &&
72 src[i + count] == ch && count < 255) {
73 count++;
74 }
75

77 dst[di++] = (unsigned char)count;
78 dst[di++] = ch;
79 i += count;
80 }

114 unsigned int checksum; /* CRC-32 체크섬 (원본 기준) */
115 unsigned char flags; /* 비트 플래그 */
116 unsigned char reserved[3]; /* 예약 (정렬용) */

128 unsigned char buf[BUF_SIZE];
129 unsigned int remaining = size;
130 while (remaining > 0) {
131 unsigned int toRead = remaining < BUF_SIZE
132 ? remaining : BUF_SIZE;
133 unsigned int r = (unsigned int)fread(
134 buf, 1, toRead, src);
135 if (r == 0) return -1;
136 fwrite(buf, 1, r, dst);
137 remaining -= r;
138 }

161/* pack: 압축 지원 아카이브 묶기 */
162int pack(const char *arcName, const char *files[],
163 int count, int compress) {
164 FILE *arc = fopen(arcName, "wb");
165 if (!arc) {
166 fprintf(stderr, "오류: 아카이브 생성 실패\n");
167 return -1;
168 }
169
170 /* 매직 넘버 + 버전 */
171 const char magic[4] = {'M','A','R','3'};
172 fwrite(magic, 1, 4, arc);
173 fwrite(&count, sizeof(int), 1, arc);
174
175 unsigned int totalOrig = 0;
176 unsigned int totalStored = 0;
177
178 for (int i = 0; i < count; i++) {
179 FILE *fp = fopen(files[i], "rb");
180 if (!fp) {
181 fprintf(stderr, "오류: %s 열기 실패\n",
182 files[i]);
183 fclose(arc);
184 return -1;
185 }
186
187 /* 원본 데이터 읽기 */
188 unsigned int fsize = (unsigned int)getFileSize(fp);
189 unsigned char *origData = (unsigned char *)
190 malloc(fsize);
191 if (!origData) {
192 fclose(fp);
193 fclose(arc);
194 return -1;
195 }
196 fread(origData, 1, fsize, fp);
197 fclose(fp);
198
199 /* CRC 계산 */
200 unsigned int crc = crc32(origData, fsize);
201
202 /* 헤더 작성 */
203 FileHeader h;
204 memset(&h, 0, sizeof(h));
205 strncpy(h.name, files[i], NAME_LEN - 1);
206 h.origSize = fsize;
207 h.timestamp = (unsigned int)time(NULL);
208 h.checksum = crc;
209 h.flags = 0;
210
211 unsigned char *storeData = origData;
212 unsigned int storeSize = fsize;
213
214 /* RLE 압축 시도 */
215 unsigned char *compData = NULL;
216 if (compress && fsize > 0) {
217 compData = (unsigned char *)
218 malloc(fsize * 2);
219 if (compData) {
220 unsigned int cLen = rleEncode(
221 origData, fsize, compData,
222 fsize * 2);
223 if (cLen > 0 && cLen < fsize) {
224 /* 압축 효과가 있을 때만 사용 */
225 storeData = compData;
226 storeSize = cLen;
227 h.flags |= FLAG_COMPRESSED;
228 }
229 }
230 }
231
232 h.storedSize = storeSize;
233
234 fwrite(&h, sizeof(FileHeader), 1, arc);
235 fwrite(storeData, 1, storeSize, arc);
236
237 totalOrig += fsize;
238 totalStored += storeSize;
239
240 printf(" 추가: %-30s %8u -> %8u 바이트",
241 h.name, fsize, storeSize);
242 if (h.flags & FLAG_COMPRESSED)
243 printf(" [압축]\n");
244 else
245 printf("\n");
246
247 free(origData);
248 if (compData) free(compData);
249 }
250
251 fclose(arc);
252 printf("\n아카이브 '%s' 생성 완료\n", arcName);
253 printf(" 파일 수: %d\n", count);
254 printf(" 원본 합계: %u 바이트\n", totalOrig);
255 printf(" 저장 합계: %u 바이트\n", totalStored);
256 if (totalOrig > 0) {
257 printf(" 압축률: %.1f%%\n",
258 (1.0 - (double)totalStored / totalOrig)
259 * 100.0);
260 }
261 return 0;
262}

307/* unpack: 전체 추출 + CRC 검증 */
308int unpack(const char *arcName) {
309 FILE *arc = fopen(arcName, "rb");
310 if (!arc) {
311 fprintf(stderr, "오류: 아카이브 열기 실패\n");
312 return -1;
313 }
314
315 char magic[4];
316 fread(magic, 1, 4, arc);
317 if (memcmp(magic, "MAR3", 4) != 0) {
318 fprintf(stderr, "오류: 유효하지 않은 아카이브\n");
319 fclose(arc);
320 return -1;
321 }
322
323 int count;
324 fread(&count, sizeof(int), 1, arc);
325 int errors = 0;
326
327 for (int i = 0; i < count; i++) {
328 FileHeader h;
329 fread(&h, sizeof(FileHeader), 1, arc);
330
331 /* 저장된 데이터 읽기 */
332 unsigned char *stored = (unsigned char *)
333 malloc(h.storedSize);
334 if (!stored) {
335 fclose(arc);
336 return -1;
337 }
338 fread(stored, 1, h.storedSize, arc);
339
340 /* 압축 해제 */
341 unsigned char *origData;
342 unsigned int origLen;
343
344 if (h.flags & FLAG_COMPRESSED) {
345 origData = (unsigned char *)
346 malloc(h.origSize);
347 if (!origData) {
348 free(stored);
349 fclose(arc);
350 return -1;
351 }
352 origLen = rleDecode(stored, h.storedSize,
353 origData, h.origSize);
354 free(stored);
355 if (origLen != h.origSize) {
356 fprintf(stderr,
357 "오류: %s 압축 해제 크기 불일치\n",
358 h.name);
359 free(origData);
360 errors++;
361 continue;
362 }
363 } else {
364 origData = stored;
365 origLen = h.storedSize;
366 }
367
368 /* CRC 검증 */
369 unsigned int crc = crc32(origData, origLen);
370 if (crc != h.checksum) {
371 fprintf(stderr,
372 "경고: %s CRC 불일치 "
373 "(기대: %08X, 실제: %08X)\n",
374 h.name, h.checksum, crc);
375 errors++;
376 }
377
378 /* 디렉토리 생성 + 파일 쓰기 */
379 ensureDir(h.name);
380 FILE *fp = fopen(h.name, "wb");
381 if (!fp) {
382 fprintf(stderr, "오류: %s 생성 실패\n",
383 h.name);
384 free(origData);
385 errors++;
386 continue;
387 }
388 fwrite(origData, 1, origLen, fp);
389 fclose(fp);
390 free(origData);
391
392 printf(" 추출: %-30s %8u 바이트",
393 h.name, origLen);
394 if (h.flags & FLAG_COMPRESSED)
395 printf(" [압축 해제]");
396 if (crc == h.checksum)
397 printf(" [CRC OK]\n");
398 else
399 printf(" [CRC FAIL]\n");
400 }
401
402 fclose(arc);
403 printf("\n추출 완료: %d개 파일", count);
404 if (errors > 0)
405 printf(" (%d개 오류)\n", errors);
406 else
407 printf(" (오류 없음)\n");
408 return 0;
409}

요약

핵심 정리
1. 아카이브 = 헤더(이름+크기) + 데이터를 파일 개수만큼 연결한 구조
2. pack: 각 파일의 헤더를 기록한 뒤 데이터를 순서대로 복사
3. unpack: 헤더를 읽어 파일 이름과 크기를 파악한 뒤 데이터를 추출
4. list: 헤더만 읽고 데이터는 fseek로 건너뛰어 목록 출력
5. extract: 이름이 일치하는 파일만 추출, 나머지는 건너뜀
6. delete: 임시 파일에 대상 제외 복사 후 원본 교체
7. RLE 압축: 연속 반복 바이트를 (개수, 바이트) 쌍으로 인코딩
8. RLE은 반복이 많은 데이터에 효과적, 랜덤 데이터에는 역효과
9. CRC-32: 다항식 나눗셈 기반 체크섬으로 데이터 무결성 검증
10. 매직 넘버로 파일 형식을 식별하여 잘못된 파일 열기를 방지
