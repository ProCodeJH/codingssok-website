JSON 파서 만들기

JSON 형식을 파싱하는 프로그램을 3단계로 구현하며, 문자열 처리, 재귀 하강 파서, 트리 자료구조를 종합적으로 활용하는 프로젝트입니다.

이 유닛에서 배우는 것

  • JSON 문법 규칙을 이해하고 문자열, 숫자, 불리언 값을 파싱할 수 있습니다.
  • 재귀 하강 파서를 구현하여 중첩된 객체와 배열을 파싱할 수 있습니다.
  • 파싱 결과를 트리 구조(JsonNode)로 저장하고 경로 기반으로 접근할 수 있습니다.
  • JSON 파일을 읽어 파싱하고 예쁜 출력(pretty print) 형태로 표시할 수 있습니다.

66int parseString(char *out, int maxLen) {
67 if (peek() != '"') {
68 printf("오류: 문자열은 '\"'로 시작해야 합니다.\n");
69 return 0;

78 case '"': out[i++] = '"'; break;
79 case '\\': out[i++] = '\\'; break;
80 case 'n': out[i++] = '\n'; break;
81 case 't': out[i++] = '\t'; break;
82 case '/': out[i++] = '/'; break;

93 pos++; /* 닫는 '"' 건너뛰기 */

101
102int parseNumber(double *out) {
103 skipWhitespace();

105 int i = 0;
106
107 if (src[pos] == '-') buf[i++] = src[pos++];
108

110 buf[i++] = src[pos++];
111

113 buf[i++] = src[pos++];
114 while (isdigit((unsigned char)src[pos]) && i < 63)
115 buf[i++] = src[pos++];
116 }
117 buf[i] = '\0';
118

120 printf("오류: 유효한 숫자가 아닙니다.\n");
121 return 0;
122 }

156/* ===== 값 파싱 (타입 자동 판별) ===== */

158int parseValue(JsonPair *pair) {
159 char ch = peek();
160
161 if (ch == '"') {
162 pair->type = JSON_STRING;

164 }

166 pair->type = JSON_NUMBER;
167 return parseNumber(&pair->numVal);
168 }
169 if (ch == 't' || ch == 'f') {
170 pair->type = JSON_BOOL;
171 return parseBool(&pair->boolVal);
172 }
173 if (ch == 'n') {
174 pair->type = JSON_NULL;
175 return parseNull();
176 }
177
178 printf("오류: 알 수 없는 값 (위치 %d, 문자 '%c')\n",
179 pos, ch);
180 return 0;

183/* ===== 단순 객체 파싱 (중첩 없음) ===== */
184
185int parseSimpleObject(JsonObject *obj) {
186 obj->count = 0;
187
188 if (!expect('{')) return 0;
189
190 if (peek() == '}') {
191 advance();
192 return 1; /* 빈 객체 */
193 }
194
195 do {
196 if (obj->count >= MAX_PAIRS) {
197 printf("오류: key-value 쌍이 너무 많습니다.\n");
198 return 0;
199 }
200
201 JsonPair *pair = &obj->pairs[obj->count];
202
203 /* key 파싱 */
204 if (!parseString(pair->key, sizeof(pair->key)))
205 return 0;
206
207 /* ':' 확인 */
208 if (!expect(':')) return 0;
209
210 /* value 파싱 */
211 if (!parseValue(pair)) return 0;
212
213 obj->count++;
214
215 /* ',' 또는 '}' 확인 */
216 if (peek() == ',') {
217 advance();
218 } else {
219 break;
220 }
221 } while (1);
222
223 return expect('}');
224}

243 case JSON_NULL:

157JsonNode *parseObject(void) {

180 node->memberCount++;

183 else break;
184 } while (1);
185

187 return node;
188}

137/* ===== 문자열 파싱 (동적 할당) ===== */
138
139char *parseStringAlloc(void) {
140 if (peek() != '"') return NULL;
141 advance();
142
143 int cap = 64;
144 char *buf = (char *)malloc(cap);
145 int len = 0;
146
147 while (src[pos] && src[pos] != '"') {
148 if (len >= cap - 1) {
149 cap *= 2;
150 buf = (char *)realloc(buf, cap);
151 }
152 if (src[pos] == '\\') {
153 pos++;
154 switch (src[pos]) {
155 case '"': buf[len++] = '"'; break;
156 case '\\': buf[len++] = '\\'; break;
157 case '/': buf[len++] = '/'; break;
158 case 'n': buf[len++] = '\n'; break;
159 case 't': buf[len++] = '\t'; break;
160 case 'r': buf[len++] = '\r'; break;
161 case 'b': buf[len++] = '\b'; break;
162 case 'f': buf[len++] = '\f'; break;
163 default: buf[len++] = src[pos]; break;
164 }
165 } else {
166 buf[len++] = src[pos];
167 }
168 pos++;
169 }
170 buf[len] = '\0';
171
172 if (src[pos] == '"') { pos++; return buf; }
173 free(buf);
174 return NULL;
175}

177/* ===== 숫자 파싱 ===== */
178
179JsonNode *parseNumber(void) {
180 updatePosition();
181 const char *start = src + pos;
182 if (src[pos] == '-') pos++;
183 while (isdigit((unsigned char)src[pos])) pos++;
184 if (src[pos] == '.') {
185 pos++;
186 while (isdigit((unsigned char)src[pos])) pos++;
187 }
188 /* 지수 처리 */
189 if (src[pos] == 'e' || src[pos] == 'E') {
190 pos++;
191 if (src[pos] == '+' || src[pos] == '-') pos++;
192 while (isdigit((unsigned char)src[pos])) pos++;
193 }
194
195 JsonNode *n = createNode(JSON_NUMBER);
196 n->numVal = strtod(start, NULL);
197 return n;
198}

200/* ===== 객체 파싱 ===== */
201
202JsonNode *parseObject(void) {
203 if (!expect('{')) return NULL;
204 JsonNode *node = createNode(JSON_OBJECT);
205
206 if (peek() == '}') { advance(); return node; }
207
208 do {
209 char *key = parseStringAlloc();
210 if (!key) { freeNode(node); return NULL; }
211 if (!expect(':')) {
212 free(key); freeNode(node); return NULL;
213 }
214 JsonNode *val = parseValue();
215 if (!val) {
216 free(key); freeNode(node); return NULL;
217 }
218 addMember(node, key, val);
219 free(key);
220
221 if (peek() == ',') advance();
222 else break;
223 } while (1);
224
225 if (!expect('}')) { freeNode(node); return NULL; }
226 return node;
227}
228
229/* ===== 배열 파싱 ===== */
230
231JsonNode *parseArray(void) {
232 if (!expect('[')) return NULL;
233 JsonNode *node = createNode(JSON_ARRAY);
234
235 if (peek() == ']') { advance(); return node; }
236
237 do {
238 JsonNode *elem = parseValue();
239 if (!elem) { freeNode(node); return NULL; }
240 addElement(node, elem);
241
242 if (peek() == ',') advance();
243 else break;
244 } while (1);
245
246 if (!expect(']')) { freeNode(node); return NULL; }
247 return node;
248}

250/* ===== 값 파싱 ===== */
251
252JsonNode *parseValue(void) {
253 char ch = peek();
254 if (ch == '"') {
255 char *s = parseStringAlloc();
256 if (!s) return NULL;
257 JsonNode *n = createNode(JSON_STRING);
258 n->strVal = s;
259 return n;
260 }
261 if (ch == '{') return parseObject();
262 if (ch == '[') return parseArray();
263 if (ch == '-' || isdigit((unsigned char)ch))
264 return parseNumber();
265
266 updatePosition();
267 if (strncmp(src + pos, "true", 4) == 0) {
268 pos += 4;
269 JsonNode *n = createNode(JSON_BOOL);
270 n->boolVal = 1;
271 return n;
272 }
273 if (strncmp(src + pos, "false", 5) == 0) {
274 pos += 5;
275 JsonNode *n = createNode(JSON_BOOL);
276 n->boolVal = 0;
277 return n;
278 }
279 if (strncmp(src + pos, "null", 4) == 0) {
280 pos += 4;
281 return createNode(JSON_NULL);
282 }
283
284 printf("오류 [%d:%d]: 예상치 못한 문자 '%c'\n",
285 lineNum, colNum, ch);
286 return NULL;
287}

335 int i;
336 switch (node->type) {

380/* ===== 경로 기반 접근 ===== */
381/* 경로 예시: "data.users[0].name" */
382
383JsonNode *jsonGet(JsonNode *root, const char *path) {
384 if (!root || !path || path[0] == '\0')
385 return root;
386
387 char token[128];
388 int ti = 0;
389 int pi = 0; /* path index */
390
391 /* 토큰 추출 (점 또는 [ 전까지) */
392 while (path[pi] && path[pi] != '.' && path[pi] != '[')
393 token[ti++] = path[pi++];
394 token[ti] = '\0';
395
396 JsonNode *next = NULL;
397
398 /* 키 이름으로 객체 멤버 접근 */
399 if (ti > 0 && root->type == JSON_OBJECT) {
400 for (int i = 0; i < root->memberCount; i++) {
401 if (strcmp(root->members[i].key, token) == 0) {
402 next = root->members[i].value;
403 break;
404 }
405 }
406 if (!next) {
407 printf("경로 오류: 키 '%s'를 찾을 수 없습니다.\n",
408 token);
409 return NULL;
410 }
411 }
412
413 /* 배열 인덱스 접근: [N] */
414 if (path[pi] == '[') {
415 JsonNode *target = next ? next : root;
416 pi++; /* '[' 건너뛰기 */
417
418 int index = 0;
419 while (path[pi] && path[pi] != ']') {
420 index = index * 10 + (path[pi] - '0');
421 pi++;
422 }
423 if (path[pi] == ']') pi++; /* ']' 건너뛰기 */
424
425 if (target->type != JSON_ARRAY ||
426 index < 0 || index >= target->elementCount) {
427 printf("경로 오류: 인덱스 [%d] 범위 초과\n", index);
428 return NULL;
429 }
430 next = target->elements[index];
431 }
432
433 /* 다음 경로로 재귀 */
434 if (path[pi] == '.') pi++; /* '.' 건너뛰기 */
435
436 if (path[pi] == '\0')
437 return next;
438
439 return jsonGet(next, path + pi);
440}

대회 기출

요약

핵심 정리
1. JSON: 문자열, 숫자, 불리언, null, 객체({}), 배열([]) -- 6가지 타입
2. 파서 기본: peek()로 현재 문자 확인, advance()로 전진, expect()로 검증
3. 문자열 파싱: '"' 사이의 문자를 읽고 이스케이프 시퀀스(\n, \t 등) 처리
4. 숫자 파싱: 부호, 정수부, 소수부, 지수부를 순서대로 읽어 변환
5. 재귀 하강 파서: parseValue가 타입 판별 후 parseObject/parseArray 등 호출
6. 상호 재귀: parseObject -> parseValue -> parseObject (중첩 구조 처리)
7. 기본 경우(base case): 문자열, 숫자, 불리언, null은 재귀 없이 즉시 반환
8. 동적 메모리: 임의 크기 JSON을 처리하려면 malloc/realloc 필수
9. 경로 접근: 점(.) 표기법 + 배열 인덱스([])로 트리 내 원하는 값에 도달
10. 메모리 해제: 트리 구조이므로 자식 노드를 먼저 재귀 해제한 뒤 부모 해제
