버전 2: 확장 -- 연결 리스트 기반 + 검색/치환 + undo 1회

C
1#include <stdio.h>
2#include <stdlib.h>
3#include <string.h>
4
5#define MAX_LEN 256
6#define FILENAME "document.txt"
7
8/* ===== 줄 노드 정의 ===== */
9typedef struct Line {
10 char text[MAX_LEN];
11 struct Line *next;
12} Line;
13
14Line *head = NULL; /* 문서의 첫 줄 */
15int lineCount = 0; /* 줄 수 */
16
17/* ===== undo 정보 (1회) ===== */
18typedef enum { UNDO_NONE, UNDO_INSERT, UNDO_DELETE, UNDO_EDIT } UndoType;
19
20typedef struct {
21 UndoType type;
22 int pos; /* 작업 위치 (1-based) */
23 char oldText[MAX_LEN]; /* 이전 내용 (삭제/수정 시) */
24} UndoInfo;
25
26UndoInfo lastUndo = { UNDO_NONE, 0, "" };
27
28/* ===== 노드 생성 ===== */
29Line *createLine(const char *text) {
30 Line *node = (Line *)malloc(sizeof(Line));
31 if (node == NULL) {
32 printf("메모리 할당 실패\n");
33 return NULL;
34 }
35 strncpy(node->text, text, MAX_LEN - 1);
36 node->text[MAX_LEN - 1] = '\0';
37 node->next = NULL;
38 return node;
39}
40
41/* ===== n번째 노드 반환 (1-based) ===== */
42Line *getLine(int pos) {
43 Line *cur = head;
44 for (int i = 1; i < pos && cur != NULL; i++)
45 cur = cur->next;
46 return cur;
47}
48
49/* ===== 전체 출력 ===== */
50void printAll(void) {
51 if (head == NULL) {
52 printf("(빈 문서)\n");
53 return;
54 }
55 printf("\n--- 문서 내용 (%d줄) ---\n", lineCount);
56 Line *cur = head;
57 int num = 1;
58 while (cur != NULL) {
59 printf("%3d | %s\n", num++, cur->text);
60 cur = cur->next;
61 }
62 printf("--- 끝 ---\n");
63}

65/* ===== 줄 추가 (맨 끝) ===== */
66void appendLine(void) {
67 char buf[MAX_LEN];
68 printf("추가할 내용: ");
69 getchar();
70 fgets(buf, MAX_LEN, stdin);
71 buf[strcspn(buf, "\n")] = '\0';
72
73 Line *node = createLine(buf);
74 if (node == NULL) return;
75
76 if (head == NULL) {
77 head = node;
78 } else {
79 Line *cur = head;
80 while (cur->next != NULL)
81 cur = cur->next;
82 cur->next = node;
83 }
84 lineCount++;
85
86 lastUndo.type = UNDO_INSERT;
87 lastUndo.pos = lineCount;
88 printf("%d번 줄에 추가되었습니다.\n", lineCount);
89}

91/* ===== 특정 위치에 줄 삽입 ===== */
92void insertLine(void) {
93 int pos;
94 printf("삽입할 줄 번호 (1-%d): ", lineCount + 1);
95 scanf("%d", &pos);
96
97 if (pos < 1 || pos > lineCount + 1) {
98 printf("잘못된 위치입니다.\n");
99 return;
100 }
101
102 char buf[MAX_LEN];
103 printf("삽입할 내용: ");
104 getchar();
105 fgets(buf, MAX_LEN, stdin);
106 buf[strcspn(buf, "\n")] = '\0';
107
108 Line *node = createLine(buf);
109 if (node == NULL) return;
110
111 if (pos == 1) {
112 node->next = head;
113 head = node;
114 } else {
115 Line *prev = getLine(pos - 1);
116 node->next = prev->next;
117 prev->next = node;
118 }
119 lineCount++;
120
121 lastUndo.type = UNDO_INSERT;
122 lastUndo.pos = pos;
123 printf("%d번 줄에 삽입되었습니다.\n", pos);
124}

126/* ===== 줄 삭제 ===== */
127void deleteLine(void) {

129 printf("삭제할 줄이 없습니다.\n");

131 }
132 int pos;
133 printf("삭제할 줄 번호 (1-%d): ", lineCount);
134 scanf("%d", &pos);
135
136 if (pos < 1 || pos > lineCount) {
137 printf("잘못된 위치입니다.\n");

140

142 if (pos == 1) {

216/* ===== 치환 (replace) ===== */
217void replaceText(void) {
218 char oldWord[MAX_LEN], newWord[MAX_LEN];
219 printf("찾을 문자열: ");
220 getchar();
221 fgets(oldWord, MAX_LEN, stdin);
222 oldWord[strcspn(oldWord, "\n")] = '\0';
223
224 printf("바꿀 문자열: ");
225 fgets(newWord, MAX_LEN, stdin);
226 newWord[strcspn(newWord, "\n")] = '\0';
227
228 int replaced = 0;
229 Line *cur = head;
230 int num = 1;
231
232 while (cur != NULL) {
233 char *pos = strstr(cur->text, oldWord);
234 if (pos != NULL) {
235 /* 치환 수행 */
236 char temp[MAX_LEN];
237 int prefix = (int)(pos - cur->text);
238 strncpy(temp, cur->text, prefix);
239 temp[prefix] = '\0';
240 strcat(temp, newWord);
241 strcat(temp, pos + strlen(oldWord));
242 strcpy(cur->text, temp);
243
244 printf("%3d | %s\n", num, cur->text);
245 replaced++;
246 }
247 cur = cur->next;
248 num++;
249 }
250
251 if (replaced == 0)
252 printf("'%s'을(를) 찾을 수 없습니다.\n", oldWord);
253 else
254 printf("%d개의 줄에서 치환되었습니다.\n", replaced);
255}

257/* ===== undo (1회) ===== */
258void undo(void) {
259 switch (lastUndo.type) {
260 case UNDO_NONE:
261 printf("되돌릴 작업이 없습니다.\n");
262 break;
263
264 case UNDO_INSERT: {
265 /* 삽입 취소 = 해당 줄 삭제 */
266 int pos = lastUndo.pos;
267 Line *target;
268 if (pos == 1) {
269 target = head;
270 head = head->next;
271 } else {
272 Line *prev = getLine(pos - 1);
273 target = prev->next;
274 prev->next = target->next;
275 }
276 free(target);
277 lineCount--;
278 printf("삽입이 취소되었습니다 (%d번 줄).\n", pos);
279 break;
280 }
281
282 case UNDO_DELETE: {
283 /* 삭제 취소 = 해당 위치에 복원 */
284 int pos = lastUndo.pos;
285 Line *node = createLine(lastUndo.oldText);
286 if (node == NULL) return;
287
288 if (pos == 1) {
289 node->next = head;
290 head = node;
291 } else {
292 Line *prev = getLine(pos - 1);
293 node->next = prev->next;
294 prev->next = node;
295 }
296 lineCount++;
297 printf("삭제가 취소되었습니다 (%d번 줄 복원).\n", pos);
298 break;
299 }
300
301 case UNDO_EDIT: {
302 /* 수정 취소 = 이전 내용 복원 */
303 Line *target = getLine(lastUndo.pos);
304 strcpy(target->text, lastUndo.oldText);
305 printf("수정이 취소되었습니다 (%d번 줄).\n",
306 lastUndo.pos);
307 break;
308 }
309 }
310 lastUndo.type = UNDO_NONE;
311}

핵심 정리
1. 텍스트 에디터: 줄(line) 단위로 텍스트를 관리하는 프로그램
2. 배열 기반: 구현 간단, 삽입/삭제 시 O(n) 이동 필요
3. 연결 리스트 기반: 삽입/삭제 O(1), 임의 접근 O(n)
4. 파일 저장: fprintf로 줄 단위 기록, fgets로 줄 단위 읽기
5. 검색(find): strstr로 부분 문자열 검색, 줄 번호 출력
6. 치환(replace): 찾은 위치 기준 앞부분+새문자열+뒷부분 조합
7. undo 1회: 마지막 작업의 종류와 데이터를 구조체에 저장
8. 다중 undo/redo: 스택 두 개로 작업 이력 관리
9. 구문 강조: C 키워드를 ANSI 이스케이프 코드로 색상 출력
10. 커서: 현재 줄 위치를 추적하여 편집 기준점으로 활용
