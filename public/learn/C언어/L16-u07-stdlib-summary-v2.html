종합 예제: 모든 라이브러리 활용

C
1#include <stdio.h>
2#include <stdlib.h>
3#include <string.h>
4#include <math.h>
5#include <ctype.h>
6#include <time.h>
7
8typedef struct {
9 char name[50];
10 double score;
11} Student;
12
13int cmp_score(const void *a, const void *b) {
14 double diff = ((Student *)b)->score - ((Student *)a)->score;
15 return (diff > 0) - (diff < 0);
16}
17
18int main(void) {
19 srand((unsigned)time(NULL));
20 int n = 5;
21 Student *students = (Student *)malloc(n * sizeof(Student));
22 if (!students) { perror("malloc"); return 1; }
23
24 /* 데이터 생성 */
25 const char *names[] = {"Alice", "Bob", "Charlie", "David", "Eve"};
26 for (int i = 0; i < n; i++) {
27 strcpy(students[i].name, names[i]);
28 students[i].score = (rand() % 4001) / 100.0 + 60.0;
29 /* toupper로 이름 첫 글자 대문자 확인 */
30 students[i].name[0] = toupper(students[i].name[0]);
31 }
32
33 /* qsort로 점수 내림차순 정렬 */
34 qsort(students, n, sizeof(Student), cmp_score);
35
36 /* 결과 출력 */
37 printf("===== 성적 순위표 =====\n");
38 double sum = 0;
39 for (int i = 0; i < n; i++) {
40 printf("%d위: %-10s %.2f점\n", i + 1, students[i].name, students[i].score);
41 sum += students[i].score;
42 }
43 double avg = sum / n;
44 printf("평균: %.2f / 표준편차: %.2f\n", avg,
45 sqrt(sum / n - avg * avg > 0 ? sum / n - avg * avg : 0));
46
47 /* 현재 시각 출력 */
48 time_t now = time(NULL);
49 char timebuf[64];
50 strftime(timebuf, sizeof(timebuf), "%Y-%m-%d %H:%M:%S", localtime(&now));
51 printf("출력 시각: %s\n", timebuf);
52
53 /* 파일 저장 */
54 FILE *fp = fopen("result.txt", "w");
55 if (fp) {
56 fprintf(fp, "성적 결과 (%s)\n", timebuf);
57 for (int i = 0; i < n; i++)
58 fprintf(fp, "%s: %.2f\n", students[i].name, students[i].score);
59 fclose(fp);
60 printf("result.txt에 저장 완료\n");
61 }
62
63 free(students);
64 return 0;
65}

에러 체험

에러 1: 헤더 파일 누락
#include
int main(void) {
int len = strlen("hello"); // 에러!
}

-> strlen은 string.h에 선언되어 있습니다.
-> #include 를 추가해야 합니다.
에러 2: math.h 링크 누락 (Linux/Mac)
gcc main.c // sqrt 등에서 undefined reference 에러
gcc main.c -lm // 정상 컴파일

-> math.h 함수 사용 시 -lm 플래그를 추가해야 합니다.
-> Windows(MSVC, MinGW)에서는 자동 링크됩니다.
에러 3: atoi 반환값 검증 누락
int n = atoi("abc"); // n = 0 (에러가 아님!)
int m = atoi(""); // m = 0

-> atoi는 변환 실패 시 0을 반환하여 정상값과 구분이 불가합니다.
-> 안전한 대안: strtol을 사용하면 에러를 감지할 수 있습니다.
에러 4: strcpy 버퍼 오버플로
char buf[5];
strcpy(buf, "Hello World"); // 버퍼 초과!

-> buf는 5바이트인데 11글자 + null = 12바이트를 복사합니다.
-> strncpy(buf, src, sizeof(buf) - 1)로 크기를 제한하십시오.
