149int partitionByTotal(Student arr[], int low, int high) {
150 int pivot = arr[high].total;
151 int i = low - 1;
152 for (int j = low; j < high; j++) {
153 if (arr[j].total >= pivot) /* 내림차순 */
154 swapStudent(&arr[++i], &arr[j]);
155 }
156 swapStudent(&arr[i + 1], &arr[high]);
157 return i + 1;
158}

160void quickSortByTotal(Student arr[], int low, int high) {
161 if (low < high) {
162 int pi = partitionByTotal(arr, low, high);
163 quickSortByTotal(arr, low, pi - 1);
164 quickSortByTotal(arr, pi + 1, high);
165 }
166}

에러 체험

시험에서 자주 하는 실수 1: 변수 초기화 누락
int sum;
for (int i = 0; i < n; i++)
sum += arr[i];

-> 지역 변수 sum을 초기화하지 않으면 쓰레기 값에 누적됩니다.
-> 반드시 int sum = 0;으로 초기화하십시오.
-> 시험에서 가장 흔한 실수이며, 디버깅이 어려운 버그입니다.
시험에서 자주 하는 실수 2: 배열 범위 초과
int arr[10];
for (int i = 0; i <= 10; i++) /* i = 10일 때 범위 초과! */
arr[i] = i;

-> arr[10]은 유효하지 않은 접근입니다. 인덱스는 0~9까지입니다.
-> 조건을 i < 10으로 수정하십시오.
-> 범위 초과는 즉시 오류를 발생시키지 않아 발견이 어렵습니다.
시험에서 자주 하는 실수 3: 동적 메모리 해제 누락
void process(void) {
int *p = (int *)malloc(100 * sizeof(int));
/* ... 작업 수행 ... */
if (error_condition) return; /* free 없이 반환! */
free(p);
}

-> 오류 조건에서 조기 반환할 때 free를 빠뜨리면 메모리 누수가 발생합니다.
-> 함수의 모든 종료 경로에서 free를 호출하거나, goto cleanup 패턴을 사용하십시오.
시험에서 자주 하는 실수 4: 문자열 널 종료 누락
char str[5];
for (int i = 0; i < 5; i++)
str[i] = 'A' + i;
printf("%s\n", str); /* 널 종료 없이 출력! */

-> 문자열 끝에 '\0'이 없으면 printf는 쓰레기 값까지 출력합니다.
-> 배열 크기를 6으로 하고 str[5] = '\0';을 추가하십시오.
시험에서 자주 하는 실수 5: 포인터 사용 전 NULL 검사 누락
int *p = (int *)malloc(n * sizeof(int));
p[0] = 10; /* malloc 실패 시 NULL 역참조! */

-> malloc은 메모리 부족 시 NULL을 반환합니다.
-> 반드시 if (p == NULL) 검사를 수행하십시오.
-> 시험 코드 분석 문제에서 자주 출제되는 패턴입니다.
