버전 3 — 세 과목 종합 판정

C
1#include <stdio.h>
2int main(void) {
3 int kor, eng, math;
4
5 printf("국어: "); scanf("%d", &kor);
6 printf("영어: "); scanf("%d", &eng);
7 printf("수학: "); scanf("%d", &math);
8
9 // 입력 검증
10 if (kor < 0 || kor > 100 || eng < 0 || eng > 100 || math < 0 || math > 100) {
11 printf("잘못된 점수가 있습니다.\n");
12 return 1;
13 }
14
15 int total = kor + eng + math;
16 double avg = (double)total / 3;
17
18 char grade;
19 if (avg >= 90) grade = 'A';
20 else if (avg >= 80) grade = 'B';
21 else if (avg >= 70) grade = 'C';
22 else if (avg >= 60) grade = 'D';
23 else grade = 'F';
24
25 printf("\n===== 성적 리포트 =====\n");
26 printf("국어: %3d\n", kor);
27 printf("영어: %3d\n", eng);
28 printf("수학: %3d\n", math);
29 printf("-----------------------\n");
30 printf("총점: %3d\n", total);
31 printf("평균: %5.1f\n", avg);
32 printf("학점: %c\n", grade);
33
34 // 최고/최저 과목
35 int max = kor;
36 if (eng > max) max = eng;
37 if (math > max) max = math;
38
39 int min = kor;
40 if (eng < min) min = eng;
41 if (math < min) min = math;
42
43 printf("최고: %d / 최저: %d / 편차: %d\n", max, min, max - min);
44 printf("========================\n");
45
46 return 0;
47}
실행 결과
국어: 92 영어: 78 수학: 85 ===== 성적 리포트 ===== 국어: 92 영어: 78 수학: 85 ----------------------- 총점: 255 평균: 85.0 학점: B 최고: 92 / 최저: 78 / 편차: 14 ========================

에러 체험

에러 1: 범위 겹침
if (score >= 80) grade = 'B';
else if (score >= 90) grade = 'A'; // 절대 도달 못함!

→ 90점도 80 이상이므로 B로 판정됩니다.
→ 높은 점수부터 검사해야 합니다.
에러 2: 평균 계산 정수 나눗셈
int avg = total / 3; // 소수점 버려짐

→ 255 / 3 = 85 (85.0이 아님)
→ double avg = (double)total / 3; 사용
에러 3: 입력 검증 누락
scanf("%d", &score);
if (score >= 90) grade = 'A';
// -50이 입력되면?

→ 잘못된 입력에 대한 검증 없이 F가 됩니다.
→ 0~100 범위 확인을 먼저 하십시오.

핵심 정리
