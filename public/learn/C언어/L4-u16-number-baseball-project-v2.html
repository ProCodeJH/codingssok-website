버전 1 — 숫자 맞추기 (단순 버전)

C
1#include <stdio.h>
2int main(void) {
3 int answer = 7;
4 int tries = 0;
5
6 printf("=== 숫자 맞추기 (1~9) ===\n");
7 while (1) {
8 int guess;
9 printf("입력: ");
10 scanf("%d", &guess);
11 tries++;
12
13 if (guess == answer) {
14 printf("정답! %d번 만에 맞추었습니다.\n", tries);
15 break;
16 } else if (guess < answer) {
17 printf("더 큽니다.\n");
18 } else {
19 printf("더 작습니다.\n");
20 }
21 }
22 return 0;
23}

버전 2 — 두 자리 숫자 야구

C
1#include <stdio.h>
2int main(void) {
3 int answer = 37; // 두 자리 정답
4 int a_ten = answer / 10; // 3
5 int a_one = answer % 10; // 7
6 int tries = 0;
7
8 printf("=== 두 자리 숫자 야구 ===\n");
9 while (1) {
10 int guess;
11 printf("두 자리 수 입력: ");
12 scanf("%d", &guess);
13
14 if (guess < 10 || guess > 99) {
15 printf("두 자리 수를 입력하십시오.\n");
16 continue;
17 }
18 tries++;
19
20 int g_ten = guess / 10;
21 int g_one = guess % 10;
22
23 int strike = 0, ball = 0;
24
25 // 스트라이크 판정 (같은 자리, 같은 숫자)
26 if (g_ten == a_ten) strike++;
27 if (g_one == a_one) strike++;
28
29 // 볼 판정 (다른 자리, 같은 숫자)
30 if (g_ten != a_ten && g_ten == a_one) ball++;
31 if (g_one != a_one && g_one == a_ten) ball++;
32
33 if (strike == 2) {
34 printf("정답! %d번 만에 맞추었습니다.\n", tries);
35 break;
36 } else if (strike == 0 && ball == 0) {
37 printf("아웃!\n");
38 } else {
39 printf("%d스트라이크 %d볼\n", strike, ball);
40 }
41 }
42 return 0;
43}
실행 결과
=== 두 자리 숫자 야구 === 두 자리 수 입력: 35 1스트라이크 0볼 두 자리 수 입력: 73 0스트라이크 2볼 두 자리 수 입력: 37 정답! 3번 만에 맞추었습니다.

버전 3 — 세 자리 숫자 야구 + 시도 제한

C
1#include <stdio.h>
2int main(void) {
3 int answer = 359;
4 int a1 = answer / 100; // 3 (백의 자리)
5 int a2 = (answer / 10) % 10; // 5 (십의 자리)
6 int a3 = answer % 10; // 9 (일의 자리)
7 int tries = 0;
8 int max_tries = 10;
9
10 printf("=== 세 자리 숫자 야구 ===\n");
11 printf("기회: %d번\n\n", max_tries);
12
13 while (tries < max_tries) {
14 int guess;
15 printf("[%d/%d] 세 자리 수: ", tries + 1, max_tries);
16 scanf("%d", &guess);
17
18 if (guess < 100 || guess > 999) {
19 printf("세 자리 수를 입력하십시오.\n");
20 continue;
21 }
22 tries++;
23
24 int g1 = guess / 100;
25 int g2 = (guess / 10) % 10;
26 int g3 = guess % 10;
27
28 int strike = 0, ball = 0;
29
30 // 스트라이크
31 if (g1 == a1) strike++;
32 if (g2 == a2) strike++;
33 if (g3 == a3) strike++;
34
35 // 볼 (같은 자리가 아닌 경우만)
36 if (g1 != a1 && (g1 == a2 || g1 == a3)) ball++;
37 if (g2 != a2 && (g2 == a1 || g2 == a3)) ball++;
38 if (g3 != a3 && (g3 == a1 || g3 == a2)) ball++;
39
40 if (strike == 3) {
41 printf("\n정답! %d번 만에 맞추었습니다!\n", tries);
42
43 if (tries <= 3) printf("평가: 천재!\n");
44 else if (tries <= 5) printf("평가: 우수!\n");
45 else if (tries <= 7) printf("평가: 보통\n");
46 else printf("평가: 아슬아슬...\n");
47 return 0;
48 }
49
50 if (strike == 0 && ball == 0) {
51 printf("아웃!\n");
52 } else {
53 printf("%dS %dB\n", strike, ball);
54 }
55 }
56
57 printf("\n게임 오버! 정답은 %d였습니다.\n", answer);
58 return 0;
59}
실행 결과
=== 세 자리 숫자 야구 === 기회: 10번 [1/10] 세 자리 수: 123 0S 1B [2/10] 세 자리 수: 345 1S 1B [3/10] 세 자리 수: 359 정답! 3번 만에 맞추었습니다! 평가: 천재!

에러 체험

에러 1: 볼 판정에서 스트라이크 중복 카운트
// 잘못된 코드: 같은 자리인데도 볼로 카운트
if (g1 == a2 || g1 == a3) ball++; // g1 == a1일 때도 볼 추가!

→ 스트라이크가 아닌 경우에만 볼을 검사하십시오.
→ if (g1 != a1 && (g1 == a2 || g1 == a3)) ball++;
에러 2: 자릿수 분해 오류
int a2 = answer / 10; // 35 (십의 자리가 아님!)

→ 십의 자리: (answer / 10) % 10
→ 359 / 10 = 35, 35 % 10 = 5
에러 3: 입력 검증 후 tries 증가
if (guess < 100) {
printf("잘못된 입력\n");
// continue 없이 tries++가 실행됨!
}
tries++;

→ 잘못된 입력에서는 continue로 건너뛰어야 합니다.
