선택 기준

대표 패턴 모음

패턴 1: 카운터
int count = 0;
for (int i = 0; i < n; i++) {
if (조건) count++;
}
→ 조건을 만족하는 개수 세기
패턴 2: 누적 합계
int sum = 0;
for (int i = 1; i <= n; i++) {
sum += i;
}
→ 합계는 0으로 초기화, 곱은 1로 초기화
패턴 3: 최대/최소
int max;
scanf("%d", &max);
for (int i = 1; i < n; i++) {
int x; scanf("%d", &x);
if (x > max) max = x;
}
→ 첫 값으로 초기화, 반복하며 갱신
패턴 4: 검색 (break)
int found = 0;
for (int i = 2; i < n; i++) {
if (n % i == 0) { found = 1; break; }
}
→ 찾으면 즉시 종료
패턴 5: 필터 (continue)
for (int i = 0; i < n; i++) {
int x; scanf("%d", &x);
if (x < 0) continue;
sum += x;
}
→ 조건에 맞지 않는 항목 건너뛰기
패턴 6: 무한 루프 + break
while (1) {
int x; scanf("%d", &x);
if (x == 0) break;
sum += x;
}
→ 종료 조건이 복잡하거나 중간에 있을 때
패턴 7: 입력 검증 (do-while)
int x;
do {
printf("1~100: ");
scanf("%d", &x);
} while (x < 1 || x > 100);
→ 유효한 입력이 들어올 때까지 반복
패턴 8: 중첩 반복
for (int i = 0; i < rows; i++) {
for (int j = 0; j < cols; j++) {
printf("*");
}
printf("\n");
}
→ 바깥=행, 안쪽=열 (2차원 출력)

에러 체험

에러 1: for문에서 do-while이 필요한 상황
for (int i = 0; i < ???; i++) {
scanf("%d", &x); // 몇 번 입력받을지 모름
}

→ 입력 개수가 불확실하면 while(1)+break 또는 do-while 사용
에러 2: break로 바깥 반복문 종료 시도
for (int i = 0; i < 3; i++) {
for (int j = 0; j < 3; j++) {
if (...) break; // 안쪽 for만 종료!
}
}

→ 바깥 종료는 플래그 변수 사용: int done = 0;
에러 3: 불필요하게 복잡한 반복
// 너무 복잡
int i = 0;
while (1) {
if (i >= 10) break;
printf("%d\n", i);
i++;
}

// 간결하게
for (int i = 0; i < 10; i++) printf("%d\n", i);
