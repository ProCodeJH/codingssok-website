Chapter 18

메뉴 프로그램 (반복+조건 종합 프로젝트)

반복문과 조건문의 모든 기법을 결합하여 메뉴 기반 종합 프로그램을 단계적으로 완성합니다.

이 유닛에서 배우는 것

  • do-while + switch로 메뉴 반복 구조를 설계할 수 있습니다.
  • for, while, break, continue를 목적에 맞게 선택하여 사용할 수 있습니다.
  • 입력 검증, 누적 계산, 결과 출력을 포함한 종합 프로그램을 완성할 수 있습니다.
  • Layer 4에서 배운 모든 반복문 기법을 하나의 프로그램에 통합할 수 있습니다.

C
1#include <stdio.h>
2int main(void) {
3 int choice;
4
5 do {
6 printf("\n=== 계산기 ===\n");

12 printf("선택: ");
13 scanf("%d", &choice);
14
15 if (choice == 0) break;
16
17 int a, b;
18 printf("두 정수 입력: ");
19 scanf("%d %d", &a, &b);
20
21 switch (choice) {
22 case 1:
23 printf("%d + %d = %d\n", a, b, a + b);
24 break;
25 case 2:
26 printf("%d - %d = %d\n", a, b, a - b);
27 break;
28 case 3:
29 printf("%d * %d = %d\n", a, b, a * b);
30 break;
31 case 4:
32 if (b == 0) {
33 printf("0으로 나눌 수 없습니다.\n");
34 } else {
35 printf("%d / %d = %d (나머지 %d)\n",
36 a, b, a / b, a % b);
37 }
38 break;
39 default:
40 printf("잘못된 선택입니다.\n");
41 }
42 } while (choice != 0);

에러 체험

에러 1: switch의 case 안에서 변수 선언
switch (choice) {
case 1:
int a; // 컴파일 에러 가능!

→ case 안에서 변수를 선언하려면 중괄호로 블록을 만드십시오.
→ case 1: { int a; ... break; }
에러 2: do-while 끝에 세미콜론 누락
do {
...
} while (choice != 0) // 세미콜론 없음!

→ do-while문은 반드시 세미콜론으로 끝나야 합니다.
→ } while (choice != 0);
에러 3: break 위치 혼동
case 1:
printf("덧셈\n");
// break 빠뜨림!
case 2:
printf("뺄셈\n"); // case 1에서 여기까지 실행

→ switch의 break는 다음 case로의 fall-through를 방지합니다.
→ 반복문의 break와 역할이 다르니 혼동하지 마십시오.
