로또 번호 생성기 (프로젝트)

배열의 중복 검사, 정렬, 비교를 결합하여 로또 번호를 생성하고 당첨 여부를 판정합니다.

이 유닛에서 배우는 것

  • 배열에서 중복을 검사하는 알고리즘을 구현할 수 있습니다.
  • 생성된 번호를 정렬하여 보기 좋게 출력할 수 있습니다.
  • 두 배열을 비교하여 일치하는 요소 수를 구할 수 있습니다.
  • 배열 탐색, 정렬, 비교를 하나의 프로젝트에 종합할 수 있습니다.

버전 1 — 중복 없는 번호 생성

C
1#include <stdio.h>
2int main(void) {
3 // 고정 번호 풀 (랜덤 대신)
4 int pool[] = {23, 5, 41, 5, 12, 23, 7, 33, 41, 12,
5 18, 7, 45, 33, 2, 38, 27, 9, 15, 42};
6 int pn = sizeof(pool) / sizeof(pool[0]);
7
8 int lotto[6];
9 int count = 0;
10
11 for (int p = 0; p < pn && count < 6; p++) {
12 // 중복 검사
13 int dup = 0;
14 for (int i = 0; i < count; i++) {
15 if (lotto[i] == pool[p]) {
16 dup = 1;
17 break;
18 }
19 }
20 if (!dup) {
21 lotto[count++] = pool[p];
22 }
23 }
24
25 printf("생성된 번호: ");
26 for (int i = 0; i < 6; i++)
27 printf("%d ", lotto[i]);
28 printf("\n");
29 return 0;
30}

생성된 번호: 23 5 41 12 7 33

버전 2 — 정렬 + 당첨 비교

C
1#include <stdio.h>
2int main(void) {
3 int my[6] = {23, 5, 41, 12, 7, 33};
4 int win[6] = {5, 12, 27, 33, 38, 45};
5
6 // 오름차순 정렬 (버블 소트)
7 for (int i = 0; i < 5; i++)
8 for (int j = 0; j < 5 - i; j++)
9 if (my[j] > my[j+1]) {
10 int t = my[j]; my[j] = my[j+1]; my[j+1] = t;
11 }
12
13 printf("내 번호: ");
14 for (int i = 0; i < 6; i++) printf("%2d ", my[i]);
15 printf("\n당첨 번호: ");
16 for (int i = 0; i < 6; i++) printf("%2d ", win[i]);
17
18 // 일치 검사
19 int match = 0;
20 int matched[6] = {0};
21 for (int i = 0; i < 6; i++)
22 for (int j = 0; j < 6; j++)
23 if (my[i] == win[j]) {
24 matched[match++] = my[i];
25 break;
26 }
27
28 printf("\n\n일치: %d개", match);
29 if (match > 0) {
30 printf(" (");
31 for (int i = 0; i < match; i++) {
32 if (i > 0) printf(", ");
33 printf("%d", matched[i]);
34 }
35 printf(")");
36 }
37
38 printf("\n결과: ");
39 if (match == 6) printf("1등!\n");
40 else if (match == 5) printf("3등\n");
41 else if (match == 4) printf("4등\n");
42 else if (match == 3) printf("5등\n");
43 else printf("낙첨\n");
44 return 0;
45}

버전 3 — 여러 게임 + 통계

C
1#include <stdio.h>
2int main(void) {
3 int win[6] = {7, 14, 21, 28, 35, 42};
4
5 // 5게임 분의 번호
6 int games[5][6] = {
7 {7, 14, 21, 28, 35, 42},
8 {3, 14, 25, 28, 36, 42},
9 {1, 8, 15, 22, 29, 43},
10 {7, 21, 33, 35, 40, 42},
11 {14, 28, 35, 37, 41, 42}
12 };
13
14 printf("당첨 번호: ");
15 for (int i = 0; i < 6; i++) printf("%2d ", win[i]);
16 printf("\n\n");
17
18 int total_match = 0;
19 int rank_cnt[6] = {0}; // 1~5등 + 낙첨
20
21 for (int g = 0; g < 5; g++) {
22 printf("게임 %d: ", g + 1);
23 for (int i = 0; i < 6; i++)
24 printf("%2d ", games[g][i]);
25
26 int match = 0;
27 for (int i = 0; i < 6; i++)
28 for (int j = 0; j < 6; j++)
29 if (games[g][i] == win[j]) {
30 match++;
31 break;
32 }
33
34 total_match += match;
35 printf("→ %d개 일치 ", match);
36
37 if (match == 6) { printf("(1등!)\n"); rank_cnt[0]++; }
38 else if (match == 5) { printf("(3등)\n"); rank_cnt[2]++; }
39 else if (match == 4) { printf("(4등)\n"); rank_cnt[3]++; }
40 else if (match == 3) { printf("(5등)\n"); rank_cnt[4]++; }
41 else { printf("(낙첨)\n"); rank_cnt[5]++; }
42 }
43
44 printf("\n=== 통계 ===\n");
45 printf("총 일치 수: %d / %d\n", total_match, 5 * 6);
46 printf("평균 일치: %.1f개\n", (double)total_match / 5);
47 char *names[] = {"1등", "2등", "3등", "4등", "5등", "낙첨"};
48 for (int i = 0; i < 6; i++)
49 if (rank_cnt[i] > 0)
50 printf("%s: %d회\n", names[i], rank_cnt[i]);
51 return 0;
52}

당첨 번호: 7 14 21 28 35 42 게임 1: 7 14 21 28 35 42 → 6개 일치 (1등!) 게임 2: 3 14 25 28 36 42 → 3개 일치 (5등) 게임 3: 1 8 15 22 29 43 → 0개 일치 (낙첨) 게임 4: 7 21 33 35 40 42 → 4개 일치 (4등) 게임 5: 14 28 35 37 41 42 → 4개 일치 (4등) === 통계 === 총 일치 수: 17 / 30 평균 일치: 3.4개 1등: 1회 4등: 2회 5등: 1회 낙첨: 1회

에러 체험

에러 1: 중복 검사 없이 번호 생성
for (int i = 0; i < 6; i++)
lotto[i] = pool[i]; // pool에 중복이 있으면 그대로 들어감!

→ 매번 이미 선택된 번호인지 검사해야 합니다.
에러 2: 일치 검사에서 이중 카운트
for (int i = 0; i < 6; i++)
for (int j = 0; j < 6; j++)
if (my[i] == win[j]) match++; // break 없으면 중복!

→ 내부 루프에서 일치하면 break하십시오.
에러 3: 2차원 배열 인덱스 혼동
games[g][i] // g번째 게임의 i번째 번호 (올바름)
games[i][g] // 게임과 번호 순서가 뒤바뀜!

→ [행][열] 순서를 정확히 지키십시오.
