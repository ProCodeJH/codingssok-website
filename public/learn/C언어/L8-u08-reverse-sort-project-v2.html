14/* 투 포인터: 정렬된 배열에서 합이 target인 쌍 찾기 */
15int findPair(int *arr, int n, int target) {
16 int *left = arr;
17 int *right = arr + n - 1;
18 while (left < right) {
19 int sum = *left + *right;
20 if (sum == target) {
21 printf("쌍 발견: %d + %d = %d\n", *left, *right, target);
22 return 1;
23 }
24 if (sum < target) left++;
25 else right--;
26 }
27 return 0;
28}

30/* 정렬된 배열 중복 제거 (제자리) */
31int removeDup(int *arr, int n) {
32 if (n <= 1) return n;
33 int *write = arr + 1;
34 for (int *read = arr + 1; read < arr + n; read++)
35 if (*read != *(read - 1))
36 *write++ = *read;
37 return write - arr;
38}

40/* 정렬된 두 배열 병합 */
41void merge(int *a, int na, int *b, int nb, int *out) {
42 int *pa = a, *pb = b, *po = out;
43 while (pa < a+na && pb < b+nb)
44 *po++ = (*pa <= *pb) ? *pa++ : *pb++;
45 while (pa < a+na) *po++ = *pa++;
46 while (pb < b+nb) *po++ = *pb++;
47}

대회 기출

요약

핵심 정리
1. 투 포인터: 양쪽 끝에서 중앙으로 좁히며 처리
2. reverse: swap(left++, right--) while left < right
3. 포인터 정렬: *i, *j로 비교, swap(i, j)로 교환
4. 합 탐색: sum < target → left++, sum > target → right--
5. 배열 회전: reverse 3번으로 O(n) 구현
6. 병합: 두 포인터로 정렬 순서 유지하며 합치기
