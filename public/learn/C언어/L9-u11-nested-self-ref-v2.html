자기참조 구조체의 규칙
typedef struct Node {
int data;
struct Node *next; ← struct Node* 사용 필수
} Node;

1. 자기 자신의 '포인터'만 멤버로 가질 수 있습니다.
(자기 자신을 값으로 가지면 크기가 무한대)
2. typedef 완료 전이므로 멤버에는 struct Node* 사용
3. typedef 후에는 Node *p; 사용 가능

2
3typedef struct Node {
4 int data;
5 struct Node *next;

8/* 순회: 모든 노드 출력 */
9void printList(const Node *head) {
10 const Node *cur = head;
11 while (cur != NULL) {
12 printf("[%d]->" , cur->data);
13 cur = cur->next;
14 }
15 printf("NULL\n");
16}
17
18/* 검색: 값이 있으면 포인터, 없으면 NULL */
19Node *findNode(Node *head, int value) {
20 Node *cur = head;
21 while (cur != NULL) {
22 if (cur->data == value) return cur;
23 cur = cur->next;
24 }
25 return NULL;
26}
27
28/* 노드 개수 세기 */
29int countNodes(const Node *head) {
30 int cnt = 0;
31 const Node *cur = head;
32 while (cur != NULL) {
33 cnt++;
34 cur = cur->next;
35 }
36 return cnt;
37}

39int main(void) {
40 Node n1={10,NULL}, n2={20,NULL}, n3={30,NULL}, n4={40,NULL};
41 n1.next=&n2; n2.next=&n3; n3.next=&n4;
42
43 printList(&n1);
44 printf("노드 수: %d\n", countNodes(&n1));
45
46 Node *found = findNode(&n1, 30);
47 if (found)
48 printf("찾음: %d\n", found->data);

50}

대회 기출

요약

핵심 정리
1. 중첩 구조체: Student 안에 Address addr;
2. 접근: s.addr.city, sp->addr.city
3. 자기참조: typedef struct Node { int data; struct Node *next; } Node;
4. 포인터만 가능: 값으로 자기 자신 포함 불가 (크기 무한)
5. 연결 리스트 순회: while (cur != NULL) cur = cur->next;
6. NULL 체크 필수: 마지막 노드의 next는 NULL
