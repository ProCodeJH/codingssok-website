4void two_sum(int arr[], int n, int target) {

6 for (int i = 0; i < n; i++)

13// 문제 2: 배열 회전
14void rotate(int arr[], int n, int k) {
15 k = k % n;
16 // 간단 방법: 임시 배열
17 int temp[100];
18 for (int i = 0; i < n; i++)
19 temp[(i+k) % n] = arr[i];
20 for (int i = 0; i < n; i++) arr[i] = temp[i];
21}

23// 문제 3: 중복 제거 (정렬된 배열)
24int remove_dups(int arr[], int n) {
25 if (n == 0) return 0;
26 int j = 0;
27 for (int i = 1; i < n; i++)
28 if (arr[i] != arr[j]) arr[++j] = arr[i];
29 return j + 1;
30}

7 int left = 0, right = strlen(s) - 1;

9 if (tolower(s[left]) != tolower(s[right])) return 0;
10 left++; right--;
11 }
12 return 1;
13}

15// 단어 수 세기
16int count_words(const char *s) {
17 int count = 0, in_word = 0;
18 for (int i = 0; s[i]; i++) {
19 if (isspace(s[i])) in_word = 0;
20 else if (!in_word) { in_word = 1; count++; }
21 }
22 return count;
23}

27 for (int i = 0; s[i]; i++) {
28 if (isupper(s[i])) s[i] = tolower(s[i]);
29 else if (islower(s[i])) s[i] = toupper(s[i]);
30 }
31}

3// 버블 정렬
4void bubble_sort(int arr[], int n) {
5 for (int i = 0; i < n-1; i++)
6 for (int j = 0; j < n-1-i; j++)
7 if (arr[j] > arr[j+1]) {
8 int t = arr[j]; arr[j] = arr[j+1]; arr[j+1] = t;
9 }
10}
11
12// 선택 정렬
13void selection_sort(int arr[], int n) {
14 for (int i = 0; i < n-1; i++) {
15 int min_idx = i;
16 for (int j = i+1; j < n; j++)
17 if (arr[j] < arr[min_idx]) min_idx = j;
18 int t = arr[i]; arr[i] = arr[min_idx]; arr[min_idx] = t;
19 }
20}
21
22// 삽입 정렬
23void insertion_sort(int arr[], int n) {
24 for (int i = 1; i < n; i++) {
25 int key = arr[i], j = i-1;
26 while (j >= 0 && arr[j] > key) { arr[j+1] = arr[j]; j--; }
27 arr[j+1] = key;
28 }
29}
30
31void print_arr(int a[], int n) {
32 for(int i=0;i<n;i++) printf("%d ",a[i]);
33 printf("\n");
34}
35
36int main() {
37 int a[] = {64,25,12,22,11};
38 int b[] = {64,25,12,22,11};
39 int c[] = {64,25,12,22,11};
40
41 bubble_sort(a, 5);
42 printf("버블: "); print_arr(a, 5);
43
44 selection_sort(b, 5);
45 printf("선택: "); print_arr(b, 5);
46
47 insertion_sort(c, 5);
48 printf("삽입: "); print_arr(c, 5);
49 return 0;
50}

5void caesar_encrypt(char *text, int shift) {
6 for (int i = 0; text[i]; i++) {
7 if (isupper(text[i]))
8 text[i] = (text[i] - 'A' + shift) % 26 + 'A';
