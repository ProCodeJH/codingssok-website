[심화 내용] 재귀 하강 파서

[심화 내용] 수식 파싱 — 재귀 하강법
문법 (BNF):
expr → term (("+"|"-") term)*
term → factor (("*"|"/") factor)*
factor → NUMBER | "(" expr ")"

원리:
각 문법 규칙 = 하나의 함수
우선순위가 낮을수록 바깥 함수

예: 2 + 3 * 4
expr() → term() = 2
→ "+" 발견
→ term() → factor()=3, "*", factor()=4 → 12
→ 2 + 12 = 14

왜 중요한가:
컴파일러, 인터프리터, 설정 파서의 기초!
연산자 우선순위를 자연스럽게 처리
[C 코드] 재귀 하강 파서 계산기
1#include <stdio.h>
2#include <stdlib.h>
3#include <ctype.h>
4
5const char *input;
6
7double parse_expr();
8
9double parse_number() {
10 while (isspace(*input)) input++;
11 double val = strtod(input, (char**)&input);
12 return val;
13}
14
15double parse_factor() {
16 while (isspace(*input)) input++;
17 if (*input == '(') {
18 input++; // '(' skip
19 double val = parse_expr();
20 if (*input == ')') input++; // ')' skip
21 return val;
22 }
23 return parse_number();
24}
25
26double parse_term() {
27 double val = parse_factor();
28 while (1) {
29 while (isspace(*input)) input++;
30 if (*input == '*') { input++; val *= parse_factor(); }
31 else if (*input == '/') { input++; val /= parse_factor(); }
32 else break;
33 }
34 return val;
35}
36
37double parse_expr() {
38 double val = parse_term();
39 while (1) {
40 while (isspace(*input)) input++;
41 if (*input == '+') { input++; val += parse_term(); }
42 else if (*input == '-') { input++; val -= parse_term(); }
43 else break;
44 }
45 return val;
46}

[심화 내용] RPN 계산기 (역폴란드 표기법)
RPN (Reverse Polish Notation):
일반: 2 + 3 * 4
RPN: 2 3 4 * +

원리: 스택 기반!
2 → 스택: [2]
3 → 스택: [2, 3]
4 → 스택: [2, 3, 4]
* → pop 3,4 → push 12 → [2, 12]
+ → pop 2,12 → push 14 → [14]
결과: 14

19double eval_rpn(const char *expr) {
20 top = -1;
21 char buf[256];
22 strncpy(buf, expr, sizeof(buf)-1);
23 buf[sizeof(buf)-1] = 0;
24
25 char *tok = strtok(buf, " ");
26 while (tok) {
27 if (isdigit(tok[0]) || (tok[0]=='-' && isdigit(tok[1]))) {
28 push(atof(tok));
29 } else {
30 double b = pop(), a = pop();
31 switch (tok[0]) {
32 case '+': push(a + b); break;
33 case '-': push(a - b); break;
34 case '*': push(a * b); break;
35 case '/': push(b != 0 ? a/b : 0); break;
36 }
37 }
38 tok = strtok(NULL, " ");
39 }
40 return pop();
41}

45 printf("2 3 4 * + = %.2f\n", eval_rpn("2 3 4 * +"));
46 printf("5 1 2 + 4 * + 3 - = %.2f\n", eval_rpn("5 1 2 + 4 * + 3 -"));
47 printf("10 2 / 3 + = %.2f\n", eval_rpn("10 2 / 3 +"));
48 return 0;
49}

RPN
Reverse Polish Notation. 역폴란드 표기법. 연산자가 피연산자 뒤에. 스택으로 평가. HP 계산기 전통!
스택 계산기

17const char* error_msg(CalcError e) {

20 case CALC_DIV_ZERO: return "Division by zero";
21 case CALC_OVERFLOW: return "Overflow";
22 case CALC_INVALID_OP: return "Invalid operator";

24 case CALC_NAN_RESULT: return "NaN result";

29CalcError safe_calc(double a, char op, double b, double *result) {
30 switch (op) {
31 case '+':
32 *result = a + b;
33 if (isinf(*result)) return CALC_OVERFLOW;
34 return CALC_OK;
35 case '-':
36 *result = a - b;
37 if (isinf(*result)) return CALC_OVERFLOW;
38 return CALC_OK;
39 case '*':
40 *result = a * b;
41 if (isinf(*result) && !isinf(a) && !isinf(b))
42 return CALC_OVERFLOW;
43 return CALC_OK;
44 case '/':
45 if (b == 0) return CALC_DIV_ZERO;
46 *result = a / b;
47 return CALC_OK;
48 default:
49 return CALC_INVALID_OP;
50 }
51}
