2차원 좌표 탐색: 피타고라스 삼조

pythagoras.c
1#include <stdio.h>
2
3int main(void) {
4 int n = 30;
5
6 for (int a = 1; a <= n; a++) {
7 for (int b = a; b <= n; b++) { // b>=a: 중복 방지
8 for (int c = b; c <= n; c++) {
9 if (a*a + b*b == c*c) {
10 printf("%d² + %d² = %d² (%d+%d=%d)\n",
11 a, b, c, a*a, b*b, c*c);
12 }
13 }
14 }
15 }
16 return 0;
17}
실행 결과
3² + 4² = 5² (9+16=25) 5² + 12² = 13² (25+144=169) 6² + 8² = 10² (36+64=100) ...
3중 for! O(n³)이지만 b>=a, c>=b로 중복을 줄였어요. 브루트포스(완전 탐색)의 기본!
[핵심] 중첩 횟수와 시간 복잡도
for 1개: O(n) — 1차원 (배열 순회)
for 2개: O(n²) — 2차원 (표, 별찍기, 2D 배열)
for 3개: O(n³) — 3차원 (좌표 탐색, 행렬 곱셈)

n=100일 때:
O(n) = 100번
O(n²) = 10,000번
O(n³) = 1,000,000번!

대회에서 시간 제한은 보통 1초(=약 1억 연산).
n=1000이면 O(n²)까지 가능, O(n³)은 시간 초과!

브루트포스
모든 경우를 다 체크! 중첩 for의 대표적 활용. O(n²) 이상.

[컴퓨팅 사고력] 브루트포스와 최적화

[사고력] 4요소 심화 연결
1. 분해(Decomposition):
구구단 표 = 줄(i) × 칸(j)으로 분해
소수 N개 = 개수 반복(while) × 소수 판별(for)로 분해

2. 패턴 인식(Pattern Recognition):
2차원 구조(표, 좌석, 맵) → 이중 for
N개 찾기(개수 미정) → while + 내부 판별
조합 탐색 → 3중 for (브루트포스)

3. 추상화(Abstraction):
중첩의 핵심: 바깥='큰 단위', 안쪽='작은 단위'
줄/칸, 학급/학생, 단/피승수 — 추상 구조는 동일!

4. 알고리즘 설계(Algorithm Design):
브루트포스(완전 탐색)를 먼저 만들고 최적화!
O(n³)→O(n²)→O(n log n) 단계적 개선이 실력!

실전: 버블 정렬 (Bubble Sort)

중첩 반복문의 대표적 활용 = 정렬 알고리즘!

bubble_sort.c
1#include <stdio.h>
2
3int main(void) {
4 int arr[] = {64, 34, 25, 12, 22, 11, 90};
5 int n = 7;
6
7 // 버블 정렬: 인접한 두 값 비교 → 교환
8 for (int i = 0; i < n-1; i++) {
9 for (int j = 0; j < n-1-i; j++) {
10 if (arr[j] > arr[j+1]) {
11 int temp = arr[j];
12 arr[j] = arr[j+1];
13 arr[j+1] = temp;
14 }
15 }
16 }
17
18 printf("정렬 결과: ");
19 for (int i = 0; i < n; i++)
20 printf("%d ", arr[i]);
21 printf("\n");
22 return 0;
23}
실행 결과
정렬 결과: 11 12 22 25 34 64 90
8줄바깥 for: n-1회 반복 (패스)

10줄앞이 크면 교환!
버블 정렬 = 중첩 for의 클래식! O(n²)이지만 이해하기 쉽고 구현도 쉽습니다.

1#include <stdio.h>
2
3int main(void) {
4 int A[2][3] = {{1,2,3},{4,5,6}};
5 int B[3][2] = {{7,8},{9,10},{11,12}};
6 int C[2][2] = {0};
7
8 // C[i][j] = Σ A[i][k]*B[k][j]
9 for (int i = 0; i < 2; i++)
10 for (int j = 0; j < 2; j++)
11 for (int k = 0; k < 3; k++)
12 C[i][j] += A[i][k] * B[k][j];
13
14 for (int i = 0; i < 2; i++) {
15 for (int j = 0; j < 2; j++)
16 printf("%4d", C[i][j]);
17 printf("\n");
18 }
19 return 0;
20}

sieve.c
1#include <stdio.h>
2#include <string.h>
3
4int main(void) {
5 int n = 100;
6 int is_prime[101];
7 memset(is_prime, 1, sizeof(is_prime));
8 is_prime[0] = is_prime[1] = 0;
9
10 for (int i = 2; i * i <= n; i++) {
11 if (is_prime[i]) {
12 for (int j = i*i; j <= n; j += i)
13 is_prime[j] = 0;
14 }
15 }
16
17 printf("소수: ");
18 for (int i = 2; i <= n; i++)
19 if (is_prime[i]) printf("%d ", i);
20 printf("\n");
21 return 0;
22}

selection_sort.c
1#include <stdio.h>
2
3int main(void) {
4 int arr[] = {64, 25, 12, 22, 11};
5 int n = 5;
6
7 for (int i = 0; i < n-1; i++) {
8 int min_idx = i;
9 for (int j = i+1; j < n; j++) {
10 if (arr[j] < arr[min_idx])
11 min_idx = j;
12 }
13 int temp = arr[i];
14 arr[i] = arr[min_idx];
15 arr[min_idx] = temp;
16 }
17
18 for (int i = 0; i < n; i++)
19 printf("%d ", arr[i]);
20 printf("\n");
21 return 0;
22}
