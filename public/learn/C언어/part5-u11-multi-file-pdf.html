왜 파일을 나눠야 할까?

코드가 500줄, 1000줄이 되면 한 파일에서 관리하기 어려워져. 기능별로 파일을 나누면 팀원과 협업도 쉽고, 수정할 부분만 다시 컴파일하면 돼서 빠르기도 해!
main.c, math_utils.h, math_utils.c로 나눈 파일 구조

헤더 파일(.h)과 소스 파일(.c)

파일 분리 원칙
헤더 파일(.h): 함수 선언(프로토타입), 구조체 정의, 매크로, extern 변수 선언
소스 파일(.c): 함수 정의(구현), 전역 변수 정의

선언은 '이런 게 있다'고 알려주는 것, 정의는 '이렇게 동작한다'고 만드는 것.
math_utils.h
1#ifndef MATH_UTILS_H
2#define MATH_UTILS_H
3
4/* 함수 선언 (프로토타입) */
5int add(int a, int b);
6int multiply(int a, int b);
7
8/* extern 변수 선언 */
9extern int globalCount;
10
11#endif
math_utils.c
1#include "math_utils.h"
2
3/* 전역 변수 정의 */
4int globalCount = 0;
5
6/* 함수 정의 */
7int add(int a, int b) {
8 globalCount++;
9 return a + b;
10}
11
12int multiply(int a, int b) {
13 globalCount++;
14 return a * b;
15}
main.c
1#include <stdio.h>
2#include "math_utils.h"
3
4int main(void) {
5 printf("%d + %d = %d\n", 3, 5, add(3, 5));
6 printf("%d x %d = %d\n", 4, 6, multiply(4, 6));
7 printf("함수 호출 횟수: %d\n", globalCount);
8 return 0;
9}
실행 결과
3 + 5 = 8 4 x 6 = 24 함수 호출 횟수: 2
math_utils.h에는 '이런 함수가 있다'는 선언만 써. 실제 코드는 math_utils.c에 써. main.c에서 헤더만 include하면 함수를 쓸 수 있어!

extern 키워드

extern은 '이 변수가 다른 파일에 정의되어 있다'고 알려주는 키워드예요. 선언만 하고 메모리를 할당하지 않아요.

extern 선언은 다른 파일의 변수를 가리킨다
extern이 없으면 각 파일이 같은 이름의 별개 변수를 만들어서 값이 따로 놀아. extern으로 '같은 변수'라고 연결해 줘야 해!

컴파일과 링킹 과정

컴파일(각 .c -> .o) + 링킹(.o들 -> 실행 파일)
빌드 과정 3단계
1. 전처리: #include, #define 등 처리
2. 컴파일: 각 .c 파일을 오브젝트 파일(.o)로 변환
3. 링킹: 모든 .o 파일을 합쳐서 하나의 실행 파일 생성

명령어: gcc main.c math_utils.c -o program
'undefined reference' 에러가 나면 링킹 단계 문제야! 함수를 선언만 하고 정의를 안 했거나, .c 파일을 컴파일 목록에 안 넣었을 가능성이 높아.

21// === stack.c ===
22// #include "stack.h"
23
24void stack_init(Stack *s) { s->top = -1; }
25
26int stack_push(Stack *s, int val) {
27 if (s->top >= STACK_MAX-1) return -1;
28 s->data[++s->top] = val;
29 return 0;
30}
31
32int stack_pop(Stack *s, int *val) {
33 if (s->top < 0) return -1;
34 *val = s->data[s->top--];
35 return 0;
36}
37
38int stack_peek(const Stack *s, int *val) {
39 if (s->top < 0) return -1;
40 *val = s->data[s->top];
41 return 0;
42}
43

45int stack_size(const Stack *s) { return s->top + 1; }

47// === main.c ===
48#include <stdio.h>
49// #include "stack.h"
50
51int main() {

53 stack_init(&s);
54

58 int val;
59 stack_peek(&s, &val);

1// === calc.h ===
2#ifndef CALC_H
3#define CALC_H
4
5typedef enum { OP_ADD, OP_SUB, OP_MUL, OP_DIV, OP_MOD, OP_POW } CalcOp;
6typedef struct { CalcOp op; double a, b; double result; int error; } CalcExpr;
7
8CalcExpr calc_eval(CalcOp op, double a, double b);
9const char* calc_op_str(CalcOp op);
10
11// 히스토리
12#define CALC_HISTORY_SIZE 50
13void calc_history_add(CalcExpr expr);
14void calc_history_print(int n);
15int calc_history_count(void);
16
17#endif
18
19// === calc.c ===
20#include <stdio.h>
21#include <math.h>
22// #include "calc.h"
23
24static CalcExpr history[CALC_HISTORY_SIZE];
25static int hist_count = 0;
26
27const char* calc_op_str(CalcOp op) {
28 static const char *ops[] = {"+","-","*","/","%","^"};
29 return ops[op];
30}
31
32CalcExpr calc_eval(CalcOp op, double a, double b) {
33 CalcExpr e = {op, a, b, 0, 0};
34 switch(op) {
35 case OP_ADD: e.result = a + b; break;
36 case OP_SUB: e.result = a - b; break;
37 case OP_MUL: e.result = a * b; break;
38 case OP_DIV:
39 if (b == 0) { e.error = 1; break; }
40 e.result = a / b; break;
41 case OP_MOD:
42 if (b == 0) { e.error = 1; break; }
43 e.result = fmod(a, b); break;
44 case OP_POW: e.result = pow(a, b); break;
45 }
46 if (!e.error) calc_history_add(e);
47 return e;
48}
49
50void calc_history_add(CalcExpr expr) {
51 if (hist_count < CALC_HISTORY_SIZE)
52 history[hist_count++] = expr;
53}
54
55void calc_history_print(int n) {
56 int start = hist_count > n ? hist_count - n : 0;
57 for (int i = start; i < hist_count; i++)
58 printf(" [%d] %.4g %s %.4g = %.4g\n", i+1,
59 history[i].a, calc_op_str(history[i].op),
60 history[i].b, history[i].result);
61}
62
63int calc_history_count(void) { return hist_count; }
64
65// === main.c ===
66// #include "calc.h"
67
68int main() {
69 CalcExpr e;
70 e = calc_eval(OP_ADD, 10, 20); printf("%.0f + %.0f = %.0f\n", e.a, e.b, e.result);
71 e = calc_eval(OP_MUL, 3.5, 4); printf("%.1f * %.0f = %.1f\n", e.a, e.b, e.result);
72 e = calc_eval(OP_POW, 2, 10); printf("%.0f ^ %.0f = %.0f\n", e.a, e.b, e.result);
73 e = calc_eval(OP_DIV, 100, 3); printf("%.0f / %.0f = %.4f\n", e.a, e.b, e.result);
74 e = calc_eval(OP_DIV, 1, 0);
75 printf("1 / 0 = %s\n", e.error ? "Error!" : "OK");
76
77 printf("\n=== History (last 5) ===\n");
78 calc_history_print(5);
79 printf("Total: %d calculations\n", calc_history_count());
80 return 0;
81}

static 함수/변수 = 파일 내부에만 보임

→ 컴파일러 최적화 가능 (인라인 등)

23
24typedef struct Entry {
25 char *key;
26 int value;
27 struct Entry *next;
28} Entry;
29
30struct HashMap {
31 Entry **buckets;
32 int capacity;
33 int count;
34};
35
36static unsigned int hash(const char *s, int cap) {
37 unsigned int h = 5381;
38 while (*s) h = h * 33 + *s++;
39 return h % cap;
40}
41
42HashMap* hm_create(int capacity) {
43 HashMap *m = (HashMap*)malloc(sizeof(HashMap));
44 m->capacity = capacity;
45 m->count = 0;
46 m->buckets = (Entry**)calloc(capacity, sizeof(Entry*));
47 return m;
48}
49
50void hm_destroy(HashMap *m) {
51 for (int i = 0; i < m->capacity; i++) {
52 Entry *e = m->buckets[i];
53 while (e) { Entry *n = e->next; free(e->key); free(e); e = n; }
54 }
55 free(m->buckets);
56 free(m);
57}
58
59int hm_set(HashMap *m, const char *key, int value) {
60 unsigned int idx = hash(key, m->capacity);
61 Entry *e = m->buckets[idx];
62 while (e) {
63 if (strcmp(e->key, key) == 0) { e->value = value; return 0; }
64 e = e->next;
65 }
66 Entry *n = (Entry*)malloc(sizeof(Entry));
67 n->key = strdup(key); n->value = value;
68 n->next = m->buckets[idx]; m->buckets[idx] = n;
69 m->count++;
70 return 1;
71}
72
73int hm_get(HashMap *m, const char *key, int *value) {
74 Entry *e = m->buckets[hash(key, m->capacity)];
75 while (e) { if (strcmp(e->key, key)==0) { *value = e->value; return 1; } e = e->next; }
76 return 0;
77}
78
79int hm_size(const HashMap *m) { return m->count; }
80
81void hm_print(const HashMap *m) {
82 for (int i = 0; i < m->capacity; i++) {
83 Entry *e = m->buckets[i];
84 while (e) { printf(" %s=%d\n", e->key, e->value); e = e->next; }
85 }
86}

90int main() {
91 HashMap *m = hm_create(8);
92 hm_set(m, "apple", 3); hm_set(m, "banana", 5);
93 hm_set(m, "cherry", 2); hm_set(m, "date", 8);
94
95 printf("Size: %d\n", hm_size(m));
96 int v;
97 if (hm_get(m, "cherry", &v)) printf("cherry=%d\n", v);
98 if (!hm_get(m, "grape", &v)) printf("grape=not found\n");
99
100 printf("\nAll entries:\n");
101 hm_print(m);
102 hm_destroy(m);
103 return 0;

[심화 내용] API 설계 원칙
불투명 타입 (Opaque Type):
헤더: typedef struct HashMap HashMap;
사용자: HashMap *m; (포인터로만 접근)
구조체 내부를 모름 → 구현 변경 자유!

API 안정성:
헤더가 바뀌지 않으면 → ABI 호환
내부 구조 변경 → 재컴파일 불필요!
→ 라이브러리 업데이트 시 바이너리 호환

PIMPL 패턴 (C++):
class Foo { struct Impl; Impl *pImpl; };
→ 컴파일 방화벽 (빌드 시간 감소)
→ 불투명 타입의 C++ 버전

참조 카운팅:
hm_retain(m) / hm_release(m)
→ 여러 곳에서 공유 시 메모리 관리
→ COM, Objective-C, Swift ARC
PIMPL
Pointer to Implementation. 구현 숨김 패턴. 헤더에 포인터만. 빌드 의존성 차단! C++의 컴파일 방화벽.
API vs ABI
API=소스 호환. ABI=바이너리 호환. API 유지+ABI 깨짐 → 재컴파일 필요. 불투명 타입은 ABI도 보호!
불투명 타입 = C의 캡슐화! 사용자는 HashMap*만 알고, 내부는 .c에 숨김!
구현 변경해도 헤더 불변 → 재컴파일 불필요! 라이브러리 API 설계의 핵심 원칙!
💡 다중 파일 설계 원칙
✅ .h: 선언만 (함수 시그니처, typedef, 상수)
✅ .c: 구현 (함수 본체, static 변수)
✅ static: 파일 내부에만 보이는 함수/변수
✅ extern: 다른 파일의 전역 변수 참조
✅ 순환 의존성 금지 (A→B→A 안 됨!)
✅ 인터페이스 최소화: 필요한 것만 공개
✅ 불투명 타입으로 구현 세부사항 숨김
translation unit
번역 단위. .c 파일+#include된 헤더 = 하나의 번역 단위. 컴파일러가 독립적으로 컴파일!
linkage
링크 속성. external(전역), internal(static), none(로컬). extern int x; = 외부 링크 참조.
incremental build
증분 빌드. 변경된 .c만 재컴파일. Make/CMake가 자동 추적! 대형 프로젝트에서 빌드 시간 90%↓.
다중 파일 = 대형 프로젝트의 기본! 1000줄 넘으면 반드시 파일 분리. 관리+재사용+협업!
object file
오브젝트 파일. .o/.obj. 기계어+심볼 테이블+재배치 정보. 링커가 합쳐서 실행파일!
name mangling
이름 맹글링. C++에서 함수 오버로딩 구분. C는 없음! extern "C"로 C 링크 사용.
visibility
심볼 가시성. __attribute__((visibility("hidden"))). 공유 라이브러리에서 내부 심볼 숨김!
다중 파일 마스터! Stack·Utils·Calculator·HashMap. 모듈화로 대형 프로젝트 관리!
weak symbol
약한 심볼. __attribute__((weak)). 다른 곳에서 정의하면 그걸 사용. 기본 구현 제공에 활용!
ELF format
Executable and Linkable Format. Linux 실행파일 형식. .text(코드) .data(데이터) .bss(0초기화)
dynamic linking
동적 링킹. 실행 시 .so/.dll 로드. 메모리 공유, 업데이트 용이. LD_LIBRARY_PATH로 경로 지정.
ELF = Linux 바이너리의 뼈대! .text+.data+.bss+심볼 테이블. readelf로 분석!
헤더 파일(.h)
함수 선언, 매크로, 구조체 정의 등을 담는 파일. 여러 .c 파일에서 공유한다.
소스 파일(.c)
함수 정의(구현)를 담는 파일. 컴파일러가 직접 처리한다.
extern
다른 파일에 정의된 변수를 현재 파일에서 참조하겠다는 선언.
오브젝트 파일(.o)
컴파일된 기계어 코드. 아직 다른 파일과 연결되지 않은 상태.
링커(linker)
여러 오브젝트 파일과 라이브러리를 합쳐 실행 파일을 만드는 프로그램.
