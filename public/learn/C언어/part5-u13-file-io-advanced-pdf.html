파일 안에서 위치 이동하기

파일을 읽을 때 항상 처음부터 끝까지 읽어야 할까? 아니야! fseek으로 원하는 위치로 점프할 수 있어. 음악 플레이어에서 원하는 부분으로 건너뛰는 것과 비슷해!
fseek으로 파일 안에서 원하는 위치로 이동할 수 있다
fseek_ftell.c
1#include <stdio.h>
2
3int main(void) {
4 FILE *fp = fopen("data.txt", "w");
5 if (fp == NULL) return 1;
6 fprintf(fp, "ABCDEFGHIJ");
7 fclose(fp);
8
9 fp = fopen("data.txt", "r");
10 if (fp == NULL) return 1;
11
12 /* 파일 끝으로 이동하여 크기 구하기 */
13 fseek(fp, 0, SEEK_END);
14 long size = ftell(fp);
15 printf("파일 크기: %ld 바이트\n", size);
16
17 /* 5번째 위치로 이동 */
18 fseek(fp, 5, SEEK_SET);
19 char ch = fgetc(fp);
20 printf("5번째 위치 문자: %c\n", ch);
21
22 fclose(fp);
23 return 0;
24}
실행 결과
파일 크기: 10 바이트 5번째 위치 문자: F
13줄fseek(fp, 0, SEEK_END) — 파일 끝에서 0바이트 위치, 즉 맨 끝으로 이동.
14줄ftell(fp) — 현재 파일 포인터의 위치(바이트 수)를 반환.
18줄fseek(fp, 5, SEEK_SET) — 파일 처음(SEEK_SET)에서 5바이트 위치로 이동.
fseek의 기준점
SEEK_SET — 파일 처음 기준
SEEK_CUR — 현재 위치 기준
SEEK_END — 파일 끝 기준

바이너리 파일: fread와 fwrite

텍스트 파일은 사람이 읽을 수 있지만, 바이너리 파일은 컴퓨터가 직접 이해하는 형태야. 구조체를 통째로 저장할 때 바이너리 파일이 훨씬 효율적이야!
텍스트: 사람이 읽는 문자열 / 바이너리: 메모리 그대로 저장

구조체를 파일에 저장하고 불러오기

struct_file.c
1#include <stdio.h>
2#include <string.h>
3
4typedef struct {
5 char name[20];
6 int age;
7 double score;
8} Student;
9
10int main(void) {
11 /* 저장 */
12 Student s1 = {"Kim", 17, 92.5};
13 Student s2 = {"Lee", 18, 88.0};
14
15 FILE *fp = fopen("students.dat", "wb");
16 if (fp == NULL) return 1;
17 fwrite(&s1, sizeof(Student), 1, fp);
18 fwrite(&s2, sizeof(Student), 1, fp);
19 fclose(fp);
20
21 /* 불러오기 */
22 Student buf;
23 fp = fopen("students.dat", "rb");
24 if (fp == NULL) return 1;
25
26 printf("=== 저장된 학생 정보 ===\n");
27 while (fread(&buf, sizeof(Student), 1, fp) == 1) {
28 printf("이름: %s, 나이: %d, 점수: %.1f\n",
29 buf.name, buf.age, buf.score);
30 }
31
32 fclose(fp);
33 return 0;
34}
실행 결과
=== 저장된 학생 정보 === 이름: Kim, 나이: 17, 점수: 92.5 이름: Lee, 나이: 18, 점수: 88.0
15줄fopen("students.dat", "wb") — wb는 바이너리 쓰기 모드.
17줄fwrite(&s1, sizeof(Student), 1, fp) — s1 구조체를 통째로 파일에 저장.
27줄fread(&buf, sizeof(Student), 1, fp) — 파일에서 Student 크기만큼 읽어서 buf에 저장.
fwrite/fread의 인자 순서를 외워! (데이터 주소, 한 개 크기, 개수, 파일포인터) 순서야. fread는 성공적으로 읽은 개수를 반환해.
구조체 멤버가 파일에 순서대로 저장되는 모습
바이너리 파일은 메모장으로 열면 글자가 깨져 보여. 그게 정상이야! 텍스트 파일과 바이너리 파일은 용도가 달라.
[주의] 바이너리 파일의 이식성
바이너리 파일은 컴퓨터마다 구조체 크기, 패딩, 바이트 순서가 다를 수 있어요. 다른 컴퓨터에서 읽으려면 텍스트 형식(CSV, JSON)이 안전해요.

4// INI 파일 파서
5typedef struct { char key[30]; char value[50]; } IniEntry;
6typedef struct { char section[30]; IniEntry entries[20]; int count; } IniSection;
7
8int parse_ini(const char *content, IniSection *sections, int max_sec) {
9 int nsec = -1;
10 char line[256];
11 const char *p = content;
12 while (*p) {
13 int i = 0;
14 while (*p && *p != '\n' && i < 255) line[i++] = *p++;
15 line[i] = 0;
16 if (*p == '\n') p++;
17
18 // 빈 줄, 주석 건너뜀
19 if (line[0] == 0 || line[0] == ';' || line[0] == '#') continue;
20
21 // 섹션 [name]
22 if (line[0] == '[') {
23 nsec++;
24 if (nsec >= max_sec) break;
25 char *end = strchr(line, ']');
26 if (end) { *end = 0; strncpy(sections[nsec].section, line+1, 29); }
27 sections[nsec].count = 0;
28 continue;
29 }
30
31 // key=value
32 if (nsec >= 0) {
33 char *eq = strchr(line, '=');
34 if (eq) {
35 *eq = 0;
36 int idx = sections[nsec].count++;
37 strncpy(sections[nsec].entries[idx].key, line, 29);
38 strncpy(sections[nsec].entries[idx].value, eq+1, 49);
39 }
40 }
41 }
42 return nsec + 1;
43}

19int load_records(const char *fn, Record *arr, int max) {
20 FILE *fp = fopen(fn, "rb");
21 if (!fp) return -1;
22 int n;
23 fread(&n, sizeof(int), 1, fp);
24 if (n > max) n = max;
25 fread(arr, sizeof(Record), n, fp);
26 fclose(fp);
27 return n;
28}

30// 랜덤 접근: n번째 레코드 직접 읽기
31int read_record(const char *fn, int index, Record *r) {
32 FILE *fp = fopen(fn, "rb");
33 if (!fp) return -1;
34 int n;
35 fread(&n, sizeof(int), 1, fp);
36 if (index >= n) { fclose(fp); return -1; }
37 fseek(fp, sizeof(int) + index * sizeof(Record), SEEK_SET);
38 fread(r, sizeof(Record), 1, fp);
39 fclose(fp);
40 return 0;
41}

14int csv_parse_line(const char *line, char fields[][100], int max) {
15 int count = 0;
16 const char *p = line;
17 while (*p && count < max) {
18 int i = 0;
19 int quoted = 0;
20 if (*p == '"'){ quoted = 1; p++; }
21 while (*p) {
22 if (quoted) {
23 if (*p == '"') { p++; if (*p != '"') break; }
24 } else {
25 if (*p == ',' || *p == '\n' || *p == '\r') break;
26 }
27 if (i < 99) fields[count][i++] = *p;
28 p++;
29 }
30 fields[count][i] = 0;
31 count++;
32 if (*p == ',') p++;
33 }
34 return count;
35}
36
37int csv_load(CsvTable *t, const char *content) {
38 char line[1024];
39 const char *p = content;
40 int row = -1;
41 while (*p) {
42 int i = 0;
43 while (*p && *p != '\n' && i < 1023) line[i++] = *p++;
44 line[i] = 0;
45 if (*p == '\n') p++;
46 if (i == 0) continue;
47
48 if (row < 0) {
49 t->ncols = csv_parse_line(line, t->headers, MAX_COLS);
50 row = 0;
51 } else {
52 csv_parse_line(line, t->data[row], MAX_COLS);
53 row++;
54 }
55 }
56 t->nrows = row;
57 return row;
58}

3#include <stdlib.h>

5#define MAX_ROWS 50
6#define MAX_COLS 8
7#define CELL_LEN 30
8
9typedef struct {
10 char name[20];
11 char cols[MAX_COLS][20];
12 char data[MAX_ROWS][MAX_COLS][CELL_LEN];
13 int ncols, nrows;
14} Table;
15
16void table_init(Table *t, const char *name, const char **cols, int ncols) {
17 strncpy(t->name, name, 19);
18 t->ncols = ncols; t->nrows = 0;
19 for (int i = 0; i < ncols; i++) strncpy(t->cols[i], cols[i], 19);
20}
21
22int table_insert(Table *t, const char **values) {
23 if (t->nrows >= MAX_ROWS) return -1;
24 for (int i = 0; i < t->ncols; i++)
25 strncpy(t->data[t->nrows][i], values[i], CELL_LEN-1);
26 return t->nrows++;
27}
28
29void table_select_where(Table *t, int col, const char *val) {
30 printf("SELECT * FROM %s WHERE %s='%s'\n", t->name, t->cols[col], val);
31 for (int i = 0; i < t->nrows; i++)
32 if (strcmp(t->data[i][col], val) == 0) {
33 printf(" ");
34 for (int j = 0; j < t->ncols; j++) printf("%-12s", t->data[i][j]);
35 printf("\n");
36 }
37}
38
39void table_order_by(Table *t, int col, int asc) {
40 for (int i = 0; i < t->nrows-1; i++)
41 for (int j = 0; j < t->nrows-1-i; j++) {
42 int cmp = strcmp(t->data[j][col], t->data[j+1][col]);
43 if (asc ? cmp > 0 : cmp < 0) {
44 for (int k = 0; k < t->ncols; k++) {
45 char tmp[CELL_LEN];
46 strcpy(tmp, t->data[j][k]);
47 strcpy(t->data[j][k], t->data[j+1][k]);
48 strcpy(t->data[j+1][k], tmp);
49 }
50 }
51 }
52}
53
54void table_print(Table *t) {
55 printf("\n=== %s (%d rows) ===\n", t->name, t->nrows);
56 for (int j = 0; j < t->ncols; j++) printf("%-12s", t->cols[j]);
57 printf("\n");
58 for (int j = 0; j < t->ncols; j++) printf("%-12s", "----------");
59 printf("\n");
60 for (int i = 0; i < t->nrows; i++) {
61 for (int j = 0; j < t->ncols; j++) printf("%-12s", t->data[i][j]);
62 printf("\n");
63 }
64}

[심화 내용] 관계형 데이터베이스
관계형 모델 (Codd, 1970):
테이블 = 릴레이션
행 = 튜플, 열 = 어트리뷰트
SQL = 관계 대수의 구현

핵심 연산:
SELECT: σ (행 필터)
PROJECT: π (열 선택)
JOIN: ⋈ (테이블 결합)
→ 이 3개로 거의 모든 쿼리 표현!

인덱스 구조:
B+ 트리: 범위 검색에 최적 (기본!)
해시 인덱스: 등값 검색에 최적
비트맵 인덱스: 카디널리티 낮은 열

트랜잭션 ACID:
Atomicity, Consistency, Isolation, Durability
WAL + 2PL로 구현
ACID
원자성+일관성+격리성+지속성. 트랜잭션의 4대 속성. 은행 이체가 반만 되면 안 되니까! WAL로 보장.
미니 DB 엔진! SELECT WHERE, ORDER BY를 구조체 배열로 구현. SQL의 원리가 이것!
💡 파일 I/O 고급 체크리스트
✅ 바이너리: fwrite/fread + sizeof
✅ 랜덤 접근: fseek + ftell (고정 크기 레코드)
✅ 설정 파일: INI, JSON, TOML 중 선택
✅ CSV: RFC 4180 인용부호 처리
✅ 직렬화: 버전 필드 포함 (하위 호환)
✅ 에러 처리: ferror, feof 확인
✅ 대용량: 청크 단위 처리 (전체 로드 금지)
ferror
파일 에러 확인. if(ferror(fp)) 읽기/쓰기 오류 감지. feof()와 함께 루프 종료 조건!
tmpfile
임시 파일. FILE *tmp = tmpfile(); 자동 삭제. 이름 없음. 안전한 임시 데이터 처리!
바이너리+텍스트+CSV+INI+JSON! 파일 포맷을 이해하면 어떤 데이터든 다룰 수 있어!
file locking
파일 잠금. flock()/fcntl(). 동시 접근 제어. 공유 잠금(읽기)+배타 잠금(쓰기). DB의 기본!
파일 I/O 고급! 바이너리·CSV·INI·미니DB. 데이터 영속성의 모든 것!
fsync
fsync(). 버퍼를 디스크에 강제 기록. 데이터베이스의 내구성(Durability) 보장! 느리지만 안전.
sparse file
희소 파일. 0으로 채워진 영역은 디스크 공간 미사용. fseek으로 큰 오프셋 점프!
fsync = 데이터가 정말 디스크에 갔는지 확인! DB의 ACID Durability 보장의 핵심!
fseek
파일 포인터의 위치를 이동하는 함수.
ftell
현재 파일 포인터의 위치(바이트 오프셋)를 반환하는 함수.
fread
파일에서 바이너리 데이터를 읽는 함수.
fwrite
바이너리 데이터를 파일에 쓰는 함수.
바이너리 파일
데이터를 메모리에 저장된 형태 그대로 파일에 저장한 것.
텍스트 파일
데이터를 사람이 읽을 수 있는 문자열 형태로 저장한 것.
이번 유닛 핵심 정리
1
fseek/ftell로 파일 포인터를 이동하고 위치를 확인한다
2
fread/fwrite로 바이너리 데이터를 읽고 쓴다
3
구조체를 fwrite로 통째로 저장하고 fread로 복원한다
4
바이너리 모드는 'b'를 붙인다: rb, wb
5
텍스트는 이식성, 바이너리는 효율성이 장점이다
