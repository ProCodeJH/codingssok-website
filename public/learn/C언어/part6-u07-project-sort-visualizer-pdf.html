프로젝트 목표

지금까지 배운 세 가지 정렬을 하나의 프로그램에서 모두 실행하고, 누가 비교를 적게 하는지, 누가 교환을 적게 하는지 직접 실험해 보자!

이 프로젝트에서는 같은 랜덤 배열에 버블 정렬, 선택 정렬, 삽입 정렬을 각각 적용하고, 비교 횟수와 교환/이동 횟수를 측정해요. 결과를 비교하면 각 알고리즘의 특성을 눈으로 확인할 수 있어요.

22SortResult bubbleSort(int arr[], int n) {
23 SortResult r = {0, 0};
24 for (int i = 0; i < n - 1; i++) {
25 int swapped = 0;
26 for (int j = 0; j < n - 1 - i; j++) {
27 r.compares++;
28 if (arr[j] > arr[j + 1]) {
29 int temp = arr[j];
30 arr[j] = arr[j + 1];
31 arr[j + 1] = temp;
32 r.swaps++;
33 swapped = 1;
34 }
35 }
36 if (!swapped) break;
37 }
38 return r;
39}

41SortResult selectionSort(int arr[], int n) {
42 SortResult r = {0, 0};
43 for (int i = 0; i < n - 1; i++) {
44 int minIdx = i;
45 for (int j = i + 1; j < n; j++) {
46 r.compares++;
47 if (arr[j] < arr[minIdx]) {
48 minIdx = j;
49 }
50 }
51 if (minIdx != i) {
52 int temp = arr[i];
53 arr[i] = arr[minIdx];
54 arr[minIdx] = temp;
55 r.swaps++;
56 }
57 }
58 return r;
59}
60
61SortResult insertionSort(int arr[], int n) {
62 SortResult r = {0, 0};
63 for (int i = 1; i < n; i++) {
64 int key = arr[i];
65 int j = i - 1;
66 while (j >= 0) {
67 r.compares++;
68 if (arr[j] > key) {
69 arr[j + 1] = arr[j];
70 r.swaps++;
71 j--;
72 } else {
73 break;
74 }
75 }
76 arr[j + 1] = key;
77 }
78 return r;
79}

성능 비교 분석

n=10 랜덤 데이터 실험 결과 예시
실험 결과에서 배우는 것
비교 횟수: 버블과 선택은 거의 같음(n(n-1)/2). 삽입은 데이터에 따라 달라짐.
교환/이동 횟수: 선택이 가장 적음(최대 n-1). 버블이 가장 많음.
데이터 특성: 거의 정렬된 데이터에는 삽입 정렬이 압도적으로 빠름!

같은 O(n^2)이라도 실제 성능은 다를 수 있다는 것을 실험으로 확인!
이론으로만 배우는 것과 직접 실험해서 숫자로 확인하는 건 완전 다른 경험이야! SIZE를 바꿔가며 실험해 봐!

1#include <stdio.h>
2
3void print_indent(int depth) { for(int i=0;i<depth;i++) printf(" "); }
4
5void merge_v(int *a, int l, int m, int r, int depth) {
6 int L[50], R[50], n1=m-l+1, n2=r-m;
7 for(int i=0;i<n1;i++) L[i]=a[l+i];
8 for(int i=0;i<n2;i++) R[i]=a[m+1+i];
9 print_indent(depth); printf("merge [");
10 for(int i=0;i<n1;i++) printf("%d%s",L[i],i<n1-1?",":"");
11 printf("] + [");
12 for(int i=0;i<n2;i++) printf("%d%s",R[i],i<n2-1?",":"");
13 printf("]\n");
14 int i=0,j=0,k=l;
15 while(i<n1&&j<n2) a[k++]=(L[i]<=R[j])?L[i++]:R[j++];
16 while(i<n1) a[k++]=L[i++];
17 while(j<n2) a[k++]=R[j++];
18 print_indent(depth); printf(" → [");
19 for(int x=l;x<=r;x++) printf("%d%s",a[x],x<r?",":"");
20 printf("]\n");
21}

25}

[C 코드] 정렬 시각화 — 비교 카운터
1#include <stdio.h>
2
3typedef struct { long comps; long swaps; } SortStats;
4
5SortStats bubble_stats(int *a, int n) {
6 SortStats s = {0,0};
7 for(int i=0;i<n-1;i++){int sw=0;
8 for(int j=0;j<n-1-i;j++){s.comps++;if(a[j]>a[j+1]){int t=a[j];a[j]=a[j+1];a[j+1]=t;s.swaps++;sw=1;}}
9 if(!sw)break;}
10 return s;
11}
12
13SortStats selection_stats(int *a, int n) {
14 SortStats s = {0,0};
15 for(int i=0;i<n-1;i++){int m=i;
16 for(int j=i+1;j<n;j++){s.comps++;if(a[j]<a[m])m=j;}
17 if(m!=i){int t=a[i];a[i]=a[m];a[m]=t;s.swaps++;}}
18 return s;
19}
20
21SortStats insertion_stats(int *a, int n) {
22 SortStats s = {0,0};
23 for(int i=1;i<n;i++){int k=a[i],j=i-1;
24 while(j>=0){s.comps++;if(a[j]>k){a[j+1]=a[j];j--;s.swaps++;}else break;}
25 a[j+1]=k;}
26 return s;
27}

31 int n = 10;
32 int a[10],b[10],c[10];
33 for(int i=0;i<n;i++) a[i]=b[i]=c[i]=orig[i];
34
35 SortStats sb = bubble_stats(a,n);
36 SortStats ss = selection_stats(b,n);
37 SortStats si = insertion_stats(c,n);
38

40 printf("%-12s %8ld %8ld\n", "Bubble", sb.comps, sb.swaps);
41 printf("%-12s %8ld %8ld\n", "Selection", ss.comps, ss.swaps);
42 printf("%-12s %8ld %8ld\n", "Insertion", si.comps, si.swaps);
43 return 0;
44}

[심화 내용] 정렬 알고리즘 총정리
O(n²) 정렬:
버블: 인접 교환, 안정, 느림
선택: 최소값→앞, 교환 적음, 불안정
삽입: 카드 끼워넣기, 거의정렬에 O(n), 안정

O(n log n) 정렬:
병합: 분할정복, 안정, 추가공간 O(n)
퀵: 파티션, 캐시 최적, 평균 O(n log n)
힙: 최대힙, in-place, 항상 O(n log n)

O(n) 정렬 (비비교):
계수: 범위 제한 정수, 안정
기수: 자릿수별, 안정
버킷: 균등 분포, 평균 O(n)

하이브리드:
Timsort: 삽입+병합 (Python/Java)
Introsort: 퀵+힙+삽입 (C++ std::sort)
in-place
제자리 정렬. 추가 공간 O(1). 힙, 퀵, 삽입=in-place. 병합=O(n) 추가 필요.
정렬 시각화로 파티션 과정이 한눈에! pivot 기준 좌우 분리가 눈앞에서!
벤치마크(benchmark)
여러 알고리즘의 성능을 같은 조건에서 비교 측정하는 것.
카운팅(counting)
비교, 교환 등 특정 연산의 횟수를 세는 것.
memcpy
메모리 블록을 복사하는 C 표준 함수. string.h에 선언.
srand / rand
난수 생성 함수. srand로 시드 설정, rand로 난수 생성.
역쌍(inversion)
배열에서 앞의 원소가 뒤의 원소보다 큰 쌍. 버블 정렬의 교환 횟수와 같다.
