줄 서기의 규칙

편의점 계산대에서 줄을 서 본 적 있지? 먼저 온 사람이 먼저 계산하고 나가. 새로 온 사람은 뒤에 서고. 이게 바로 큐야!

큐(Queue)는 줄 서기와 같은 자료구조예요. 먼저 들어온 데이터가 먼저 나가는 FIFO(First In, First Out) 구조입니다. 스택과 정반대죠!

큐: 먼저 넣은 것이 먼저 나온다 (FIFO)

큐의 두 가지 핵심 연산

enqueue: rear쪽에 추가 / dequeue: front쪽에서 제거
큐의 핵심 연산
enqueue(x) -- 큐의 뒤쪽(rear)에 데이터 x를 추가한다.
dequeue() -- 큐의 앞쪽(front)에서 데이터를 꺼내서 반환한다.
front / rear -- 큐의 앞과 뒤를 가리키는 변수.

배열로 큐 구현하기

queue_array.c
1#include <stdio.h>
2#define MAX 100
3
4int queue[MAX];
5int front = 0, rear = 0;
6
7void enqueue(int x) {
8 if (rear >= MAX) {
9 printf("Queue Full!\n");
10 return;
11 }
12 queue[rear++] = x;
13}
14
15int dequeue(void) {
16 if (front >= rear) {
17 printf("Queue Empty!\n");
18 return -1;
19 }
20 return queue[front++];
21}
22
23int main(void) {
24 enqueue(10);
25 enqueue(20);
26 enqueue(30);
27 printf("%d\n", dequeue());
28 printf("%d\n", dequeue());
29 enqueue(40);
30 printf("%d\n", dequeue());
31 return 0;
32}
실행 결과
10 20 30
5줄front = 0, rear = 0 -- front와 rear가 같으면 큐가 비어 있다는 뜻이에요.
12줄queue[rear++] = x -- rear 위치에 데이터를 넣고 rear를 1 증가시켜요.
20줄queue[front++] -- front 위치의 값을 반환하고 front를 1 증가시켜요.
그런데 이 방식에는 문제가 있어! dequeue를 하면 front가 앞으로 이동하는데, 그 앞쪽 공간은 다시 쓸 수 없어. 배열 공간이 낭비되지!

원형 큐로 공간 낭비 해결

원형 큐(Circular Queue)는 배열의 끝에 도달하면 다시 앞으로 돌아가는 구조예요. (rear + 1) % SIZE로 계산하면 자연스럽게 원형이 됩니다.

원형 큐: (rear+1) % SIZE로 공간을 재활용
circular_queue.c
1#include <stdio.h>
2#define SIZE 5
3
4int queue[SIZE];
5int front = 0, rear = 0, count = 0;
6
7void enqueue(int x) {
8 if (count >= SIZE) {
9 printf("Queue Full!\n");
10 return;
11 }
12 queue[rear] = x;
13 rear = (rear + 1) % SIZE;
14 count++;
15}
16
17int dequeue(void) {
18 if (count <= 0) {
19 printf("Queue Empty!\n");
20 return -1;
21 }
22 int val = queue[front];
23 front = (front + 1) % SIZE;
24 count--;
25 return val;
26}
27
28int main(void) {
29 enqueue(1); enqueue(2); enqueue(3);
30 printf("%d\n", dequeue());
31 enqueue(4); enqueue(5); enqueue(6);
32 while (count > 0)
33 printf("%d ", dequeue());
34 printf("\n");
35 return 0;
36}
실행 결과
1 2 3 4 5 6
% SIZE가 핵심이야! 나머지 연산으로 인덱스가 배열 크기를 넘으면 다시 0으로 돌아가. 이러면 앞쪽 빈 공간을 재활용할 수 있어!

실전 활용

실생활 속 큐
프린터 대기열 -- 먼저 보낸 문서가 먼저 인쇄된다.
은행/병원 번호표 -- 먼저 뽑은 번호가 먼저 호출된다.

1#include <stdio.h>
2
3#define Q_SIZE 10
4typedef struct { int data[Q_SIZE]; int front, rear, count; } Queue;
5
6void q_init(Queue *q) { q->front = q->rear = q->count = 0; }
7int q_enqueue(Queue *q, int v) {
8 if (q->count >= Q_SIZE) return -1;
9 q->data[q->rear] = v;
10 q->rear = (q->rear + 1) % Q_SIZE;
11 q->count++;
12 return 0;
13}
14int q_dequeue(Queue *q) {
15 if (q->count <= 0) return -1;
16 int v = q->data[q->front];
17 q->front = (q->front + 1) % Q_SIZE;
18 q->count--;
19 return v;
20}
21int q_peek(Queue *q) { return q->count>0?q->data[q->front]:-1; }
22int q_empty(Queue *q) { return q->count==0; }
23
24// BFS로 미로 최단 경로
25#define N 5
26int maze[N][N] = {
27 {0,0,1,0,0},
28 {0,0,0,0,1},
29 {1,0,1,0,0},
30 {0,0,0,1,0},
31 {0,1,0,0,0}
32};
33
34typedef struct { int r, c, dist; } Pos;
35
36int bfs_shortest(int sr, int sc, int er, int ec) {
37 int visited[N][N] = {0};
38 Pos queue[100]; int front=0, rear=0;
39 queue[rear++] = (Pos){sr, sc, 0};
40 visited[sr][sc] = 1;
41 int dr[] = {-1,1,0,0}, dc[] = {0,0,-1,1};
42
43 while (front < rear) {
44 Pos cur = queue[front++];
45 if (cur.r == er && cur.c == ec) return cur.dist;
46 for (int d = 0; d < 4; d++) {
47 int nr = cur.r+dr[d], nc = cur.c+dc[d];
48 if (nr>=0&&nr<N&&nc>=0&&nc<N && !maze[nr][nc] && !visited[nr][nc]) {
49 visited[nr][nc] = 1;
50 queue[rear++] = (Pos){nr, nc, cur.dist+1};
51 }
52 }
53 }
54 return -1;
55}

61 printf("Queue: ");

BFS
너비 우선 탐색. 큐로 구현. 가까운 것부터 탐색. 최단 경로 보장! DFS는 스택(재귀).
deque
Double-Ended Queue. 양쪽 삽입/삭제. 스택+큐 동시! C++ std::deque. 슬라이딩 윈도우에 활용.

BFS = 큐의 킬러 앱! 미로 최단 경로, 소셜 네트워크 촌수, 웹 크롤링 전부 BFS!
[C 코드] 큐 — 우선순위 큐 (힙 기반)
1#include <stdio.h>
2
3typedef struct { int data[100]; int size; } PriorityQueue;
4
5void pq_init(PriorityQueue *pq) { pq->size = 0; }
6void pq_swap(int *a, int *b) { int t=*a; *a=*b; *b=t; }
7
8void pq_push(PriorityQueue *pq, int val) {
9 pq->data[pq->size] = val;
10 int i = pq->size++;
11 while (i > 0) {
12 int parent = (i-1)/2;
13 if (pq->data[parent] > pq->data[i]) { pq_swap(&pq->data[parent], &pq->data[i]); i = parent; }
14 else break;
15 }
16}
17
18int pq_pop(PriorityQueue *pq) {
19 int min = pq->data[0];
20 pq->data[0] = pq->data[--pq->size];
21 int i = 0;
22 while (1) {
23 int smallest = i, l = 2*i+1, r = 2*i+2;
24 if (l < pq->size && pq->data[l] < pq->data[smallest]) smallest = l;
25 if (r < pq->size && pq->data[r] < pq->data[smallest]) smallest = r;
26 if (smallest != i) { pq_swap(&pq->data[i], &pq->data[smallest]); i = smallest; }
27 else break;
28 }
29 return min;
30}
31

35 for (int i = 0; i < 8; i++) pq_push(&pq, vals[i]);
36
37 printf("Priority Queue (min-heap) extraction:\n");

1#include <stdio.h>

3
4typedef struct { char name[20]; int burst; int remaining; } Task;
5
6// Round Robin 스케줄링
7void round_robin(Task *tasks, int n, int quantum) {
8 int time = 0;
9 Task queue[20]; int front=0, rear=0;
10 for(int i=0;i<n;i++) { queue[rear++]=tasks[i]; }
11
12 printf("Round Robin (quantum=%d):\n", quantum);
13 while (front < rear) {
14 Task *t = &queue[front];
15 int run = t->remaining < quantum ? t->remaining : quantum;
16 time += run;
17 t->remaining -= run;
18 printf(" t=%3d: %s ran %dms (remaining %d)\n",
19 time, t->name, run, t->remaining);
20 front++;
21 if (t->remaining > 0) queue[rear++] = *t; // 뒤로 재삽입
22 }
23 printf(" Total time: %dms\n", time);
24}

[C 코드] 큐 — 우선순위 큐 (힙)
1#include <stdio.h>
2
3typedef struct { int data[100]; int size; } MinHeap;
4
5void heap_init(MinHeap *h) { h->size = 0; }
6
7void heap_push(MinHeap *h, int val) {
8 h->data[h->size] = val;
9 int i = h->size++;
10 while (i > 0) {
11 int parent = (i-1)/2;
12 if (h->data[parent] <= h->data[i]) break;
13 int t = h->data[parent]; h->data[parent] = h->data[i]; h->data[i] = t;
14 i = parent;
15 }
16}
17
18int heap_pop(MinHeap *h) {
19 int val = h->data[0];
20 h->data[0] = h->data[--h->size];
21 int i = 0;
22 while (1) {
23 int left=2*i+1, right=2*i+2, smallest=i;
24 if (left<h->size && h->data[left]<h->data[smallest]) smallest=left;
25 if (right<h->size && h->data[right]<h->data[smallest]) smallest=right;
26 if (smallest==i) break;
27 int t=h->data[i]; h->data[i]=h->data[smallest]; h->data[smallest]=t;
28 i = smallest;
29 }
30 return val;
31}
32
33int main() {
34 MinHeap h; heap_init(&h);
35 int vals[] = {5, 3, 8, 1, 9, 2, 7};
36 printf("Push: ");
37 for (int i = 0; i < 7; i++) { printf("%d ", vals[i]); heap_push(&h, vals[i]); }
38 printf("\nPop (priority order): ");
39 while (h.size > 0) printf("%d ", heap_pop(&h));
40 printf("\n");
41 return 0;
42}

부모 ≤ 자식. 루트 = 최소값!
삽입: 끝에 추가 → 위로 올라가기 (sift up)
추출: 루트 제거 → 끝 원소를 루트에 → 아래로 내려가기 (sift down)

시간 복잡도:
push: O(log n)
pop: O(log n)
peek: O(1)

활용:
다익스트라 최단 경로
허프만 압축
이벤트 드리븐 시뮬레이션
OS 작업 스케줄러
priority queue
우선순위 큐. 가장 높은(낮은) 우선순위 먼저 나옴. 힙으로 구현! 삽입/추출 O(log n).
Min Heap: 항상 최소값이 맨 위! push하면 위로 올라가고, pop하면 아래로 내려감!
[C 코드] 큐 — 작업 스케줄러 시뮬레이션
1#include <stdio.h>
2
3typedef struct { char name[20]; int burst; int wait; } Process;
4
5void fcfs_schedule(Process *procs, int n) {
6 int time = 0;
7 double total_wait = 0;
8 printf("=== FCFS Scheduling ===\n");
9 for (int i = 0; i < n; i++) {
10 procs[i].wait = time;
11 printf(" t=%d: %s (burst=%d, wait=%d)\n", time, procs[i].name, procs[i].burst, procs[i].wait);
12 time += procs[i].burst;
13 total_wait += procs[i].wait;
14 }
15 printf(" Avg wait: %.1f\n", total_wait / n);
16}
17
18void sjf_schedule(Process *procs, int n) {
19 // SJF: 짧은 작업 우선
20 for(int i=0;i<n-1;i++) for(int j=0;j<n-1-i;j++)
21 if(procs[j].burst>procs[j+1].burst) { Process t=procs[j]; procs[j]=procs[j+1]; procs[j+1]=t; }
22 int time = 0;
23 double total_wait = 0;
24 printf("\n=== SJF Scheduling ===\n");
25 for (int i = 0; i < n; i++) {
26 procs[i].wait = time;
27 printf(" t=%d: %s (burst=%d, wait=%d)\n", time, procs[i].name, procs[i].burst, procs[i].wait);
28 time += procs[i].burst;
29 total_wait += procs[i].wait;
30 }
31 printf(" Avg wait: %.1f\n", total_wait / n);
32}
33
34int main() {
35 Process p1[] = {{"P1",6,0},{"P2",8,0},{"P3",7,0},{"P4",3,0}};
36 fcfs_schedule(p1, 4);
37 Process p2[] = {{"P1",6,0},{"P2",8,0},{"P3",7,0},{"P4",3,0}};
38 sjf_schedule(p2, 4);
39 return 0;
40}

FCFS
First Come First Served. 먼저 온 순서. 공정하지만 비효율적! 짧은 작업이 긴 작업 뒤에서 대기.
SJF
Shortest Job First. 짧은 작업 우선. 평균 대기 최소! 하지만 실행 시간 예측 어려움.
SJF가 FCFS보다 평균 대기 짧음! 10.2→7.0. 짧은 작업 먼저 처리하면 전체 효율↑!
circular buffer
원형 버퍼. 오디오/네트워크 패킷 버퍼. producer-consumer 패턴. lock-free 구현 가능!
round robin
라운드 로빈. 시간 할당량(quantum)만큼 실행→큐 뒤로. 공정한 CPU 스케줄링!
라운드 로빈: 각 프로세스에 10ms씩! 시간 초과→큐 뒤로. 공정하지만 문맥 교환 비용!
[C 코드] 큐 — 우선순위 큐 (힙)
1#include <stdio.h>
2
3typedef struct { int data[100]; int size; } MinHeap;
4
5void heap_init(MinHeap *h) { h->size = 0; }
6
7void heap_push(MinHeap *h, int val) {
8 int i = h->size++;
9 h->data[i] = val;
10 while (i > 0) {
11 int parent = (i-1)/2;
12 if (h->data[parent] > h->data[i]) {
13 int t = h->data[parent]; h->data[parent] = h->data[i]; h->data[i] = t;
14 i = parent;
15 } else break;
16 }
17}
18
19int heap_pop(MinHeap *h) {
20 int min = h->data[0];
21 h->data[0] = h->data[--h->size];
22 int i = 0;
23 while (1) {
24 int left = 2*i+1, right = 2*i+2, smallest = i;
25 if (left < h->size && h->data[left] < h->data[smallest]) smallest = left;
26 if (right < h->size && h->data[right] < h->data[smallest]) smallest = right;
27 if (smallest == i) break;
28 int t = h->data[i]; h->data[i] = h->data[smallest]; h->data[smallest] = t;
29 i = smallest;
30 }
31 return min;
32}
33
34// 우선순위 큐로 K개 정렬 리스트 병합
35void merge_k_sorted() {
36 int lists[3][4] = {{1,4,7,10},{2,5,8,11},{3,6,9,12}};
37 MinHeap h; heap_init(&h);
38
39 // 모든 원소 push
40 for (int i = 0; i < 3; i++)
41 for (int j = 0; j < 4; j++)
42 heap_push(&h, lists[i][j]);
43
44 printf("Merged: ");
45 while (h.size > 0) printf("%d ", heap_pop(&h));
46 printf("\n");
47}
48

51 int vals[] = {15, 10, 20, 8, 12, 25, 5};
52
53 printf("Push: ");
54 for (int i = 0; i < 7; i++) {
55 heap_push(&h, vals[i]);
56 printf("%d ", vals[i]);
57 }
58 printf("\n\nPop (ascending): ");
59 while (h.size > 0) printf("%d ", heap_pop(&h));
60 printf("\n\n");

62 merge_k_sorted();
63 return 0;
64}

[심화 내용] 힙과 그래프 알고리즘
우선순위 큐 = 힙:
push: O(log n) — 아래→위 올라가기
pop: O(log n) — 위→아래 내려가기
peek: O(1) — 루트만 보기

그래프에서의 활용:
다익스트라: 최단 거리 정점 추출
프림: 최소 비용 간선 추출
허프만: 최소 빈도 노드 추출
A*: f(n)=g(n)+h(n) 최소 노드 추출

K개 정렬 리스트 병합:
각 리스트에서 최소값을 힙에 유지
→ O(N log K), N=전체 원소, K=리스트 수

다익스트라. 최단 경로. 우선순위 큐로 최소 거리 정점 추출. O(E log V). 네비게이션!
Huffman coding
허프만 코딩. 빈도 기반 가변 길이 코드. 우선순위 큐로 최소 빈도 합치기. 데이터 압축!
우선순위 큐 = "항상 최소/최대 먼저!" 일반 큐는 FIFO, 우선순위 큐는 값 순서!
힙 push: 맨 아래에 넣고 위로 올리기. pop: 루트 빼고 맨 아래를 루트에 넣고 아래로!
큐(Queue)
FIFO 구조의 자료구조. 먼저 넣은 데이터가 먼저 나온다.
FIFO
First In, First Out. 먼저 들어간 것이 먼저 나온다.
enqueue
큐의 뒤쪽(rear)에 데이터를 추가하는 연산.
dequeue
큐의 앞쪽(front)에서 데이터를 꺼내는 연산.
front
큐에서 데이터가 나가는 쪽(앞)을 가리키는 변수.
rear
큐에서 데이터가 들어오는 쪽(뒤)을 가리키는 변수.
원형 큐(Circular Queue)
배열의 끝에서 다시 앞으로 돌아가는 큐. 나머지 연산(%)으로 구현.
BFS
Breadth-First Search. 너비 우선 탐색. 큐를 사용하는 그래프 탐색 알고리즘.
