배열 스택의 한계

배열 스택은 크기를 미리 정해야 해서 MAX를 넘으면 Overflow가 나지. 연결 리스트로 만들면 메모리가 허락하는 한 무한히 push할 수 있어!

연결 리스트로 스택 구현

스택: head 앞에 삽입/삭제 (O(1)) / 큐: head에서 삭제, tail에 삽입
ll_stack.c
1#include <stdio.h>
2#include <stdlib.h>
3
4typedef struct Node {
5 int data;
6 struct Node* next;
7} Node;
8
9Node* head = NULL;
10
11void push(int val) {
12 Node* n = (Node*)malloc(sizeof(Node));
13 n->data = val;
14 n->next = head;
15 head = n;
16}
17
18int pop(void) {
19 if (head == NULL) {
20 printf("Stack Empty!\n");
21 return -1;
22 }
23 Node* temp = head;
24 int val = temp->data;
25 head = head->next;
26 free(temp);
27 return val;
28}
29
30int peek(void) {
31 if (head == NULL) return -1;
32 return head->data;
33}
34
35int main(void) {
36 push(10);
37 push(20);
38 push(30);
39 printf("peek: %d\n", peek());
40 printf("pop: %d\n", pop());
41 printf("pop: %d\n", pop());
42 printf("peek: %d\n", peek());
43 return 0;
44}
실행 결과
peek: 30 pop: 30 pop: 20 peek: 10
push는 head 앞에 새 노드를 넣고, pop은 head를 제거하면 끝! 배열 스택의 top++, top--와 같은 역할을 포인터가 하는 거야.

연결 리스트로 큐 구현

ll_queue.c
1#include <stdio.h>
2#include <stdlib.h>
3
4typedef struct Node {
5 int data;
6 struct Node* next;
7} Node;
8
9Node* front = NULL;
10Node* rear = NULL;
11
12void enqueue(int val) {
13 Node* n = (Node*)malloc(sizeof(Node));
14 n->data = val;
15 n->next = NULL;
16 if (rear == NULL) {
17 front = rear = n;
18 } else {
19 rear->next = n;
20 rear = n;
21 }
22}
23
24int dequeue(void) {
25 if (front == NULL) {
26 printf("Queue Empty!\n");
27 return -1;
28 }
29 Node* temp = front;
30 int val = temp->data;
31 front = front->next;
32 if (front == NULL) rear = NULL;
33 free(temp);
34 return val;
35}
36
37int main(void) {
38 enqueue(10);
39 enqueue(20);
40 enqueue(30);
41 printf("%d\n", dequeue());
42 printf("%d\n", dequeue());
43 enqueue(40);
44 printf("%d\n", dequeue());
45 printf("%d\n", dequeue());
46 return 0;
47}
실행 결과
10 20 30 40
16줄큐가 비어 있을 때(rear == NULL) front와 rear 모두 새 노드를 가리키게 해요.
19줄rear->next에 새 노드를 연결하고, rear를 새 노드로 갱신해요.
32줄front가 NULL이 되면 큐가 비었으므로 rear도 NULL로 설정해요.
연결 리스트 큐는 원형 큐가 필요 없어! 배열처럼 공간 낭비가 없거든. 대신 노드마다 포인터 메모리가 추가로 필요하다는 건 기억해 둬.

이중 연결 리스트

이중 연결 리스트(Doubly Linked List)는 각 노드에 next뿐만 아니라 prev(이전 노드) 포인터도 있어요. 앞뒤 양방향으로 이동할 수 있습니다.

이중 연결 리스트: 양방향 이동 가능, 삭제 시 이전 노드를 바로 찾을 수 있다
단일: next만 / 이중: prev + next -- 더 유연하지만 메모리와 구현이 더 복잡
doubly_node.c
1#include <stdio.h>
2#include <stdlib.h>
3
4typedef struct DNode {
5 int data;
6 struct DNode* prev;
7 struct DNode* next;
8} DNode;
9
10DNode* createDNode(int val) {
11 DNode* n = (DNode*)malloc(sizeof(DNode));
12 n->data = val;
13 n->prev = NULL;
14 n->next = NULL;
15 return n;
16}
17
18int main(void) {
19 DNode* a = createDNode(10);
20 DNode* b = createDNode(20);
21 DNode* c = createDNode(30);
22
23 a->next = b; b->prev = a;
24 b->next = c; c->prev = b;
25
26 /* 정방향 출력 */
27 DNode* cur = a;
28 while (cur) {
29 printf("%d ", cur->data);
30 cur = cur->next;
31 }
32 printf("\n");
33
34 /* 역방향 출력 */
35 cur = c;
36 while (cur) {
37 printf("%d ", cur->data);
38 cur = cur->prev;
39 }
40 printf("\n");
41 return 0;
42}
실행 결과
10 20 30 30 20 10
배열 기반 vs 연결 리스트 기반 정리
배열 기반 스택/큐
- 구현 간단, 접근 빠름
- 크기 고정 (MAX 필요)
- KOI 실전에서 주로 사용

연결 리스트 기반 스택/큐
- 크기 제한 없음 (동적)
- 구현 복잡, 포인터 관리 필요
- 실무 프로젝트에서 주로 사용
[KOI] 실전 선택 가이드
KOI 시험에서는 시간이 중요하므로 배열 기반을 추천해요. 빠르게 구현할 수 있고 디버깅도 쉬워요. 하지만 연결 리스트 개념은 알아야 심화 문제를 이해할 수 있어요.

[C 코드] 이중 연결 리스트 — 양방향 순회
1#include <stdio.h>
2#include <stdlib.h>
3
4typedef struct DNode {
5 int data;
6 struct DNode *prev; // ← 이전
7 struct DNode *next; // → 다음
8} DNode;
9
10typedef struct { DNode *head, *tail; int count; } DList;
11
12DList dlist_new() { return (DList){NULL, NULL, 0}; }
13
14void dlist_push_back(DList *l, int data) {
15 DNode *n = (DNode*)malloc(sizeof(DNode));
16 n->data = data; n->next = NULL; n->prev = l->tail;
17 if (l->tail) l->tail->next = n;
18 else l->head = n;
19 l->tail = n; l->count++;
20}
21
22void dlist_push_front(DList *l, int data) {
23 DNode *n = (DNode*)malloc(sizeof(DNode));
24 n->data = data; n->prev = NULL; n->next = l->head;
25 if (l->head) l->head->prev = n;
26 else l->tail = n;
27 l->head = n; l->count++;
28}
29
30void dlist_delete(DList *l, DNode *node) {
31 if (node->prev) node->prev->next = node->next;
32 else l->head = node->next;
33 if (node->next) node->next->prev = node->prev;
34 else l->tail = node->prev;
35 free(node); l->count--;
36}
37
38void dlist_print_forward(DList *l) {
39 printf("→ [%d]: ", l->count);
40 for (DNode *p = l->head; p; p = p->next) printf("%d ", p->data);
41 printf("\n");
42}
43
44void dlist_print_backward(DList *l) {
45 printf("← [%d]: ", l->count);
46 for (DNode *p = l->tail; p; p = p->prev) printf("%d ", p->data);
47 printf("\n");
48}
49
50void dlist_free(DList *l) {
51 DNode *p = l->head;
52 while (p) { DNode *n = p->next; free(p); p = n; }
53 l->head = l->tail = NULL; l->count = 0;
54}
55
56int main() {
57 DList l = dlist_new();
58 for (int i = 1; i <= 5; i++) dlist_push_back(&l, i*10);
59 dlist_push_front(&l, 5);
60
61 dlist_print_forward(&l);
62 dlist_print_backward(&l);
63
64 // 3번째 노드 삭제
65 DNode *p = l.head; for(int i=0;i<2;i++) p=p->next;
66 printf("\nDelete %d\n", p->data);
67 dlist_delete(&l, p);
68 dlist_print_forward(&l);
69
70 dlist_free(&l);
71 return 0;
72}

쉽게 이해하기: 이중 연결 리스트
단일 vs 이중:
단일: [A]→[B]→[C]→NULL
B에서 A로 못 돌아감!

이중: NULL←[A]⇄[B]⇄[C]→NULL
B에서 A로도 C로도 갈 수 있음!

장점:
• 역방향 순회 가능
• 삭제가 O(1) (노드 포인터만 있으면)
• 양쪽 끝 접근 O(1) (head, tail)

단점:
• 포인터 2개 → 메모리 2배 사용
• 삽입/삭제 시 포인터 4개 수정 (prev+next 양쪽)

doubly linked
이중 연결 리스트. prev+next 포인터. 양방향 순회! 삭제 O(1). LRU 캐시의 핵심 자료구조.
LRU cache
Least Recently Used. 해시맵+이중리스트. 접근→맨앞이동, 가득→맨뒤삭제. O(1)!
이중 리스트: prev+next 포인터 2개! 양방향 이동 가능. 삭제도 노드만 알면 O(1)!
포인터 4개 수정! del->prev->next = del->next; del->next->prev = del->prev; 끝!
[C 코드] 이중 연결 리스트 — LRU 캐시
1#include <stdio.h>
2#include <stdlib.h>
3
4#define CACHE_SIZE 3
5
6typedef struct CNode {
7 int key, value;
8 struct CNode *prev, *next;
9} CNode;
10
11typedef struct { CNode *head, *tail; int count; } LRUCache;
12
13LRUCache lru_new() { return (LRUCache){NULL, NULL, 0}; }
14
15void lru_remove(LRUCache *c, CNode *n) {
16 if (n->prev) n->prev->next = n->next; else c->head = n->next;
17 if (n->next) n->next->prev = n->prev; else c->tail = n->prev;
18 c->count--;
19}
20
21void lru_push_front(LRUCache *c, CNode *n) {
22 n->prev = NULL; n->next = c->head;
23 if (c->head) c->head->prev = n;
24 c->head = n;
25 if (!c->tail) c->tail = n;
26 c->count++;
27}
28
29int lru_get(LRUCache *c, int key) {
30 for (CNode *n = c->head; n; n = n->next) {
31 if (n->key == key) {
32 lru_remove(c, n);
33 lru_push_front(c, n); // 최근 사용→맨 앞
34 return n->value;
35 }
36 }
37 return -1; // 미스
38}
39
40void lru_put(LRUCache *c, int key, int val) {
41 for (CNode *n = c->head; n; n = n->next) {
42 if (n->key == key) { n->value = val; lru_remove(c, n); lru_push_front(c, n); return; }
43 }
44 if (c->count >= CACHE_SIZE) { CNode *old = c->tail; lru_remove(c, old); free(old); }
45 CNode *n = (CNode*)malloc(sizeof(CNode));
46 n->key = key; n->value = val;
47 lru_push_front(c, n);
48}
49
50void lru_print(LRUCache *c) {
51 printf("[%d/%d] ", c->count, CACHE_SIZE);
52 for (CNode *n = c->head; n; n = n->next) printf("(%d:%d) ", n->key, n->value);
53 printf("\n");
54}
55
56int main() {
57 LRUCache c = lru_new();
58 lru_put(&c, 1, 100); lru_print(&c);
59 lru_put(&c, 2, 200); lru_print(&c);
60 lru_put(&c, 3, 300); lru_print(&c);
61 printf("get(2) = %d\n", lru_get(&c, 2)); lru_print(&c);
62 lru_put(&c, 4, 400); // 1이 퇴출!
63 printf("put(4,400):\n"); lru_print(&c);
64 printf("get(1) = %d (evicted!)\n", lru_get(&c, 1));
65 return 0;
66}

cache eviction
캐시 퇴출. 가득 차면 무엇을 뺄까? LRU=가장 오래전 사용, LFU=가장 적게 사용, FIFO=먼저 들어온 것.
LRU = 이중 리스트의 킬러 앱! 사용→맨 앞 이동, 가득→맨 뒤 삭제. CPU 캐시 원리!
[심화 내용] Linux 커널 list_head
침투적 연결 리스트:
일반: Node { data; next; } → 데이터가 노드 안
침투적: Data { ...; list_head list; } → 리스트가 데이터 안!

장점:
1. 하나의 데이터가 여러 리스트에 동시 참여!
2. malloc 불필요 (데이터와 함께 할당)
3. container_of로 데이터 역추적

container_of 매크로:
#define container_of(ptr,type,member) \
((type*)((char*)(ptr)-offsetof(type,member)))

list_head 포인터 → 부모 구조체 포인터 획득!
container_of
리스트 노드→부모 구조체 포인터 역추적. offsetof로 멤버 오프셋 계산 후 뺄셈! Linux 핵심.
Linux list_head: 데이터 안에 리스트 포인터! 하나의 struct가 여러 리스트에 동시 참여!
skip list
스킵 리스트. 연결 리스트+다단계 인덱스. O(log n) 검색! Redis sorted set이 사용. BST 대안.
XOR linked list
XOR 리스트. prev XOR next를 하나의 포인터에. 메모리 절반! 하지만 디버깅 어려움.
memory pool for nodes
노드용 메모리 풀. malloc 호출 줄이기. 고정 크기 노드→슬랩 할당. 게임 엔진 필수!
Skip List: 리스트 위에 고속도로! 레벨별 바로가기로 O(n)→O(log n). Redis가 선택!
[C 코드] 연결 리스트 — 병합 정렬
1#include <stdio.h>
2#include <stdlib.h>
3
4typedef struct Node { int data; struct Node *next; } Node;
5
6Node* create(int d) { Node *n=malloc(sizeof(Node)); n->data=d; n->next=NULL; return n; }
7
8// 리스트를 반으로 분할
9void split(Node *head, Node **a, Node **b) {
10 Node *slow=head, *fast=head->next;
11 while(fast && fast->next) { slow=slow->next; fast=fast->next->next; }
12 *a = head;
13 *b = slow->next;
14 slow->next = NULL;
15}
16
17// 두 정렬 리스트 병합
18Node* merge(Node *a, Node *b) {
19 if(!a) return b;
20 if(!b) return a;
21 if(a->data <= b->data) { a->next=merge(a->next,b); return a; }
22 else { b->next=merge(a,b->next); return b; }
23}
24
25// 병합 정렬
26void merge_sort(Node **head) {
27 if(!*head || !(*head)->next) return;
28 Node *a, *b;
29 split(*head, &a, &b);
30 merge_sort(&a);
31 merge_sort(&b);
32 *head = merge(a, b);
33}

35void print_list(Node *h) { for(Node *p=h;p;p=p->next) printf("%d→",p->data); printf("NULL\n"); }

41 list->next->next->next->next=create(3);

[심화 내용] 연결 리스트 정렬
왜 병합 정렬이 최적인가:
Quick Sort: 랜덤 접근 필요 → 리스트에서 느림
Merge Sort: 순차 접근만! → 리스트에 최적

분할:
Fast-Slow 포인터로 중간 찾기
slow는 1칸, fast는 2칸 → slow가 중간!

병합:
두 정렬 리스트 합치기 = O(n)
포인터 연결만! 추가 메모리 0!

복잡도:
시간: O(n log n)
공간: O(log n) — 재귀 스택만
fast-slow pointer
빠른/느린 포인터. 느린=1칸, 빠른=2칸. 빠른이 끝→느린이 중간! 순환 감지, 중간점 찾기!
연결 리스트 병합 정렬: 추가 메모리 0! 포인터만 바꿔서 병합. 배열보다 유리한 점!
circular list
원형 리스트. 마지막→처음 연결. 라운드 로빈 스케줄링! 시작/끝 구분 없음.
XOR linked list
XOR 리스트. prev^next를 한 포인터에 저장. 이중 연결 리스트의 메모리를 절반으로!
원형 이중 리스트: 끝→처음→끝 순환! Linux 커널의 list_head가 정확히 이 구조!
LRU 캐시 = 해시맵 + 이중리스트. 접근→맨앞 이동 O(1), 가득→맨뒤 삭제 O(1)!
skip list
스킵 리스트. 다단계 연결 리스트. 검색 O(log n)! Redis의 Sorted Set. BST 대안.
unrolled linked list
언롤드 리스트. 노드에 배열(여러 원소) 저장. 캐시 친화적! 리스트+배열의 장점 결합.
Skip List = 연결 리스트의 이진 탐색! 여러 레벨의 지름길. Redis가 B-Tree 대신 선택한 이유!
이중 연결 리스트 삭제: prev->next=del->next; del->next->prev=del->prev; free(del); 3줄!
[C 코드] 연결 리스트 — 스택+큐+정렬 통합
1#include <stdio.h>
2#include <stdlib.h>
3
4typedef struct Node { int data; struct Node *next; } Node;
5
6// 연결 리스트 기반 스택
7typedef struct { Node *top; int size; } LStack;
8void ls_init(LStack *s) { s->top=NULL; s->size=0; }
9void ls_push(LStack *s, int v) {
10 Node *n=malloc(sizeof(Node)); n->data=v; n->next=s->top;
11 s->top=n; s->size++;
12}
13int ls_pop(LStack *s) {
14 if(!s->top) return -1;
15 Node *old=s->top; int v=old->data;
16 s->top=old->next; free(old); s->size--;
17 return v;
18}
19
20// 연결 리스트 기반 큐
21typedef struct { Node *front, *rear; int size; } LQueue;
22void lq_init(LQueue *q) { q->front=q->rear=NULL; q->size=0; }
23void lq_enqueue(LQueue *q, int v) {
24 Node *n=malloc(sizeof(Node)); n->data=v; n->next=NULL;
25 if(q->rear) q->rear->next=n; else q->front=n;
26 q->rear=n; q->size++;
27}
28int lq_dequeue(LQueue *q) {
29 if(!q->front) return -1;
30 Node *old=q->front; int v=old->data;
31 q->front=old->next; if(!q->front) q->rear=NULL;
32 free(old); q->size--; return v;
33}
34
35// 순환 검출 (Floyd)
36int has_cycle(Node *head) {
37 Node *slow=head, *fast=head;
38 while(fast && fast->next) {
39 slow = slow->next;
40 fast = fast->next->next;
41 if(slow == fast) return 1;
42 }
43 return 0;
44}
45
46int main() {
47 // 스택
48 LStack s; ls_init(&s);
49 for(int i=1;i<=5;i++) ls_push(&s,i*10);
50 printf("Stack pop: ");
51 while(s.size) printf("%d ",ls_pop(&s));
52 printf("\n");
53
54 // 큐
55 LQueue q; lq_init(&q);
56 for(int i=1;i<=5;i++) lq_enqueue(&q,i*10);
57 printf("Queue deq: ");
58 while(q.size) printf("%d ",lq_dequeue(&q));
59 printf("\n");
60
61 // 순환 검출
62 Node *a=malloc(sizeof(Node)), *b=malloc(sizeof(Node)), *c=malloc(sizeof(Node));
63 a->data=1; b->data=2; c->data=3;
64 a->next=b; b->next=c; c->next=NULL;
65 printf("\nCycle (no loop): %s\n", has_cycle(a)?"Yes":"No");
66 c->next = b; // 순환 생성!
67 printf("Cycle (b→c→b): %s\n", has_cycle(a)?"Yes":"No");
68 return 0;
69}

[심화 내용] 자료구조 통합
배열 vs 연결 리스트로 구현:
|자료구조|배열|연결리스트|
|스택|push/pop O(1)|push/pop O(1)|
|큐|원형 필요|자연스러움|
|정렬|퀵소트|병합정렬|

연결 리스트의 진정한 강점:
• 동적 크기 (최대 크기 불필요)
• 중간 삽입/삭제 O(1) (위치 알 때)
• 여러 리스트 병합 O(1)

배열의 진정한 강점:
• 랜덤 접근 O(1)
• 캐시 지역성 (연속 메모리)
• 공간 오버헤드 없음 (포인터 X)
Floyd tortoise-hare
플로이드 토끼와 거북이. slow=1칸, fast=2칸. 순환이면 반드시 만남! O(n) 시간, O(1) 공간.
sentinel node
감시 노드. 더미 head/tail 노드. 빈 리스트 특별 처리 불필요! 코드 간결화. Linux 커널 사용.
스택/큐를 연결 리스트로! 크기 제한 없고 동적. 배열은 고정 크기, 리스트는 무한 확장!
이중 연결 리스트
각 노드에 prev(이전)과 next(다음) 두 개의 포인터가 있는 연결 리스트.
prev 포인터
이전 노드를 가리키는 포인터. 역방향 이동을 가능하게 한다.
허상 포인터(Dangling Pointer)
이미 해제된 메모리를 가리키는 포인터. 사용하면 오류가 발생한다.
동적 자료구조
실행 중에 크기가 변할 수 있는 자료구조. 연결 리스트가 대표적이다.
tail
연결 리스트의 마지막 노드를 가리키는 포인터.
