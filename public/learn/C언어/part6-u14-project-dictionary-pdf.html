1void addWord(const char* eng, const char* mean) {
2 Word* newWord = (Word*)malloc(sizeof(Word));
3 strcpy(newWord->english, eng);
4 strcpy(newWord->meaning, mean);
5 newWord->next = NULL;
6
7 /* 알파벳순 정렬 삽입 */

9 newWord->next = head;

1#include <stdio.h>
2#include <stdlib.h>
3#include <string.h>
4
5#define ALPHA 26
6
7typedef struct TrieNode {
8 struct TrieNode *children[ALPHA];
9 int is_end;
10 int count; // 검색 빈도
11} TrieNode;
12
13TrieNode* trie_new() {
14 TrieNode *n = (TrieNode*)calloc(1, sizeof(TrieNode));
15 return n;
16}
17
18void trie_insert(TrieNode *root, const char *word) {
19 TrieNode *cur = root;
20 for (int i = 0; word[i]; i++) {
21 int idx = word[i] - 'a';
22 if (!cur->children[idx]) cur->children[idx] = trie_new();
23 cur = cur->children[idx];
24 }
25 cur->is_end = 1;
26}
27
28int trie_search(TrieNode *root, const char *word) {
29 TrieNode *cur = root;
30 for (int i = 0; word[i]; i++) {
31 int idx = word[i] - 'a';
32 if (!cur->children[idx]) return 0;
33 cur = cur->children[idx];
34 }
35 if (cur->is_end) cur->count++;
36 return cur->is_end;
37}
38
39// 자동완성: prefix로 시작하는 모든 단어 출력
40void trie_collect(TrieNode *node, char *prefix, int depth, int max) {
41 if (!node || max <= 0) return;
42 if (node->is_end) printf(" %s\n", prefix);
43 for (int i = 0; i < ALPHA; i++) {
44 if (node->children[i]) {
45 prefix[depth] = 'a' + i;
46 prefix[depth+1] = 0;
47 trie_collect(node->children[i], prefix, depth+1, max);
48 }
49 }
50}
51
52void trie_autocomplete(TrieNode *root, const char *prefix) {
53 TrieNode *cur = root;
54 for (int i = 0; prefix[i]; i++) {
55 int idx = prefix[i] - 'a';
56 if (!cur->children[idx]) { printf(" (no suggestions)\n"); return; }
57 cur = cur->children[idx];
58 }
59 char buf[100];
60 strcpy(buf, prefix);
61 trie_collect(cur, buf, strlen(prefix), 10);
62}

[심화 내용] 트라이와 문자열 자료구조
트라이 (Trie):
각 노드 = 문자 1개
루트→리프 = 하나의 단어
검색/삽입: O(문자열 길이)!

공간 최적화:
Compressed Trie (Patricia Tree)
공통 접두사를 하나의 노드로 압축
→ 메모리 절약

Ternary Search Tree:
트라이 + BST 하이브리드
메모리 효율적, 검색 빠름

실전 사용:
• 검색 엔진 자동완성
• 스펠 체커
• IP 라우팅 (Longest Prefix Match)
• DNA 서열 검색

[C 코드] 사전 — 해시맵 + 체이닝
1#include <stdio.h>
2#include <stdlib.h>
3#include <string.h>
4
5#define BUCKETS 16
6
7typedef struct Entry { char key[30]; char value[100]; struct Entry *next; } Entry;
8typedef struct { Entry *buckets[BUCKETS]; int count; } Dict;
9
10unsigned hash(const char *s) { unsigned h=5381; while(*s) h=h*33+*s++; return h%BUCKETS; }
11
12Dict* dict_new() { return (Dict*)calloc(1, sizeof(Dict)); }
13
14void dict_set(Dict *d, const char *key, const char *val) {
15 unsigned idx = hash(key);
16 for (Entry *e = d->buckets[idx]; e; e = e->next)
17 if (strcmp(e->key, key)==0) { strncpy(e->value, val, 99); return; }
18 Entry *e = (Entry*)malloc(sizeof(Entry));
19 strncpy(e->key, key, 29); strncpy(e->value, val, 99);
20 e->next = d->buckets[idx]; d->buckets[idx] = e; d->count++;
21}
22
23const char* dict_get(Dict *d, const char *key) {
24 for (Entry *e = d->buckets[hash(key)]; e; e = e->next)
25 if (strcmp(e->key, key)==0) return e->value;
26 return NULL;
27}
28
29void dict_stats(Dict *d) {
30 int max_chain = 0, used = 0;
31 for (int i = 0; i < BUCKETS; i++) {
32 int len = 0;
33 for (Entry *e = d->buckets[i]; e; e = e->next) len++;
34 if (len > 0) used++;
35 if (len > max_chain) max_chain = len;
36 }
37 printf("\nDict stats: %d entries, %d/%d buckets used, max chain=%d\n",
38 d->count, used, BUCKETS, max_chain);
39}

[심화 내용] 해시 테이블 vs 트라이 vs BST
해시 테이블:
평균 O(1) 검색/삽입/삭제
정렬 순회 불가, 최악 O(n)

트라이:
O(m) 검색 (m=문자열 길이)
자동완성, 접두사 검색 최적
메모리 많이 사용

BST (균형 트리):
O(log n) 검색/삽입/삭제
정렬 순회 가능! (중위 순회)
범위 질의 가능

선택 기준:
정확 매칭만 → 해시 테이블
접두사 검색 → 트라이
정렬+범위 → BST (Red-Black, AVL)
Red-Black Tree
레드-블랙 트리. 자가 균형 BST. 삽입/삭제 시 색 규칙으로 균형 유지. C++ std::map!
해시=O(1)빠름, 트라이=접두사최적, BST=정렬+범위. 용도에 맞게 선택!
consistent hashing
일관된 해싱. 서버 추가/제거 시 최소 키만 이동. 분산 캐시(Memcached, Redis Cluster) 핵심!
perfect hashing
완전 해싱. 충돌 0! 키 집합이 고정일 때. O(1) 최악 보장. gperf가 자동 생성!
해시맵 크기 소수! 16 대신 17. 소수면 해시 분포 균등. 충돌 감소! 실전 팁!
[C 코드] 사전 — 해시맵 + 충돌 처리
1#include <stdio.h>
2#include <stdlib.h>
3#include <string.h>
4
5#define BUCKETS 32
6
7typedef struct Entry { char key[50]; char value[200]; struct Entry *next; } Entry;
8typedef struct { Entry *buckets[BUCKETS]; int count; } Dict;
9
10unsigned hash(const char *s) { unsigned h=5381; while(*s) h=h*33+*s++; return h%BUCKETS; }
11
12Dict* dict_new() { Dict *d=calloc(1,sizeof(Dict)); return d; }
13
14void dict_set(Dict *d, const char *key, const char *value) {
15 unsigned idx = hash(key);
16 Entry *e = d->buckets[idx];
17 while(e) { if(strcmp(e->key,key)==0){ strncpy(e->value,value,199); return; } e=e->next; }
18 e = malloc(sizeof(Entry));
19 strncpy(e->key,key,49); strncpy(e->value,value,199);
20 e->next = d->buckets[idx]; d->buckets[idx] = e; d->count++;
21}
22
23const char* dict_get(Dict *d, const char *key) {
24 Entry *e = d->buckets[hash(key)];
25 while(e) { if(strcmp(e->key,key)==0) return e->value; e=e->next; }
26 return NULL;
27}
28
29void dict_stats(Dict *d) {
30 int used=0, maxchain=0;
31 for(int i=0;i<BUCKETS;i++) {
32 int len=0; Entry *e=d->buckets[i]; while(e){len++;e=e->next;}
33 if(len>0)used++; if(len>maxchain)maxchain=len;
34 }
35 printf("Dict: %d entries, %d/%d buckets used, max chain=%d\n",
36 d->count, used, BUCKETS, maxchain);
37}

40 Dict *d = dict_new();
41 dict_set(d,"pointer","변수의 메모리 주소를 저장하는 변수");
42 dict_set(d,"struct","여러 타입을 묶는 사용자 정의 타입");
43 dict_set(d,"malloc","힙에서 동적으로 메모리를 할당하는 함수");
44 dict_set(d,"array","같은 타입 원소의 연속 저장 공간");
45 dict_set(d,"stack","LIFO 자료구조. push/pop으로 관리");
46
47 const char *keys[]={"pointer","struct","queue"};
48 for(int i=0;i<3;i++) {
49 const char *v = dict_get(d,keys[i]);
50 printf("%s: %s\n", keys[i], v?v:"(not found)");
51 }
52 dict_stats(d);
53 return 0;
54}
실행 결과
pointer: 변수의 메모리 주소를 저장하는 변수 struct: 여러 타입을 묶는 사용자 정의 타입 queue: (not found) Dict: 5 entries, 5/32 buckets used, max chain=1
load factor
부하 계수. entries/buckets. 0.75 넘으면 리사이징! Java HashMap 기본. 충돌 확률 관리.
해시맵 사전! key→hash→bucket→값. O(1) 검색! 충돌은 체이닝으로 처리!
rehashing
리해싱. 부하계수 초과→버킷 2배→전체 재배치. O(n) 비용! 분할 상환으로 평균 O(1) 유지.
perfect hashing
완벽 해싱. 충돌 0! 정적 데이터에만 가능. 2단계 해싱. 검색어 사전, 예약어 테이블에 활용.
해시맵 성능: 좋은 해시 함수 + 적절한 부하 계수 = O(1) 검색! 나쁜 해시 = O(n) 체이닝!
cuckoo hashing
뻐꾸기 해싱. 해시 2개, 충돌→기존 원소 쫓아냄. 최악 O(1) 검색! 하지만 삽입 복잡.
사전 프로젝트 = Part 6 총정리! 해시맵+트라이+정렬+검색. 자료구조의 종합예술!
[C 코드] 사전 — 해시 테이블 + 단어 빈도
1#include <stdio.h>
2#include <stdlib.h>
3#include <string.h>
4#include <ctype.h>
5
6#define TABLE_SIZE 64
7
8typedef struct Entry { char word[30]; int count; struct Entry *next; } Entry;
9Entry *table[TABLE_SIZE];
10
11unsigned int hash(const char *s) {
12 unsigned int h = 5381;
13 while (*s) h = h * 33 + tolower(*s++);
14 return h % TABLE_SIZE;
15}
16
17void add_word(const char *word) {
18 unsigned int idx = hash(word);
19 Entry *e = table[idx];
20 while (e) { if (strcasecmp(e->word, word)==0) { e->count++; return; } e=e->next; }
21 Entry *n = (Entry*)calloc(1, sizeof(Entry));
22 strncpy(n->word, word, 29);
23 for(char *p=n->word;*p;p++) *p=tolower(*p);
24 n->count = 1;
25 n->next = table[idx]; table[idx] = n;
26}
27
28void count_words(const char *text) {
29 char word[30];
30 int wi = 0;
31 for (int i = 0; ; i++) {
32 if (isalpha(text[i])) {
33 if (wi < 29) word[wi++] = text[i];
34 } else {
35 if (wi > 0) { word[wi] = 0; add_word(word); wi = 0; }
36 if (!text[i]) break;
37 }
38 }
39}
40
41void print_top(int n) {
42 Entry *all[500]; int total = 0;
43 for (int i = 0; i < TABLE_SIZE; i++)
44 for (Entry *e = table[i]; e; e = e->next)
45 if (total < 500) all[total++] = e;
46 // 정렬
47 for (int i = 0; i < total-1; i++)
48 for (int j = 0; j < total-1-i; j++)
49 if (all[j]->count < all[j+1]->count) {
50 Entry *t = all[j]; all[j] = all[j+1]; all[j+1] = t;
51 }
52 printf("Top %d words:\n", n);
53 for (int i = 0; i < n && i < total; i++)
54 printf(" %-15s %d\n", all[i]->word, all[i]->count);
55 printf("Total unique: %d\n", total);
56}
57
58int main() {
59 const char *text = "the quick brown fox jumps over the lazy dog "
60 "the dog barked at the fox and the fox ran away "
61 "a quick brown dog chased the fox";
62 count_words(text);
63 print_top(8);
64 return 0;
65}

word frequency
단어 빈도. 해시 테이블로 O(1) 카운팅. 검색 엔진의 TF-IDF, 자연어 처리의 기본!
TF-IDF
Term Frequency-Inverse Document Frequency. 단어 빈도×문서 희소성. 검색 엔진의 핵심 랭킹!
단어 빈도 = 해시 테이블의 실전! 텍스트→단어 분리→해시→카운트. 검색 엔진의 기본!
정렬 삽입
새 데이터를 정렬된 위치에 바로 넣는 방식. 별도의 정렬 과정이 불필요하다.
strcmp
두 문자열을 사전순으로 비교하는 함수. 같으면 0, 앞이면 음수, 뒤면 양수.
srand / rand
srand는 난수 시드를 설정하고, rand는 난수를 생성한다.
모듈화
프로그램을 기능별 함수로 나누는 것. 유지보수와 재사용이 쉬워진다.

프로그램이 종료되어도 데이터가 유지되는 것. 파일 I/O로 구현한다.
