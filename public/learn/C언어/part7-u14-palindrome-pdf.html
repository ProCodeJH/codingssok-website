회문(Palindrome)이란?

"level", "madam", "racecar" 같은 단어들의 공통점이 뭔지 알아? 앞에서 읽어도 뒤에서 읽어도 똑같아! 이런 단어를 회문이라고 해!

회문(Palindrome)은 앞에서 읽어도 뒤에서 읽어도 같은 문자열이에요. "level", "noon", "12321" 같은 것들이 회문이에요. 프로그래밍 대회에서 자주 나오는 문제 유형이에요.

undefined

회문 판별 알고리즘

회문을 판별하는 방법은 간단해요. 왼쪽 끝(left)과 오른쪽 끝(right)의 문자를 비교하면서 가운데로 이동해요. 하나라도 다르면 회문이 아니에요.

1#include <stdio.h>
2#include <string.h>
3
4int is_palindrome(const char *str) {
5 int left = 0;
6 int right = strlen(str) - 1;
7
8 while (left < right) {
9 if (str[left] != str[right]) {
10 return 0; // 다르면 회문 아님
11 }
12 left++;
13 right--;
14 }
15 return 1; // 모두 같으면 회문!
16}
17
18int main(void) {
19 char *words[] = { "level", "hello", "madam", "racecar", "cat" };
20 int n = 5;
21
22 for (int i = 0; i < n; i++) {
23 printf("\"%s\" -> %s\n", words[i],
24 is_palindrome(words[i]) ? "Palindrome!" : "Not palindrome");
25 }
26 return 0;
27}
실행 결과
"level" -> Palindrome! "hello" -> Not palindrome "madam" -> Palindrome! "racecar" -> Palindrome! "cat" -> Not palindrome
undefined
시간 복잡도는 O(n/2) = O(n)이야. 문자열의 절반만 비교하면 되니까 아주 효율적이지!

N-gram: 연속된 N개 묶음

N-gram은 문자열에서 연속된 N개의 문자(또는 단어)를 묶은 것이에요. 검색 엔진, 맞춤법 검사, AI 자연어 처리에서 많이 사용돼요.

1#include <stdio.h>
2#include <string.h>
3
4void print_ngrams(const char *str, int n) {
5 int len = strlen(str);
6 printf("%d-gram of \"%s\":\n", n, str);
7
8 for (int i = 0; i <= len - n; i++) {
9 printf(" ");
10 for (int j = 0; j < n; j++) {
11 printf("%c", str[i + j]);
12 }
13 printf("\n");
14 }
15}
16
17int main(void) {
18 print_ngrams("hello", 2);
19 printf("\n");
20 print_ngrams("hello", 3);
21 return 0;
22}
실행 결과
2-gram of "hello": he el ll lo 3-gram of "hello": hel ell llo
N-gram은 AI에서 정말 많이 쓰여. 예를 들어 "나는 학교에 ___" 다음에 올 단어를 예측할 때, 앞의 단어들(N-gram)을 보고 판단하는 거야!
[KOI] 문자열 문제 풀이 팁
KOI에서 자주 나오는 문자열 문제 유형: 회문 판별, 부분 문자열 찾기, 문자열 뒤집기, 아나그램 등. 이런 문제는 대부분 두 포인터(two pointer) 기법이나 문자 배열 조작으로 풀 수 있어요.

연습문제

OX 퀴즈

"abcba"는 회문이다.

정답: ⭕ O
앞에서 읽어도 a-b-c-b-a, 뒤에서 읽어도 a-b-c-b-a로 같으므로 회문이에요.
객관식

문자열 "hello"의 2-gram은 몇 개인가?

  1. 3개
  2. 4개
  3. 5개
  4. 6개
정답: 2
길이가 5인 문자열의 2-gram 개수는 5-2+1 = 4개 (he, el, ll, lo)에요.
단답형

회문 판별 알고리즘의 시간 복잡도는?

정답: O(n)
문자열 길이의 절반만 비교하므로 O(n/2)이고, 빅오 표기법에서 상수를 제거하면 O(n)이에요.
객관식

다음 중 회문이 아닌 것은?

  1. "noon"
  2. "deed"
  3. "book"
  4. "civic"
정답: 3
book은 앞에서 읽으면 b-o-o-k, 뒤에서 읽으면 k-o-o-b로 달라요.
OX 퀴즈

N-gram에서 N이 클수록 더 많은 조각이 생긴다.

정답: ❌ X
N이 클수록 조각 수는 줄어들어요. 길이 L인 문자열의 N-gram 개수는 L-N+1이에요.

6int is_palindrome(const char *s) {

9 return 1;
10}

12// 대소문자/공백 무시
13int is_palindrome_ignore(const char *s) {
14 int l = 0, r = strlen(s) - 1;
15 while (l < r) {
16 while (l < r && !isalnum(s[l])) l++;
17 while (l < r && !isalnum(s[r])) r--;
18 if (tolower(s[l]) != tolower(s[r])) return 0;
19 l++; r--;
20 }
21 return 1;
22}

24// 숫자 팰린드롬
25int is_num_palindrome(int n) {
26 if (n < 0) return 0;
27 int original = n, reversed = 0;
28 while (n > 0) { reversed = reversed * 10 + n % 10; n /= 10; }
29 return original == reversed;
30}

32// 가장 긴 팰린드롬 부분문자열 (확장법)
33void longest_palindrome(const char *s) {
34 int n = strlen(s), best_start = 0, best_len = 1;
35 for (int center = 0; center < n; center++) {
36 // 홀수 길이
37 int l = center, r = center;
38 while (l >= 0 && r < n && s[l] == s[r]) { l--; r++; }
39 if (r - l - 1 > best_len) { best_len = r-l-1; best_start = l+1; }
40 // 짝수 길이
41 l = center; r = center + 1;
42 while (l >= 0 && r < n && s[l] == s[r]) { l--; r++; }
43 if (r - l - 1 > best_len) { best_len = r-l-1; best_start = l+1; }
44 }
45 printf(" Longest palindrome in \"%s\": \"%.*s\" (len=%d)\n", s, best_len, s+best_start, best_len);
46}

왼쪽(l)과 오른쪽(r)에서 동시에 시작

5// === 팰린드롬 분할 (Palindrome Partitioning) ===
6// 문자열을 모두 팰린드롬인 부분으로 나누기
7int is_pal(const char *s, int l, int r) {
8 while (l < r) { if (s[l++] != s[r--]) return 0; }
9 return 1;
10}
11
12void partition_helper(const char *s, int start, int n, char parts[][20], int depth) {
13 if (start == n) {
14 printf(" ");
15 for (int i = 0; i < depth; i++) printf("[%s]", parts[i]);
16 printf("\n");
17 return;
18 }
19 for (int end = start; end < n; end++) {
20 if (is_pal(s, start, end)) {
21 int len = end - start + 1;
22 strncpy(parts[depth], s+start, len);
23 parts[depth][len] = 0;
24 partition_helper(s, end+1, n, parts, depth+1);
25 }
26 }
27}

33}

48// === 회문 만들기 (최소 문자 추가) ===
49int min_insertions(const char *s) {
50 int n = strlen(s);
51 // dp[i][j] = s[i..j]를 팰린드롬으로 만드는 최소 삽입
52 int dp[50][50] = {0};
53 for (int len = 2; len <= n; len++) {
54 for (int i = 0; i <= n-len; i++) {
55 int j = i + len - 1;
56 if (s[i] == s[j]) dp[i][j] = dp[i+1][j-1];
57 else dp[i][j] = 1 + (dp[i+1][j]<dp[i][j-1]?dp[i+1][j]:dp[i][j-1]);
58 }
59 }
60 return dp[0][n-1];
61}
62
63// === 연결 리스트 팰린드롬 ===
64typedef struct Node { int val; struct Node *next; } Node;
65
66Node* make_list(int *vals, int n) {
67 Node *head = NULL, *tail = NULL;
68 for (int i = 0; i < n; i++) {
69 Node *node = (Node*)malloc(sizeof(Node));
70 node->val = vals[i]; node->next = NULL;
71 if (!head) head = tail = node;
72 else { tail->next = node; tail = node; }
73 }
74 return head;
75}
76
77int list_palindrome(Node *head) {
78 // 배열에 복사 후 투 포인터
79 int arr[100], n = 0;
80 for (Node *p = head; p; p = p->next) arr[n++] = p->val;
81 int l = 0, r = n-1;
82 while (l < r) { if (arr[l++] != arr[r--]) return 0; }
83 return 1;
84}

[심화 내용] 팰린드롬 DP
최소 삽입 DP:
dp[i][j] = s[i..j]를 팰린드롬으로 만드는 최소 삽입
s[i]==s[j] → dp[i][j] = dp[i+1][j-1]
s[i]!=s[j] → 1 + min(dp[i+1][j], dp[i][j-1])

팰린드롬 분할:
백트래킹 + is_palindrome 체크
모든 가능한 분할을 탐색
→ 면접에서 자주 출제!

Manacher O(n):
이전 팰린드롬 정보를 활용한 확장
중심 확장의 중복 제거
→ 최장 팰린드롬 부분 문자열 O(n)!
palindrome partition
팰린드롬 분할. 문자열을 모두 팰린드롬인 부분으로 나누기. 백트래킹으로 탐색. 면접 단골!
min insertions
최소 삽입. DP로 O(n²). LPS(최장 팰린드롬 부분수열)와 관련. n - LPS길이 = 최소 삽입 수!
팰린드롬 소수 20개! 2,3,5,7,11,101... 한 자리+두 자리 소수는 쉽고, 세 자리부터 희소!
[C 코드] 팰린드롬 — Manacher 알고리즘
1#include <stdio.h>
2#include <string.h>
3
4// === Manacher O(n) ===
5void manacher(const char *s) {
6 int n = strlen(s);
7 // 변환: "abc" -> "^#a#b#c#$"
8 int tn = 2*n + 3;
9 char t[205];
10 t[0] = 94; t[1] = 35; // ^ #
11 for (int i = 0; i < n; i++) { t[2*i+2] = s[i]; t[2*i+3] = 35; }
12 t[tn-1] = 36; t[tn] = 0; // $
13
14 int p[205]; // p[i] = i 중심 팰린드롬 반지름
15 int center = 0, right = 0;
16 int max_len = 0, max_center = 0;
17
18 for (int i = 1; i < tn-1; i++) {
19 int mirror = 2*center - i;
20 p[i] = (i < right) ? (p[mirror] < right-i ? p[mirror] : right-i) : 0;
21 while (t[i+p[i]+1] == t[i-p[i]-1]) p[i]++;
22 if (i+p[i] > right) { center = i; right = i+p[i]; }
23 if (p[i] > max_len) { max_len = p[i]; max_center = i; }
24 }
25
26 int start = (max_center - max_len) / 2;

30// === 팰린드롬 생성기 ===
31void make_palindrome(const char *s, char *out) {
32 int n = strlen(s);
33 strcpy(out, s);
34 for (int i = n-2; i >= 0; i--)
35 out[2*n-2-i] = s[i];
36 out[2*n-1] = 0;
37}

39// === 회문 카운터 (DP) ===
40int count_palindromic_substrings(const char *s) {
41 int n = strlen(s), count = 0;
42 for (int c = 0; c < n; c++) {
43 // 홀수 길이
44 int l=c, r=c;
45 while (l>=0 && r<n && s[l]==s[r]) { count++; l--; r++; }
46 // 짝수 길이
47 l=c; r=c+1;
48 while (l>=0 && r<n && s[l]==s[r]) { count++; l--; r++; }
49 }
50 return count;
51}

54int is_date_palindrome(int y, int m, int d) {
55 char buf[20];
56 sprintf(buf, "%04d%02d%02d", y, m, d);
57 int n = strlen(buf), l=0, r=n-1;
58 while (l < r) { if (buf[l++] != buf[r--]) return 0; }
59 return 1;
60}
