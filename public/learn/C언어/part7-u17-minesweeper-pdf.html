게임 설계

지뢰찾기 게임의 핵심 규칙을 정리해 볼게요.

지뢰찾기 규칙
1. 9x9 보드에 10개의 지뢰가 랜덤으로 배치된다.
2. 사용자가 좌표를 입력하면 해당 칸을 연다.
3. 지뢰를 밟으면 게임 오버!
4. 지뢰가 아닌 칸을 열면 주변 8칸의 지뢰 개수가 표시된다.
5. 지뢰가 아닌 모든 칸을 열면 승리!

1#include <stdio.h>
2#include <stdlib.h>
3#include <time.h>
4
5#define SIZE 9
6#define MINES 10
7
8// 보드 배열
9int board[SIZE][SIZE]; // 지뢰=-1, 숫자=주변 지뢰 수
10int visible[SIZE][SIZE]; // 0=닫힘, 1=열림
11
12// 8방향 이동 (상, 하, 좌, 우, 대각 4방향)
13int dx[] = {-1, -1, -1, 0, 0, 1, 1, 1};
14int dy[] = {-1, 0, 1, -1, 1, -1, 0, 1};
15
16// 보드 초기화
17void init_board(void) {
18 for (int i = 0; i < SIZE; i++) {
19 for (int j = 0; j < SIZE; j++) {
20 board[i][j] = 0;
21 visible[i][j] = 0;
22 }
23 }
24}
25
26// 지뢰 배치
27void place_mines(void) {
28 srand((unsigned)time(NULL));
29 int placed = 0;
30 while (placed < MINES) {
31 int r = rand() % SIZE;
32 int c = rand() % SIZE;
33 if (board[r][c] != -1) {
34 board[r][c] = -1;
35 placed++;
36 }
37 }
38}

40// 주변 지뢰 개수 계산
41void calc_numbers(void) {
42 for (int i = 0; i < SIZE; i++) {
43 for (int j = 0; j < SIZE; j++) {
44 if (board[i][j] == -1) continue;
45 int count = 0;
46 for (int d = 0; d < 8; d++) {
47 int ni = i + dx[d];
48 int nj = j + dy[d];
49 if (ni >= 0 && ni < SIZE && nj >= 0 && nj < SIZE) {
50 if (board[ni][nj] == -1) count++;
51 }
52 }
53 board[i][j] = count;
54 }
55 }
56}

67 printf("%d | ", i);

86void open_cell(int r, int c) {

91

95 open_cell(r + dx[d], c + dy[d]);

핵심 알고리즘 해설

이 프로그램에서 가장 중요한 알고리즘은 주변 8칸 탐색과 빈 칸 자동 열기(재귀)야!
핵심 알고리즘
1. 8방향 탐색: dx[], dy[] 배열로 상하좌우+대각선 8방향을 표현. 범위 체크 필수!
2. 지뢰 배치: rand()로 랜덤 좌표 생성. 중복 방지를 위해 이미 지뢰인지 확인.
3. 빈 칸 자동 열기: 숫자가 0인 칸을 열면 주변 8칸도 재귀적으로 연다. Flood Fill 알고리즘!
4. 승리 판정: 지뢰가 아닌 모든 칸이 열렸는지 확인.
open_cell의 재귀 호출은 Flood Fill 알고리즘이야. 그림 그리기 프로그램의 '페인트 통' 기능과 같은 원리! AI에서도 BFS/DFS 탐색에 사용돼.

연습문제

OX 퀴즈

지뢰찾기에서 주변 8칸을 탐색할 때 범위 체크를 하지 않으면 배열 범위를 벗어날 수 있다.

정답: ⭕ O
보드의 가장자리 칸은 주변 8칸 중 일부가 보드 밖에 있어요. 범위 체크(0 이상, SIZE 미만)를 반드시 해야 해요.
객관식

9x9 보드에 10개의 지뢰가 있을 때, 승리 조건은?

  1. 지뢰를 모두 찾으면 승리
  2. 지뢰가 아닌 71개의 칸을 모두 열면 승리
  3. 10턴 안에 끝내면 승리
  4. 점수가 100점 이상이면 승리
정답: 2
9x9=81칸 중 지뢰 10개를 뺀 71개의 안전한 칸을 모두 열면 승리예요.
단답형

open_cell에서 사용하는 재귀 알고리즘의 이름은?

정답: Flood Fill (플러드 필)
빈 칸(0)을 만나면 주변으로 퍼져나가면서 칸을 여는 것이 물이 퍼지는 것과 비슷해서 Flood Fill이라고 해요.
객관식

dx[] = {-1,-1,-1,0,0,1,1,1}에서 dx[0]=-1, dy[0]=-1이 나타내는 방향은?

  1. 오른쪽 위
  2. 왼쪽 위
  3. 왼쪽 아래
  4. 오른쪽 아래
정답: 2
행이 -1(위로), 열이 -1(왼쪽으로) 이동하므로 왼쪽 위 대각선이에요.
객관식

지뢰찾기에서 8x8 보드의 특정 칸 주변 지뢰 수를 세려면 최대 몇 칸을 확인해야 하는가?

  1. 4칸
  2. 6칸
  3. 8칸
  4. 9칸
정답: 2
한 칸의 주변은 상하좌우 + 대각선 4방향 = 최대 8칸입니다.

[C 코드] 지뢰찾기 — 핵심 로직
1#include <stdio.h>
2#include <stdlib.h>
3#include <time.h>
4
5#define ROWS 8
6#define COLS 8
7#define MINES 10
8
9typedef struct {
10 int mine; // 지뢰 여부
11 int revealed; // 공개 여부
12 int flagged; // 깃발 여부
13 int adjacent; // 인접 지뢰 수
14} Cell;
15
16Cell board[ROWS][COLS];
17
18void init_board() {
19 // 초기화
20 for (int r = 0; r < ROWS; r++)
21 for (int c = 0; c < COLS; c++)
22 board[r][c] = (Cell){0, 0, 0, 0};
23
24 // 지뢰 배치
25 srand(42);
26 int placed = 0;
27 while (placed < MINES) {
28 int r = rand() % ROWS, c = rand() % COLS;
29 if (!board[r][c].mine) { board[r][c].mine = 1; placed++; }
30 }
31
32 // 인접 지뢰 수 계산
33 int dr[] = {-1,-1,-1,0,0,1,1,1};
34 int dc[] = {-1,0,1,-1,1,-1,0,1};
35 for (int r = 0; r < ROWS; r++)
36 for (int c = 0; c < COLS; c++) {
37 if (board[r][c].mine) continue;
38 int cnt = 0;
39 for (int d = 0; d < 8; d++) {
40 int nr = r+dr[d], nc = c+dc[d];
41 if (nr>=0&&nr<ROWS&&nc>=0&&nc<COLS && board[nr][nc].mine) cnt++;
42 }
43 board[r][c].adjacent = cnt;
44 }
45}
46
47// 빈 칸 자동 공개 (Flood Fill)
48void reveal(int r, int c) {
49 if (r<0||r>=ROWS||c<0||c>=COLS) return;
50 if (board[r][c].revealed || board[r][c].flagged) return;
51 board[r][c].revealed = 1;
52 if (board[r][c].adjacent == 0 && !board[r][c].mine) {
53 for (int dr=-1;dr<=1;dr++)
54 for (int dc=-1;dc<=1;dc++)
55 if(dr||dc) reveal(r+dr, c+dc);
56 }
57}
58
59void print_board(int show_all) {
60 printf(" ");
61 for(int c=0;c<COLS;c++) printf("%d ",c);
62 printf("\n");
63 for(int r=0;r<ROWS;r++) {
64 printf("%d ",r);
65 for(int c=0;c<COLS;c++) {
66 Cell *cell = &board[r][c];
67 if(show_all && cell->mine) printf("💣");
68 else if(cell->flagged) printf("🚩");
69 else if(!cell->revealed) printf("■ ");
70 else if(cell->mine) printf("💣");
71 else if(cell->adjacent==0) printf(" ");
72 else printf("%d ",cell->adjacent);
73 }
74 printf("\n");
75 }
76}
77
78int main() {
79 init_board();
80 printf("=== 지뢰찾기 %dx%d (지뢰 %d개) ===\n\n", ROWS, COLS, MINES);
81
82 // 시뮬레이션: 몇 칸 공개
83 reveal(0, 0);
84 reveal(4, 4);
85 reveal(7, 7);
86
87 printf("Player view:\n");
88 print_board(0);
89 printf("\nAnswer:\n");
90 print_board(1);
91 return 0;
92}

Flood Fill
홍수 채우기. 연결된 빈 영역 자동 공개. 재귀/BFS/스택으로 구현. 그림판의 페인트 통!
CSP
Constraint Satisfaction Problem. 제약 조건 만족. 지뢰찾기, 스도쿠, 그래프 색칠에 활용!
지뢰찾기 = 2D 배열+재귀의 종합! 인접 카운트, Flood Fill, 상태 관리 전부!
Flood Fill: 빈 칸(0)이면 8방향 재귀! 숫자 칸이면 멈춤. 클릭 한 번에 영역 확장!
[C 코드] 지뢰찾기 — 자동 풀이 + 확률
1#include <stdio.h>
2#include <stdlib.h>
3
4#define R 8
5#define C 8
6#define M 10
7
8typedef struct { int mine,rev,flag,adj; } Cell;
9Cell bd[R][C];
10int dr[]={-1,-1,-1,0,0,1,1,1}, dc[]={-1,0,1,-1,1,-1,0,1};
11
12void init() {
13 for(int r=0;r<R;r++) for(int c=0;c<C;c++) bd[r][c]=(Cell){0,0,0,0};
14 srand(42); int p=0;
15 while(p<M){int r=rand()%R,c=rand()%C;if(!bd[r][c].mine){bd[r][c].mine=1;p++;}}
16 for(int r=0;r<R;r++) for(int c=0;c<C;c++) if(!bd[r][c].mine) {
17 int cnt=0;
18 for(int d=0;d<8;d++){int nr=r+dr[d],nc=c+dc[d];
19 if(nr>=0&&nr<R&&nc>=0&&nc<C&&bd[nr][nc].mine)cnt++;}
20 bd[r][c].adj=cnt;
21 }
22}
23
24void reveal(int r, int c) {
25 if(r<0||r>=R||c<0||c>=C||bd[r][c].rev||bd[r][c].flag) return;
26 bd[r][c].rev=1;
27 if(bd[r][c].adj==0&&!bd[r][c].mine)
28 for(int d=0;d<8;d++) reveal(r+dr[d],c+dc[d]);
29}
30
31// 자동 풀이: 확정적 단계
32int auto_solve_step() {
33 int progress = 0;
34 for(int r=0;r<R;r++) for(int c=0;c<C;c++) {
35 if(!bd[r][c].rev || bd[r][c].adj==0) continue;
36 int hidden=0, flagged=0;
37 for(int d=0;d<8;d++){int nr=r+dr[d],nc=c+dc[d];
38 if(nr>=0&&nr<R&&nc>=0&&nc<C){
39 if(!bd[nr][nc].rev&&!bd[nr][nc].flag) hidden++;
40 if(bd[nr][nc].flag) flagged++;
41 }}
42 // 남은 숫자 == 남은 숨김 → 전부 지뢰!
43 if(bd[r][c].adj - flagged == hidden && hidden > 0) {
44 for(int d=0;d<8;d++){int nr=r+dr[d],nc=c+dc[d];
45 if(nr>=0&&nr<R&&nc>=0&&nc<C&&!bd[nr][nc].rev&&!bd[nr][nc].flag)
46 { bd[nr][nc].flag=1; progress++; }}
47 }
48 // 숫자 == 깃발 수 → 나머지 안전!
49 if(flagged == bd[r][c].adj && hidden > 0) {
50 for(int d=0;d<8;d++){int nr=r+dr[d],nc=c+dc[d];
51 if(nr>=0&&nr<R&&nc>=0&&nc<C&&!bd[nr][nc].rev&&!bd[nr][nc].flag)
52 { reveal(nr,nc); progress++; }}
53 }
54 }
55 return progress;
56}
57

59 printf(" ");
60 for(int c=0;c<C;c++) printf("%d ",c);
61 printf("\n");
62 for(int r=0;r<R;r++){printf("%d ",r);
63 for(int c=0;c<C;c++){
64 if(bd[r][c].flag) printf("🚩");
65 else if(!bd[r][c].rev) printf("■ ");
66 else if(bd[r][c].mine) printf("💣");
67 else if(bd[r][c].adj==0) printf(" ");

69 }printf("\n");}
70}

[심화 내용] 지뢰찾기의 수학
확정적 풀이 (Deterministic):
숫자 == 인접 숨김 → 전부 지뢰!
숫자 == 깃발 수 → 나머지 안전!
→ 반복하면 많은 경우 해결

확률적 풀이:
확정 불가능할 때 → 확률 계산
각 칸의 지뢰 확률 추정
가장 안전한 칸 선택

NP-완전 문제!
지뢰찾기 일관성 판정 = NP-complete
Richard Kaye (2000) 증명
→ 일반적인 다항 시간 알고리즘 불가능!

• CSP (제약 전파)
• 백트래킹 (가능한 배치 탐색)

8typedef struct { int mine,rev,flag,adj; } Cel;

10int dx[]={-1,-1,-1,0,0,1,1,1}, dy[]={-1,0,1,-1,1,-1,0,1};

12void minit() {
13 memset(b,0,sizeof(b));
14 srand(42); int p=0;
15 while(p<MINES){int r=rand()%SZ,c=rand()%SZ;if(!b[r][c].mine){b[r][c].mine=1;p++;}}
16 for(int r=0;r<SZ;r++) for(int c=0;c<SZ;c++) if(!b[r][c].mine) {
17 int cnt=0;
18 for(int d=0;d<8;d++){int nr=r+dx[d],nc=c+dy[d];
19 if(nr>=0&&nr<SZ&&nc>=0&&nc<SZ&&b[nr][nc].mine)cnt++;}
20 b[r][c].adj=cnt;
21 }
22}

24void mreveal(int r, int c) {
25 if(r<0||r>=SZ||c<0||c>=SZ||b[r][c].rev||b[r][c].flag) return;
26 b[r][c].rev=1;
27 if(b[r][c].adj==0&&!b[r][c].mine)
28 for(int d=0;d<8;d++) mreveal(r+dx[d],c+dy[d]);
29}

31// 확률 추정: 각 미공개 칸의 지뢰 확률
32void estimate_probs(float prob[SZ][SZ]) {
33 memset(prob, 0, SZ*SZ*sizeof(float));
34
35 for(int r=0;r<SZ;r++) for(int c=0;c<SZ;c++) {
36 if(!b[r][c].rev || b[r][c].adj==0) continue;
37 int hidden=0, flagged=0;
38 for(int d=0;d<8;d++){int nr=r+dx[d],nc=c+dy[d];
39 if(nr>=0&&nr<SZ&&nc>=0&&nc<SZ){
40 if(!b[nr][nc].rev&&!b[nr][nc].flag) hidden++;
41 if(b[nr][nc].flag) flagged++;
42 }}
43 if(hidden==0) continue;
44 float mine_prob = (float)(b[r][c].adj - flagged) / hidden;
45 for(int d=0;d<8;d++){int nr=r+dx[d],nc=c+dy[d];
46 if(nr>=0&&nr<SZ&&nc>=0&&nc<SZ&&!b[nr][nc].rev&&!b[nr][nc].flag)
47 prob[nr][nc] = prob[nr][nc]>mine_prob ? prob[nr][nc] : mine_prob;
48 }
49 }
50}

52void print_probs(float prob[SZ][SZ]) {
53 printf(" Mine probability:\n ");
54 for(int c=0;c<SZ;c++) printf("%4d",c);
55 printf("\n");
56 for(int r=0;r<SZ;r++) {
57 printf("%d ",r);
58 for(int c=0;c<SZ;c++) {
59 if(b[r][c].rev) printf(" -- ");
60 else if(b[r][c].flag) printf(" FL ");
61 else if(prob[r][c]>0) printf("%3.0f%%", prob[r][c]*100);
62 else printf(" ? ");
63 }
64 printf("\n");
65 }
66}

68// 최적 다음 수: 가장 낮은 확률의 칸
69void suggest_move(float prob[SZ][SZ]) {
70 float min_prob = 2.0;
71 int best_r = -1, best_c = -1;
72 for(int r=0;r<SZ;r++) for(int c=0;c<SZ;c++) {
73 if(b[r][c].rev || b[r][c].flag) continue;
74 float p = prob[r][c] > 0 ? prob[r][c] : 0.5;
75 if(p < min_prob) { min_prob=p; best_r=r; best_c=c; }
76 }
77 if(best_r >= 0)
78 printf(" Best move: (%d,%d) risk=%.0f%%\n", best_r, best_c, min_prob*100);
79}

확률 추정 방법: 숫자 칸의 남은 지뢰 수 / 남은 숨김 칸 수 = 각 이웃의 지뢰 확률 여러 숫자 칸의 정보 결합 → max 사용 더 정확한 방법: Monte Carlo 시뮬레이션 가능한 지뢰 배치를 무작위 생성 각 칸에 지뢰가 나타나는 비율 → 확률 최적 전략: 1. 확정적 규칙 먼저 (100% 안전/지뢰) 2. 가장 낮은 확률의 칸 선택 3. 코너/에지 우선 (정보가 적은 곳) NP-완전이므로: 완벽한 풀이는 불가능하지만 실용적 확률 추정은 매우 효과적!
