1#include <stdio.h>
2#include <stdlib.h>
3#include <string.h>
4#include <ctype.h>
5
6/*
7 * 미니 수식 인터프리터
8 * 지원: +, -, *, /, 괄호, 변수(a-z)
9 * 문법: expr = term((+|-) term)*
10 * term = factor((*|/) factor)*
11 * factor = NUMBER | VARIABLE | ( expr )
12 */
13
14typedef enum { T_NUM, T_VAR, T_OP, T_LPAR, T_RPAR, T_END } TType;
15typedef struct { TType type; double num; char ch; } Token;
16
17const char *src;
18Token cur;
19double vars[26]; // a-z
20
21void next_token() {
22 while (isspace(*src)) src++;
23 if (!*src) { cur = (Token){T_END,0,0}; return; }
24 if (isdigit(*src) || (*src=='.')) {
25 cur.type = T_NUM;
26 cur.num = strtod(src, (char**)&src);
27 return;
28 }
29 if (*src >= 'a' && *src <= 'z') {
30 cur = (Token){T_VAR, 0, *src++};
31 return;
32 }
33 if (*src=='(') { cur = (Token){T_LPAR,0,*src++}; return; }
34 if (*src==')') { cur = (Token){T_RPAR,0,*src++}; return; }
35 cur = (Token){T_OP, 0, *src++};
36}
37
38double parse_expr();
39
40double parse_factor() {
41 if (cur.type == T_NUM) { double v=cur.num; next_token(); return v; }
42 if (cur.type == T_VAR) { double v=vars[cur.ch-'a']; next_token(); return v; }
43 if (cur.type == T_LPAR) {
44 next_token();
45 double v = parse_expr();
46 next_token(); // skip )
47 return v;
48 }
49 return 0;
50}
51
52double parse_term() {
53 double v = parse_factor();
54 while (cur.type == T_OP && (cur.ch=='*'||cur.ch=='/')) {
55 char op = cur.ch; next_token();
56 double r = parse_factor();
57 v = op=='*' ? v*r : v/r;
58 }
59 return v;
60}

65 char op = cur.ch; next_token();

[심화 내용] 재귀 하강 파서
수식 문법 (BNF):
expr → term ((+|-) term)*
term → factor ((*|/) factor)*
factor → NUMBER | '(' expr ')'

우선순위가 자연스럽게 표현됨!
3+4*2 → expr→term(3) + term(4*2=8) = 11
괄호 → factor에서 다시 expr 호출 (재귀!)

컴파일러 구조:
1. Lexer: 문자열 → 토큰
2. Parser: 토큰 → AST (여기까지 구현!)
3. Optimizer: AST 변환
4. Code Gen: AST → 기계어/바이트코드

실전:
GCC, Clang, V8 모두 재귀 하강 파서 사용
→ C 파서를 이해하면 컴파일러를 이해한다!
recursive descent
재귀 하강 파서. 문법 규칙 1개 = 함수 1개. 읽기 쉽고 디버깅 쉬움! GCC, Clang이 사용.
operator precedence
연산자 우선순위. */>+- 를 문법 구조로 표현. term(*/안에서 factor) → expr(+-안에서 term). 자연스러운 우선순위!
재귀 하강: expr→term→factor→expr(괄호). 문법 규칙 = 함수! 컴파일러의 핵심 기법!
3+4*2=11! term이 *를 먼저 처리하니까 4*2=8, 그 다음 3+8=11. 우선순위가 문법에 내장!
