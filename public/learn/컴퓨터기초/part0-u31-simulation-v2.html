1#include <stdio.h>
2
3#define N 5
4
5int main(void) {
6 int grid[N][N] = {0};

11

17 dir = (dir + 1) % 4;
18 x += dx[dir]; y += dy[dir];
19 }
20
21 for (int i = 0; i < N; i++) {
22 for (int j = 0; j < N; j++)
23 printf("%3d", grid[i][j]);
24 printf("\n");
25 }

27}

[심화 내용] 몬테카를로 시뮬레이션

[심화 내용] 몬테카를로 방법 랜덤을 이용해 근사값을 구하는 시뮬레이션!

예: 원주율(π) 구하기
1. 1×1 정사각형 안에 랜덤 점 N개 찍기
2. 원 안에 들어간 점 개수 = M
3. π ≈ 4 × M / N

N이 클수록 정확한 값에 가까워져요!

활용: 주식 예측, 물리 시뮬레이션, AI 게임(알파고)
Cmonte_carlo_pi.c
1#include <stdio.h>
2#include <stdlib.h>
3#include <time.h>
4
5int main(void) {
6 srand(time(NULL));
7 int N = 1000000; // 100만 번 시도
8 int inside = 0;
9
10 for (int i = 0; i < N; i++) {
11 double x = (double)rand() / RAND_MAX; // 0~1
12 double y = (double)rand() / RAND_MAX;
13 if (x*x + y*y <= 1.0) // 원 안?
14 inside++;
15 }
16
17 double pi = 4.0 * inside / N;
18 printf("π ≈ %.6f\n", pi); // 약 3.141592
19 return 0;
20}
