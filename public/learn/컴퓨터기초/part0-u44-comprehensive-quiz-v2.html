17 int temp[100];

26 int j = 0;
27 for (int i = 1; i < n; i++)
28 if (arr[i] != arr[j]) arr[++j] = arr[i];

30}

3// 버블 정렬
4void bubble_sort(int arr[], int n) {
5 for (int i = 0; i < n-1; i++)
6 for (int j = 0; j < n-1-i; j++)
7 if (arr[j] > arr[j+1]) {
8 int t = arr[j]; arr[j] = arr[j+1]; arr[j+1] = t;
9 }
10}
11
12// 선택 정렬
13void selection_sort(int arr[], int n) {
14 for (int i = 0; i < n-1; i++) {
15 int min_idx = i;
16 for (int j = i+1; j < n; j++)
17 if (arr[j] < arr[min_idx]) min_idx = j;
18 int t = arr[i]; arr[i] = arr[min_idx]; arr[min_idx] = t;
19 }
20}
21
22// 삽입 정렬
23void insertion_sort(int arr[], int n) {
24 for (int i = 1; i < n; i++) {
25 int key = arr[i], j = i-1;
26 while (j >= 0 && arr[j] > key) { arr[j+1] = arr[j]; j--; }
27 arr[j+1] = key;
28 }
29}
30
31void print_arr(int a[], int n) {
32 for(int i=0;i<n;i++) printf("%d ",a[i]);
33 printf("\n");
34}

38 int b[] = {64,25,12,22,11};
39 int c[] = {64,25,12,22,11};

42 printf("버블: "); print_arr(a, 5);
43
44 selection_sort(b, 5);

46

5void caesar_encrypt(char *text, int shift) {
6 for (int i = 0; text[i]; i++) {
7 if (isupper(text[i]))
8 text[i] = (text[i] - 'A' + shift) % 26 + 'A';
9 else if (islower(text[i]))
10 text[i] = (text[i] - 'a' + shift) % 26 + 'a';
11 }
12}
13
14void caesar_decrypt(char *text, int shift) {
15 caesar_encrypt(text, 26 - shift);
16}
